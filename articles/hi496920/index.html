                {
                    DeviceDTO deviceDTO = new DeviceDTO()

                        Id = device.Id,
                        Status = device.Status,
                        Name = device.Name
                    };
                    deviceDTOs.Add(deviceDTO);
                }

                DeviceActivator(serialPort, deviceDTOs);

                IFormatProvider formatter = new NumberFormatInfo { NumberDecimalSeparator = "." };
