    val depthMat = Mat(depthFrame.height, depthFrame.width, CvType.CV_16UC1)
    val size = (depthMat.total() * depthMat.elemSize()).toInt()
    val returnBuff = ByteArray(size)
    depthFrame.getData(returnBuff)

    depthMat.put(0, 0, shorts)
    return depthMat
}
