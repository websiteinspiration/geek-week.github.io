word2int_変数に 単語またはタグを追加することで、衝突は必ず解決され、同じインデックスを持つ2つの異なる単語を参照することはありません。何もありません。

int32_t Dictionary::find(const std::string& w, uint32_t h) const {
  int32_t word2intsize = word2int_.size();
  int32_t id = h % word2intsize;
  while (word2int_[id] != -1 && words_[word2int_[id]].word != w) {
    id = (id + 1) % word2intsize;
  }
  return id;
}

int32_t Dictionary::find(const std::string& w) const {
  return find(w, hash(w));
}

void Dictionary::add(const std::string& w) {
  int32_t h = find(w);
  ntokens_++;
  if (word2int_[h] == -1) {
    entry e;
    e.word = w;
    e.count = 1;
    e.type = getType(w);
    words_.push_back(e);
    word2int_[h] = size_++;
  } else {
    words_[word2int_[h]].count++;
  }
}

initNgrams 関数は、N-gram文字のすべての組み合わせを定義し、それらをngramベクトルに追加します。最後に、N-gramのすべてのハッシュが計算され、ngramベクトルに加算されて、バケットのサイズが形成されます。つまり、N-gram文字のハッシュは、N-gramワードのハッシュを追加した後に追加されます。その結果、それらのインデックスは単語のインデックスと重複しませんが、互いに重複する可能性があります。

一般に、テキストの各単語には、サブワード ...文字のNグラムを指定できます。
埋め込み行列Aは以下を表示します。

  • テキストに使用可能な辞書からの各単語の埋め込みを含む最初のnwords_行。
  • 文字のNグラムごとに埋め込みを含む行のバケットを追跡する


void Dictionary::initNgrams() {
  for (size_t i = 0; i < size_; i++) {
    std::string word = BOW + words_[i].word + EOW;
    words_[i].subwords.clear();
    words_[i].subwords.push_back(i);
    if (words_[i].word != EOS) {
      computeSubwords(word, words_[i].subwords);
    }
  }
}

void Dictionary::computeSubwords(const std::string& word,
                               std::vector<int32_t>& ngrams) const {
  for (size_t i = 0; i < word.size(); i++) {
    std::string ngram;
    if ((word[i] & 0xC0) == 0x80) continue;
    for (size_t j = i, n = 1; j < word.size() && n <= args_->maxn; n++) {
      ngram.push_back(word[j++]);
      while (j < word.size() && (word[j] & 0xC0) == 0x80) {
        ngram.push_back(word[j++]);
      }
      if (n >= args_->minn && !(n == 1 && (i == 0 || j == word.size()))) {
        int32_t h = hash(ngram) % args_->bucket;
        pushHash(ngrams, h);
      }
    }
  }
}

void Dictionary::pushHash(std::vector<int32_t>& hashes, int32_t id) const {
  if (pruneidx_size_ == 0 || id < 0) return;
  if (pruneidx_size_ > 0) {
    if (pruneidx_.count(id)) {
      id = pruneidx_.at(id);
    } else {
      return;
    }
  }
  hashes.push_back(nwords_ + id);
}

その結果、モデルのトレーニングを開始する部分が得られます。


void FastText::trainThread(int32_t threadId) {
  std::ifstream ifs(args_->input);
  utils::seek(ifs, threadId * utils::size(ifs) / args_->thread);

  Model model(input_, output_, args_, threadId);
  if (args_->model == model_name::sup) {
    model.setTargetCounts(dict_->getCounts(entry_type::label));
  } else {
    model.setTargetCounts(dict_->getCounts(entry_type::word));
  }

  const int64_t ntokens = dict_->ntokens();
  int64_t localTokenCount = 0;
  std::vector<int32_t> line, labels;
  while (tokenCount_ < args_->epoch * ntokens) {
    real progress = real(tokenCount_) / (args_->epoch * ntokens);
    real lr = args_->lr * (1.0 - progress);
    if (args_->model == model_name::sup) {
      localTokenCount += dict_->getLine(ifs, line, labels);
      supervised(model, lr, line, labels);
    } else if (args_->model == model_name::cbow) {
      localTokenCount += dict_->getLine(ifs, line, model.rng);
      cbow(model, lr, line);
    } else if (args_->model == model_name::sg) {
      localTokenCount += dict_->getLine(ifs, line, model.rng);
      skipgram(model, lr, line);
    }
    if (localTokenCount > args_->lrUpdateRate) {
      tokenCount_ += localTokenCount;
      localTokenCount = 0;
      if (threadId == 0 && args_->verbose > 1)
        loss_ = model.getLoss();
    }
  }
  if (threadId == 0)
    loss_ = model.getLoss();
  ifs.close();
}

上記の関数では、入力データからテキストを読み取り、word2int_を使用して、各単語のインデックスを次々に決定します。コードに示されているように、これらの単語を構成するN-gramをwords変数に追加します。そして最後に、ラベルをラベルベクトルに直接追加します。

テキスト（コーパスからの文）を完全に読み取り、作成されたベクターに直接追加すると、N-gramを処理するコードの一部が得られます。これはaddWordNgrams関数です。

void Dictionary::addWordNgrams(std::vector<int32_t>& line,
                               const std::vector<int32_t>& hashes,
                               int32_t n) const {
  for (int32_t i = 0; i < hashes.size(); i++) {
    uint64_t h = hashes[i];
    for (int32_t j = i + 1; j < hashes.size() && j < i + n; j++) {
      h = h * 116049371 + hashes[j];
      pushHash(line, h % args_->bucket);
    }
  }
}

さらに調べます。ハッシュ変数は、テキスト内の各単語のハッシュのセットです。行には、文内の単語の数と使用されたN-gramの数が含まれます。パラメータnはパラメータwordNgramsであり、単語のN-gramの最大長を示します。単語のNグラムはそれぞれ、再帰的な式によって計算された独自のハッシュを取得します

h=h∗116049371+hashes[j]

。この数式は、文字列に適用されるFNVハッシュアルゴリズムです。Nグラムの単語の各単語のハッシュを取得し、それらを追加します。したがって、一意のハッシュのセットが取得されます。その結果、この値（かなり大きくなる場合があります）はモジュロで送信されます。

したがって、単語のN-gramは文字のN-gramとほぼ同じ方法で計算されますが、わずかな違いがあります-特定の単語をハッシュしません。驚くべき動き。

文を読んだ後、監視あり関数が呼び出されます。オファーに複数のタグがある場合は、それらの1つをランダムに選択します。


void FastText::supervised(
    Model& model,
    real lr,
    const std::vector<int32_t>& line,
    const std::vector<int32_t>& labels) {
  if (labels.size() == 0 || line.size() == 0) return;
  std::uniform_int_distribution<> uniform(0, labels.size() - 1);
  int32_t i = uniform(model.rng);
  model.update(line, labels[i], lr);
}

そして最後に、モデル更新機能。

void Model::computeHidden(const std::vector<int32_t>& input, Vector& hidden) const {
  assert(hidden.size() == hsz_);
  hidden.zero();
  for (auto it = input.cbegin(); it != input.cend(); ++it) {
    if(quant_) {
      hidden.addRow(*qwi_, *it);
    } else {
      hidden.addRow(*wi_, *it);
    }
  }
  hidden.mul(1.0 / input.size());
}

void Model::update(const std::vector<int32_t>& input, int32_t target, real lr) {
  assert(target >= 0);
  assert(target < osz_);
  if (input.size() == 0) return;
  computeHidden(input, hidden_);
  if (args_->loss == loss_name::ns) {
    loss_ += negativeSampling(target, lr);
  } else if (args_->loss == loss_name::hs) {
    loss_ += hierarchicalSoftmax(target, lr);
  } else {
    loss_ += softmax(target, lr);
  }
  nexamples_ += 1;

  if (args_->model == model_name::sup) {
    grad_.mul(1.0 / input.size());
  }
  for (auto it = input.cbegin(); it != input.cend(); ++it) {
    wi_->addRow(grad_, *it, 1.0);
  }
}

監視あり 関数を通過する入力変数には、文のすべてのコンポーネント（単語、単語のNグラムと記号）のインデックスのリストがあります。目標は、出力にクラスを定義することです。computeHidden関数は、wi_マトリックスでコンポーネントを検索して平均化することにより、入力の各コンポーネントのすべての埋め込みを決定します（addRowが合計され、サイズで除算されます）。修正後hidden_ベクターを、介して、活性化のためにそれらを送るソフトマックスとラベルを定義します。

コードのこのセクションでは、softmaxアクティベーション関数の使用法を示します。対数損失も計算されます。

void Model::computeOutputSoftmax(Vector& hidden, Vector& output) const {
  if (quant_ && args_->qout) {
    output.mul(*qwo_, hidden);
  } else {
    output.mul(*wo_, hidden);
  }
  real max = output[0], z = 0.0;
  for (int32_t i = 0; i < osz_; i++) {
    max = std::max(output[i], max);
  }
  for (int32_t i = 0; i < osz_; i++) {
    output[i] = exp(output[i] - max);
    z += output[i];
  }
  for (int32_t i = 0; i < osz_; i++) {
    output[i] /= z;
  }
}

void Model::computeOutputSoftmax() {
  computeOutputSoftmax(hidden_, output_);
}

real Model::softmax(int32_t target, real lr) {
  grad_.zero();
  computeOutputSoftmax();
  for (int32_t i = 0; i < osz_; i++) {
    real label = (i == target) ? 1.0 : 0.0;
    real alpha = lr * (label - output_[i]);
    grad_.addRow(*wo_, i, alpha);
    wo_->addRow(hidden_, i, alpha);
  }
  return -log(output_[target]);
}

この方法を使用すると、Nグラムの単語や文字のサイズが大きくなることはありません。成長は、すでに利用可能なバケットの数によって制限されます。
