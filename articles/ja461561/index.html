Arduinoタイマー

こんにちは。私の名前はNikita Sergeevichです。14歳です。この投稿では、完成したデバイスがはんだ付けされ、父親によって組み立てられたので、半年前に開発したデバイス（およびアプリケーション）についてお話ししたいと思います。あなたはそのような装置を一度以上見たかもしれませんが、それは私を気にしませんが。

これらはすべて、鶏小屋の照明を自動的にオン/オフするために作成されましたが、他の目的にも適用できます。





完成して組み立てられたデバイスは、指定された時間から指定された時間までの正確な分まで、指定されたデジタルピンarduino 1または0に送信できます（期間外では、タイマーを追加するときにアプリケーションで指定された値と反対の値が提供されます）。

すべてのパラメータはアプリケーションに示されています。そして、ここからさらに詳しく...

Bluetoothモジュールに接続してタイマーを追加する必要があるAndroid用のアプリケーションがあります。アプリケーションは直感的で問題はないと思います。





.apkファイルは、このプロジェクトのフォルダーにあります。

必要なアイテム：

  1. arduino;
  2. rtcモジュールds1307;
  3. BluetoothモジュールHC-05;
  4. 接続要素（例：ワイヤー）;
  5. arduinoのファームウェア。私の意見ではスケッチは複雑なので、詳細には触れない方が良いでしょう。
  6. Android用アプリケーション。

5.スタジオにスケッチします。

#include <Wire.h>                                            //     i2c
#include <iarduino_RTC.h>                                    //  rtc ( c,    (Timer->firmware_for_timer->iarduino_RTC))     
#include <EEPROM.h>                                          //     

iarduino_RTC time(RTC_DS1307);                               //       

int values;                                                  //   

unsigned int to_minutes(unsigned int hours, unsigned int minutes)//       
{
  return (60*hours)+minutes;
}

void setup()
{
  time.begin();                                             //  
  Serial.begin(9600);                                       //  
 
}

void loop()
{
  if(Serial.available() > 0)                                // - 
  {
    values = Serial.read();
    time.settime(0,-1, values);                             //   
    delay(50);
    values = Serial.read();
    time.settime(0, values);                                //   
    delay(50);

    //  
    for (int i = 0 ; i < EEPROM.length() ; i++)
    {
      EEPROM.write(i, 0);
    }
    //////////////////////////////////
    
    values = Serial.read();                                 
    delay(50);
    
    //    ,   
    while (values > 0)                                        //   
    {
      for(int i = 0; i < 7; ++i)
      {
        EEPROM.write(EEPROM.read(EEPROM.length() - 1) * 7 + i, Serial.read());
        delay(50);
      }

      EEPROM.write(EEPROM.length() - 1, EEPROM.read(EEPROM.length() - 1) + 1);

      values = Serial.read();
      delay(50);
    }
    /////////////////////////////////////////
  }

  //   
  for(int i = 0; i < EEPROM.read(EEPROM.length() - 1); ++i)
  {
    if(EEPROM.read(i * 7 + 0) == 1)                                                                               //   
    {
      bool find = 0;                                                                                              //
      time.gettime();
      if((to_minutes(EEPROM.read(i * 7 + 1), EEPROM.read(i * 7 + 2)) <= to_minutes(time.Hours, time.minutes)) && //    
         (to_minutes(EEPROM.read(i * 7 + 3), EEPROM.read(i * 7 + 4)) > to_minutes(time.Hours, time.minutes)))
      {
        pinMode(EEPROM.read(i * 7 + 5), OUTPUT);                                                                  //    
        digitalWrite(EEPROM.read(i * 7 + 5), EEPROM.read(i * 7 + 6));                                             //    
      }                           
      else                                                                                                        //    
      {
        //     
        for (int j = 0; j < EEPROM.read(EEPROM.length() - 1); ++j)
        {
          if(EEPROM.read(i * 7 + 5) == EEPROM.read(j * 7 + 5) && (i != j) && (EEPROM.read(j * 7 + 0) == 1))
          {
            //             
            if((to_minutes(EEPROM.read(j * 7 + 1), EEPROM.read(j * 7 + 2)) <= to_minutes(time.Hours, time.minutes)) &&
               (to_minutes(EEPROM.read(j * 7 + 3), EEPROM.read(j * 7 + 4)) > to_minutes(time.Hours, time.minutes))) //    
            {
              pinMode(EEPROM.read(i * 7 + 5), OUTPUT);                                                             //    
              digitalWrite(EEPROM.read(i * 7 + 5), EEPROM.read(j * 7 + 6));                                        //    
              find = 1;                                                                                            // 
            }
          }
        }

        if(find == 0)                                                                                              //       
        {
          pinMode(EEPROM.read(i * 7 + 5), OUTPUT);                                                                 //    
          digitalWrite(EEPROM.read(i * 7 + 5), !EEPROM.read(i * 7 + 6));                                           //     
        }
      }
    }
  }
}

しかし、すぐに理解することをお勧めします。アプリケーションを初めて入力するときは、Bluetoothモジュールに接続する必要があり
ます。これを行うには、[デバイスの同期]をクリックして、モジュールを選択します。「自動的に接続する」チェックボックスをオンにすると、アプリケーションを開くと、携帯電話自体がBluetoothモジュールに接続しようとします（このチップはalex gyverから取得したものです）。タイマーの名前をクリックすると、このタイマーを変更するための画面が開きます。ここで（タイマーのリストはここにあります）右下隅にある[+]をクリックすると、タイマーの追加画面が表示されます。

  • 彼の名前、
  • 何時から（時分）
  • 何時（時分）
  • 制御ピン
  • 彼への価値
  • および状態（適切なタイミングでarduinoがオンまたはオフになります）。
