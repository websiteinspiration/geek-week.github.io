次にshowResult、ユーザーがゲーム全体を通過できた後、結果を描画するために必要な関数を別のファイルに入れることにしました。これは非常に普遍的な関数で、最初の引数は情報が書き込まれlocalStorage受信される入力に、2番目の引数は書き込まれる情報です（この場合、これはプレーヤーのエラーの数です）。さらに、私はからのキーのソートを実装しlocalStorageました。これにより、テーブル内のデータは、作成された時間でソートされました。関数の2番目の部分の表には、プレーヤーの以前の試行のデータが表示されます。ただし、単語が少なく、コードが多い：

export default function showResult(target_El, content){
    localStorage.setItem(+new Date, content);   
    (function drawOnLoad() {
        let temp_arr = [];
        for (let i = 0; i < localStorage.length; i++) { 
            temp_arr.push(+localStorage.key(i));
        }
        temp_arr.sort();
        for(let i = 0; i< temp_arr.length; i++){
            let item_time = new Date(temp_arr[i]);
            target_El.insertAdjacentHTML('afterend', 
            `<th>${item_time.getDate()} / ${item_time.getMonth()}  ${item_time.getHours()} : ${item_time.getMinutes()} </th>
            <th> ${localStorage.getItem(String(temp_arr[i]))}</th>
            `);
        }
    })();
}

  var errors_count = 0;
  var count_right = 0;

  function press(e) {
    let elements_arr = document.querySelectorAll(".game-button"); //     
    if (e.key == elements_arr[0].id) {
      //      querySelector,     
      elements_arr[0].remove();
      count_right++; //
