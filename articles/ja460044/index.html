Arduinoのタイマー付きオーブンヒーター温度制御

ポリマークレイを焼くための炉が必要でした。それほど長くは検索されなかったが、選択はキッチン用のKEDR電気オーブンに落ちた。 600ワットの電力、最高温度250度、レギュレータなし。動作温度が100〜130度の範囲で必要とされたため、初めてサーモメカニカルレギュレーターが設置されました。しかし、全体の問題は、炉の加速度が非常に大きく（ヒーターをオフにした後、温度がさらに20〜50度上昇し続ける）、レギュレーターのオンとオフの範囲が非常に大きいことでした。つまり、温度を130度に設定すると、許容範囲ではない100〜160度の範囲になります。

Arduino IDEとC ++での作業の原則を分析して数か月後、要件を完全に満たすプロジェクトが生まれました。デバイスは、アラームがトリガーされた後、設定に応じて、タイマーが5〜35分間に設定されると、設定温度を100〜150度に維持できます。

これはプログラムのブロック図です。



デバイスには、デフォルトで炉に取り付けられた4つのシリアルヒーターが含まれます。これは、対応する電源の他のヒーター、SSR-25 DAソリッドステートリレー、Arduino Pro Mini、逆シュミットトリガーを介して接続されたボタン付きインクリメンタルエンコーダー、ディスプレイWH1602Dおよび2つのNTCに置き換えることができます。 100kオームのサーミスタMF58。

これがコメント付きのスケッチコードです。

int *pMinute = &setTimeMinute; // ,     ...?

void clkEnc(){ //    ,   

pinClkValue = digitalRead(encCLK); 
pinDtValue = digitalRead(encDT);
cli(); 
if (!pinClkValue && pinDtValue) state = 1; 
if (!pinClkValue && !pinDtValue) state = -1; 
if (pinClkValue && state != 0) {
if (state == 1 && !pinDtValue || state == -1 && pinDtValue) { 
count += state*5; 
state = 0; }}
sei(); }

count = 0;                             //  
button_not_press = true;               //   ,     while,

if (count <= 100) count = 100;           //      100

else if(*pSetTemp<100) lcd.print(" ");
lcd.print(*pSetTemp);

lcd.print("C");

lcd.print(*pMinute);             //

if (*pMinute != 0){second = 0; timerStart();} //   ,    
timeEnd(); } //   , !!!

void tempMonitor(){                  //   !!! 
//          .
if(millis() > normalModeTime + 10){ heater_state = !heater_state; normalModeTime = millis();} 
//         
if(millis() > currentTime + 500){
lcd.setCursor(4, 1);
if(*pCurTemp<10) lcd.print("  ");
else if(*pCurTemp<100) lcd.print(" ");
lcd.print(*pCurTemp);
lcd.print("\1");
lcd.print("C");
lcd.print(" ");  

int US = analogRead(temp_upside_pin);     //   1 
int DS = analogRead(temp_downside_pin);   //   2 
int MID = (US+DS)/2;                      //   
float tr = (5 / 1023.0) * MID ;           //   
float VR = 5 - tr;                        //   
float RT = tr/(VR / 99000);               //   
float ln = log(RT / 100000);               
float TX = (1 / ((ln / B) + (1 / (25+273.15)))); //    —  
TX = round(TX - 273.15);                  //       
*pCurTemp = int(TX);                      //   
currentTime = millis(); }


//      ( - hyst)
if(preheating_state){ if(*pCurTemp < (*pSetTemp - hyst)) digitalWrite(heater_pin, HIGH); 
//     ( - hyst), 
//         (pSetTemp). 
else if(*pCurTemp > (*pSetTemp - hyst)) digitalWrite(heater_pin, LOW); 
if (*pCurTemp >= *pSetTemp){ preheating_state = false; Serial.print(preheating_state); }

//            
//   ,      
//      ,    
// .    ,
//     . 
if (millis() > timeHyst + 30000){  
changeTemp = *pCurTemp;           //   30 ,    
timeHyst = millis(); }
if(changeTemp >= *pCurTemp){ 
if(!digitalRead(heater_pin)){
digitalWrite(heater_pin, HIGH);}}}

//   (pSetTemp) 
//          
//    5    
if(!preheating_state){             
if(*pCurTemp < *pSetTemp - 5) {    
digitalWrite(heater_pin, heater_state);}
else digitalWrite(heater_pin, LOW); }}
