線形アドレッシング


Intelプロセッサには、保護モードx32とIA-32e x64の2つの主要な動作モードがあります。
一般に、Zubkovはこれについて非常によく理解できるように書いています。原則としてIntel Manualも可能ですが、複雑ではなく冗長で大きなものですが、読むことをお勧めします。
彼らはシステムプログラミングのために別のボリュームを持っています、私はそれを読んで読んでください。
最初はロシア語の情報がはるかに多いため、要点を簡単に検討します。
アドレス指定には、リニアとページの2つのタイプがあります。線形とは、全体として物理空間全体が連続的に記述され、物理空間と一致することを意味します。これは、原則として、セグメント記述子のベースがゼロに等しいためです。
この場合、カーネルモードでは、メモリを記述する3つの記述子（コード、スタック、およびデータ）を作成する必要があります。それらはいくつかのハードウェア保護によって区別されます。
このような各セグメントには、ゼロのベースと、マシンワードの最大サイズによって対処される制限があります。スタックは反対方向に大きくなり、そのために記述子にもフラグがあります。
したがって、このフォーマットの3つのレコードを使用して、必要なすべてのものに対処します。

/*
 * Global descriptor table entry
 */
struct GDT_entry_t {
    u16 limit_low: 16;
    u16 base_low: 16;
    u8 base_middle: 8;
    u8 type: 4; /* whether code (0b1010), data (0b0010), stack (0b0110) or tss (0b1001) */
    u8 s: 1; /* whether system descriptor */
    u8 dpl: 2; /* privilege level */
    u8 p: 1; /* whether segment prensent */
    u8 limit_high: 4;
    u8 a: 1; /* reserved for operation system */
    u8 zero: 1; /* zero */
    u8 db: 1; /* whether 16 or 32 segment */
    u8 g: 1; /* granularity */
    u8 base_high: 8;
} attribute(packed);


各セグメントレジスタ（cs、ds、ss）は、GDTに独自の記述子を持っているため、コードセクションに何かを書き込むと、記述子に保護が書き込まれているため、エラーが発生します。
これを機能させるには、次の形式の構造をGDTRレジスタにロードする必要があります。

/*
 * Global descriptor table pointer
 */
struct GDT_pointer_t {
    u16 limit;
    u32 base;
} attribute(packed);


制限は、GDTテーブルの最後から1を引いた値です。ベースは、メモリ内での最初です。
GDTは次のようにレジスターに読み込まれ ます。その後すぐに、データ記述子（ゼロ保護リング）を示すすべてのセグメントレジスターにカーネルデータセレクターを読み込みます。 その後、すべてにページングを含める準備が整いますが、それについては後で詳しく説明します。 ちなみに、マルチブートローダーは、GDTをすぐにセットアップすることをお勧めします。 ビデオチュートリアルで、これらすべてを技術的に正しく行う方法を参照してください。

/*
* Load global descriptor table
* void asm_gdt_load(void *gdt_ptr)
*/
asm_gdt_load:
mov 4(%esp),%eax # eax = gdt_ptr
lgdt (%eax)
mov $0x10,%eax
mov %ax,%ds
mov %ax,%es
mov %ax,%fs
mov %ax,%gs
mov %ax,%ss
jmp $0x08,$asm_gdt_load_exit
asm_gdt_load_exit:
ret







割り込み処理


GDTと同様に、割り込みテーブルには独自のIDTRレジスタがあり、そのIDTRレジスタにも同様のポインタをロードする必要がありますが、すでにIDT上にあります。
割り込みテーブル自体は、次のエントリで説明されています。

/*
 * Interrupt table entry
 */
struct IDT_entry_t {
    u16 offset_lowerbits;
    u16 selector;
    u8 zero;
    u8 type_attr;
    u16 offset_higherbits;
};


割り込みを正確に処理する必要があるため、通常、割り込みゲートウェイはタイプとして機能します。トラップとコールゲートウェイはTSSおよび保護リングに近いため、まだ検討していません。
これらのテーブルを使用するためのインターフェースを作成しましょう。設定して、一度忘れるだけです。

/*
 * Api
 */
extern void gdt_init();
extern void idt_init();


次に、IDTレコード自体にリストされている割り込みハンドラーを宣言します。
まず、ハードウェアエラーハンドラを記述します。

/*
 * Api - IDT
 */
extern void ih_double_fault();
extern void ih_general_protect();
extern void ih_page_fault();
extern void ih_alignment_check();
extern void asm_ih_double_fault();
extern void asm_ih_general_protect();
extern void asm_ih_page_fault();
extern void asm_ih_alignment_check();


次に、キーボード割り込みハンドラ：

/*
 * Api - IRQ
 */
extern void ih_keyboard();
extern void asm_ih_keyboard();


IDTテーブルを初期化する時が来ました。
次のようになります。

extern void idt_init()
{
    size_t idt_address;
    size_t idt_ptr[2];

    pic_init();

    /* fill idt */
    idt_fill_entry(INT_DOUBLE_FAULT, (size_t)asm_ih_double_fault);
    idt_fill_entry(INT_GENERAL_PROTECT, (size_t)asm_ih_general_protect);
    idt_fill_entry(INT_ALIGNMENT_CHECK, (size_t)asm_ih_alignment_check);
    idt_fill_entry(INT_KEYBOARD, (size_t)asm_ih_keyboard);

    /* load idt */
    idt_address = (size_t)IDT;
    idt_ptr[0] = (LOW_WORD(idt_address) << 16) + (sizeof(struct IDT_entry_t) * IDT_SIZE);
    idt_ptr[1] = idt_address >> 16;
    asm_idt_load(idt_ptr);
}


ここでは、3つのハードウェアエラーハンドラと1つの割り込みを登録しました。
これを機能させるには、ベースと制限を含む特別なインデックスをIDTRレジスタにロードする 必要があります。テーブル内のレコード数を理解するには制限が必要です。 キーボード割り込みハンドラを作成する時が来ました： 注：以下のコードのどこでも、「下半分」はLinuxの「上半分」と同等です。そして「アッパー」は、それぞれ反対です。申し訳ありませんが、それは私の頭の中では逆でした：D 実際には、彼はコードを高レベルのハンドラーに渡します。 次に、この割り込みを処理する要求を登録した、対応するドライバーの下半分のハンドラーを呼び出します。 私たちの場合、それはキャラクターデバイスドライバーになります。

/*
* Load interrupt table
* void asm_idt_load(unsigned long *addr)
*/
asm_idt_load:
push %edx
mov 8(%esp), %edx
lidt (%edx)
pop %edx
ret





/*
* Handle IRQ1
* void asm_ih_keyboard(unsigned int)
*/
asm_ih_keyboard:
pushal
call ih_keyboard
popal
iretl







下半分は、他の割り込みを遅くせずに割り込みをすばやく処理するために必要です。時間がある場合、上半分のプロセッサーは、すでに混雑している（中断されている）可能性があるため、徐々に追加の作業を実行します。

/*
 * Api - Keyboard interrupt handler
 */
extern void ih_keyboard()
{
    printf("[IH]: irq %u\n", 1);

    u_char status = asm_read_port(KEYBOARD_STATUS_PORT);
    if (status & 0x01) {
        char keycode = asm_read_port(KEYBOARD_DATA_PORT);

        if (keycode < 1) {
            goto end;
        }

        /* call low half (bottom) interrupt handler */
    }

end:
    asm_write_port(PIC1_CMD_PORT, 0x20); /* end of interrupt */
}


これで、キーボードのキーを押すと、カーネルシステムログに対応するエントリが表示されます。
