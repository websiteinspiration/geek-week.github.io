注：以下のコードでは、css-modules、sass、classnamesも使​​用しています。

UIButton.tsx

import * as React from 'react';
import { VisualComponentProps } from 'Library/Controls';
import * as css from './Button.sass';
import cn from 'classnames';

//  () 
export interface ButtonBasicProps {
	disabled?: boolean;
}
export interface ButtonProps extends ButtonBasicProps, VisualComponentProps {}

export function UIButton(props: ButtonProps) {
	//   undefined,    
	// "Nothing was returned from render."
	if (props.doNotRender) return props.fallback || null;

	//     
	const rootClassNames = cn(
		// ,   sass
		css.Button,
		// ,     props
		props.className,
		//     
		props.disabled && css._disabled
	);

	return (
		<div
			children={props.children}
			className={rootClassNames}
			style={props.style}
		/>
	)
}

App.tsx

renderComponent() {
	const {props, state} = this;
	const needRenderSecond = true;

	return (
		<PageLayout>
			<UIButton
				children={'This is a button'}
				style={{marginRight: needRenderSecond ? 5 : null}}
			/>
			<UIButton
				disabled
				children={'This is another button'}
				doNotRender={!needRenderSecond}
			/>
		</PageLayout>
	);
}

考察と結論


divなどのコンポーネントを操作して、さまざまなラッパー、コンポジション、特殊化を作成し、それらに埋め込まれた元の機能のフレームワークを超えると便利です。

設計システムには条件付きの黄色いボタンがなく、開発者がそれらを作成する必要があるため、問題はコンポーネントではなく、このニーズが生み出すという事実にあると主張できます。しかし、現実にはそのような状況が発生し、かなり頻繁に発生します。 「……しかし生きなければならない！生きなければならない」さらに、CSSカスケードの原則は常に実際に実装できるとは限りません。また、スタイルが別のセレクターの高い特異性（または上記）と単純に重複する場合があります。ここでスタイルは役に立ちます。
