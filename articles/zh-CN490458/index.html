    std::vector<int> v0;

    for (size_t i = 0; i < m; ++i) {

        v1[j + 1] = std::min(substCost, std::min(v0[j + 1], v1[j]) + 1);
      }
      std::swap(v0, v1);
    }
    return v0[n];
  }
