状態の存在：ホバー、：アクティブ、および：focu *。非常に重要なポイントです。インターフェースとの相互作用の間、快適さは彼にかかっていました。ユーザーは常に自分のアクションに関するフィードバックを受け取る必要があります。この項目は、アンケートとのやり取りを通じてチェックされました。 「電話をかける」ボタンをクリックして視覚的に何も起こらなかった場合（リクエストが送信された場合でも）、これは悪いことです。それは何度も何度もクリックするからです。その結果、10件のリクエストが送信され、10回コールバックされます。モバイルデバイスにマウスがないことを忘れないでください。これは、ホバーがないことを意味します。そして、意味論について要点を満たした人々に関係のないもう一つのこと。コントロールがインタラクティブな要素でない場合、コントロールにカーソルを合わせると、カーソルは標準のままになります。ホバーに対する応答を指定した場合でも、それは非常に整然として見えます。カーソルを過小評価しないでください：ポインター。

正しい解決策：

const solution = function(graph, START, FINISH)  {
    //      
    const costs = Object.assign({[FINISH]: Infinity}, graph[START]);

    //    
    const parents = { [FINISH]: null };
    Object.keys(graph[START]).reduce((acc, child) => (acc[child] = START) && acc, parents)

    const visited = [];
    let node;

    //  «» ,  
    do {
        node = lowestCostNode(costs, visited);
        let children = graph[node];
        for (let n in children) {
            let newCost = costs[node] + children[n];

            //        
            if (!costs[n] || costs[n] > newCost) {
                costs[n] = newCost;
                parents[n] = node;
            }
        }
        visited.push(node);
    } while (node)

    return {
        distance: costs[FINISH],
        path: optimalPath(parents)
    };

    //     «» 
    function optimalPath(parents) {
        let optimalPath = [FINISH];
        let parent = parents[FINISH];
        while (parent && parent !== START) {
            optimalPath.push(parent);
            parent = parents[parent];
        }
        optimalPath.push(START);
        return optimalPath.reverse();
    }

    //       
    function lowestCostNode(costs, visited) {
        return Object.keys(costs).reduce((lowest, node) => {
            if (lowest === null || costs[node] < costs[lowest]) {
                if (!visited.includes(node)) {
                    lowest = node;
                }
            }

            return lowest;
        }, null);
    };
};

作り上げたカレンダー。はい、彼はまだ補われる必要がありました。文字どおりに条件を理解しすぎて、CSSコードを1行も挿入しない人もいました。個人的には見えませんでしたが、すべてうまくいったとしてもポイントは減りませんでした。

ソースからイベントのリストを取得します。これは組版のタスクではないので、それに組み込まれたイベントのリストはカウントされませんでした。いつでも会議をキャンセルしたり、再スケジュールしたり、新しい会議を追加したりできます。そのため、外部からデータを受け取り、受け取ったJSONに基づいてレイアウトをレンダリングする必要がありました。データを取得することは、（fetchメソッドを使用するかXMLHttpRequestを使用して）何らかの方法で重要でした。人がフェッチのためにポリフィルを追加し、readmeで彼の選択をマークした場合、これはプラスとしてカウントされます。
