＃02-そして、バイト全体では十分ではありません... | 変化の十字架

名誉と部分的に敬意を表する。このような何かで、魅惑的なデモシーンの世界に浸ります。今日は、サイズコーディングの分野における特定の作業について説明します。実際のところ、一部のリリースは狭い範囲でカルトステータスを示しただけでなく、人々の心に直接かつ明確に影響を与え、IDA Proを学び、コードを見て、細かいところまで浸透するように強いられていました。そのような魔法がどのように機能するかは、単に理解できず非常に興味深いものでした。

これはQueue Members Groupによるクロスです -遠く離れた1996年からのPC向け128バイトのイントロ ：これが作品のビデオです：







一言で言えば、祖父はどのように戦ったのか... FIDO会議DEMO.DESIGNでのEnlight'95の成功の後、PC向けの128バイトの作品を作成するための仮想コンテストが発表されました。彼は数か月間歩き、主催者は結果を収集し、投票し、勝者を決定しました。Demo design'96のリリースアーカイブ全体は、Pouetで確認できます。勝者はマッドマックス/キューメンバーグループからのクロスでした。QMGグループはサマラ出身で、当時の男たちは1つの工科大学の学生で、マッドマックス自身がチーム全体のリーダーでした。

コンテストは全国的に若者の心を「揺さぶる」ため、Enlight'96の主催者は背中にクロスダンププリントアウトを備えたTシャツのプリントをプリントしました。ここにいくつかのアーカイブ写真があります：







そして、Enlight'96に関するビデオからのショット：



2020年3月に、グループ= R MDA =（私はここに正式に代表します）は、この歴史的作品のデジタル保存のためのプロジェクトを正常に実装しました。私たちはビデオを作成することから始めました...はい、ありふれたビデオで、その後YouTubeにアップロードされました。それまでの25年間、ビデオバージョンのcrossは存在しませんでした。それを視聴するには、強力なDosBoxカンフーが必要でしたが、誰もが持っているわけではありませんでした。

次に、ソースコードのクロスを含むすべての追加資料を1つのアーカイブに収集しました。残念ながら、著者はその時点でソースコードを公開していませんでした。さらに、彼の言葉から、作品への最新の変更はHEXエディターでまったく行われていませんでした。したがって、以下のソースコードは、元のコードとは逆のコメントです。

; NASM or FASM

        org 100h

        mov bx,es
        add bh,10h              ; bx = cs + 1000h
        mov es,bx               ; es = segment behind 64k of our code (extra video buffer)
        mov ds,bx               ; ds = es

        xor ax,ax
        mov cx,ax
        dec cx                  ; cx = -1
        rep stosb               ; clear extra video buffer
        dec ax                  ; ax = -1
        mov di,0A2D0h           ; set pointer to pixel at 130,80 (col,row)
        mov cl,80
        rep stosw               ; fill 80*2 = 160 pixels by 0FFh color (draw horizontal line)

        mov bx,19A0h            ; set pointer to pixel at 20,160 (col,row)
v:      mov [bx],ax             ; draw vertical line...
        add bx,320
        ja v                    ; ...until bottom

        mov ax,13h
        int 10h                 ; set 320x200 (256 colors) graphical video mode

        mov dx,3C8h
        xor al,al
        out dx,al               ; start palette set from 0 color
        inc dx
        mov cx,0C0h*3
        rep outsb               ; set first 0C0h (192) colors to black
p:
        out dx,al               ; red component
        outsb                   ; green = 0
        outsb                   ; blue = 0
        inc ax                  ; increase color (1, 2, 3..63)
        cmp al,64
        jb p                    ; set colors 0C0h..0FFh to red gradient (from dark to saturated)

        mov ch,0A0h
        mov ds,cx               ; ds = 0A000h = vedeo segment

l:      ; bx = pixel pointer, dx = pixel pointer delta value (initial values doesn't matter)
        cmp bh,0FAh
        jae s                   ; jump if current pixel is out of screen
        mov al,[bx]             ; else get color of pixel
        cmp al,0C0h
        jb s                    ; jump if pixel color is out of red gradient range (0C0h..0FFh)

        ; flame effect
        dec ax                  ; else decrase color by 1 (make it darker)
        mov [bx+1],al
        mov [bx-1],al
        mov [bx+320],al         ; di = 320
        mov [bx-320],al         ; and fill pixels around the current by new color

        mov si,-640             ; position delta: 2 lines up
        test dh,80h
        jnz n                   ; jump if dx is negative (poor pseudorandom boolean check)
        add si,2                ; wind effect
n:      mov [bx+si],al          ; add pixel 2 lines higher than current pixel considering possible wind
s:
        ; restore cross
        mov al,[es:bx]          ; get color from extra video buffer
        or [bx],al              ; replace pixel on screen by 0FFh is al = 0FFh (slow cross redraw)
        add bx,dx               ; increase pixel pointer by (pseudorandom) delta value
        inc dx                  ; increase delta value

        or bx,bx
        jnz l                   ; loop until full screen will be processed

        in al,60h               ; read keyboard scan code
        cmp al,1
        jnz l                   ; repeat if Esc key is not pressed

        mov al,3
        int 10h                 ; set text video mode

        ret                     ; return to DOS


コードをソースに戻す過程で、主要なx86プログラマーが最適化のアイデアを得たことは明らかです。これがcross2の結果です。1バイト多く（126ではなく127）かかりますが、最適化されたものはすべてテキストとサウンドに費やされました。これらの作業を視覚的に混乱させないように、色を緑に変更しました。



ここで歴史的遺産をあざけることに目的はありません。サイズコーディングを数バイトにわたって最適化できるだけです。時間はほぼ30年とかなり経過し、最適化は重要であることが判明しました-20バイト（コードのほぼ16％）。私たちの仕事はオリジナルよりも1バイト長くなります（13バイトのテキストがコードに追加され、6バイトの出力コード、2バイトのサウンド= 21-1バイトのサイズの差が差し引かれます）。

実際にはすべて！ソースcross2はPouetのアーカイブにあり、オリジナルと視覚的に比較できます。このプロジェクトの開発中に助言や資料を提供してくれたPeter [frog] SobolevとMike [Mad Max] Shirobokovに感謝します。
