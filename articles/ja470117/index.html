二重列挙可能


Enumerableから始めましょう：

//    ,  .
protocol Enumerable {
    associatedtype Element where Self.Element == Self.Enum.Element
    associatedtype Enum: Enumerator
    
    func getEnumerator() -> Self.Enum
}

protocol DualOfEnumerable {
    //  Enumerator :
    // getEnumerator() -> Self.Enum
    //    :
    // getEnumerator(Void) -> Enumerator
    //
    //  , :
    // : Void; : Enumerator
    // getEnumerator(Void) → Enumerator
    //
    //     Void   Enumerator.
    //   -      Enumerator,   Void.
    // :  Enumerator; : Void
    func subscribe(DualOfEnumerator)
}

彼がgetEnumerator()受け入れVoidて与えたのでEnumerator、今私たちは[ダブル]を受け入れてEnumerator与えVoidます。

これは奇妙なことです。行かないで。

ダブル列挙子


そして、何DualOfEnumerator？

//    ,  .
protocol Enumerator: Disposable {
    associatedtype Element
    
    // : Void; : Bool, Error
    func moveNext() throws -> Bool
    // : Void; : Element
    var current: Element { get }
}

protocol DualOfEnumerator {
    // : Bool, Error; : Void
    //   Error   
    func enumeratorIsDone(Bool)
    // : Element, : Void
    var nextElement: Element { set }
}

ここにはいくつかの問題があります：

  • Swiftには設定のみのプロパティという概念はありません。
  • 何が起こっthrowsたのEnumerator.moveNext()？
  • 何が起こっているのDisposableですか？

設定のみのプロパティの問題を修正するために、それを実際の機能、つまり関数として扱うことができます。少し修正しましょうDualOfEnumerator：

protocol DualOfEnumerator {
    // : Bool; : Void, Error
    //   Error   
    func enumeratorIsDone(Bool)
    // : Element, : Void
    func next(Element)
}

で問題を解決するために、でthrows発生する可能性のあるエラーを分離してみましょう。moveNext()別の関数として処理しますerror()。

protocol DualOfEnumerator {
    // : Bool, Error; : Void
    func enumeratorIsDone(Bool)
    func error(Error)
    
    // : Element, : Void
    func next(Element)
}

別のことができます：反復の完了の署名を見てください：

func enumeratorIsDone(Bool)

おそらく、同様のことが次第に起こります。

enumeratorIsDone(false)
enumeratorIsDone(false)
//    
enumeratorIsDone(true)

そして今、すべてを単純化しましょう。そして、我々はenumeratorIsDone...すべてが本当に準備ができているときだけ呼び出します。このアイデアに基づいて、コードを簡略化します。

protocol DualOfEnumerator {
    func enumeratorIsDone()
    func error(Error)
    
    func next(Element)
}

protocol DualOfEnumerator {

とてもよく理解しやすいです。

タイプ名はどうですか？彼らはただひどいです。それらを少し変更しましょう。

  • DualOfEnumerator-オブジェクトの線形グループに起こることに続くもの。彼は線形グループを観察していると言えます。
  • DualOfEnumerable観察の対象です。私たちが見ているもの。したがって、それは観測可能と呼ぶことができます。

最後の変更を行い、次のようにします。

protocol Observable {
    func subscribe(Observer) → Disposable
}

protocol Observer {
    func onComplete()
    func onError(Error)
    func onNext(Element)
}

それは...単なる非現実的なつまらないものです。Combineを使用してUIControlからUIControl.Eventを

取得する一般的な方法は次のとおりです。

class ControlPublisher<T: UIControl>: Publisher {
    typealias ControlEvent = (control: UIControl, event: UIControl.Event)
    typealias Output = ControlEvent
    typealias Failure = Never
    
    let subject = PassthroughSubject<Output, Failure>()
    
    convenience init(control: UIControl, event: UIControl.Event) {
        self.init(control: control, events: [event])
    }
    
    init(control: UIControl, events: [UIControl.Event]) {
        for event in events {
            control.addTarget(self, action: #selector(controlAction), for: event)
        }
    }
    
    @objc private func controlAction(sender: UIControl, forEvent event: UIControl.Event) {
        subject.send(ControlEvent(control: sender, event: event))
    }
    
    func receive<S>(subscriber: S) where S :
        Subscriber,
        ControlPublisher.Failure == S.Failure,
        ControlPublisher.Output == S.Input {
        
            subject.receive(subscriber: subscriber)
    }
}

これは、オブジェクトが他のオブジェクトの指定されたプロパティの変更の通知を受信できるようにするメカニズムです。
