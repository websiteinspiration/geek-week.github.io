私は長年WPFを使用していました。MVVMパターンは、おそらく最も便利なアーキテクチャパターンの1つです。MVCはほぼ同じだと思いました。新しい職場でMVCを実際に使用しているのを見たとき、その複雑さと、基本的な使いやすさの欠如に驚きました。最も厄介なのは、フォームがオーバーロードされたときにのみ検証が行われることです。エラーが発生しているフィールドを強調表示する赤いフレームはなく、エラーのリストを含むアラートのみが表示されます。エラーが多い場合は、検証を繰り返すために、いくつかのエラーを修正して保存する必要があります。保存ボタンは常にアクティブです。リンクされたリストは実際にはjsを介して実装されますが、それは複雑で混乱を招きます。モデル、ビュー、コントローラーは密結合されているため、すべてをテストしてください雄大非常に難しい。
これに対処する方法?? 面白そうな方にはキャットの下でお願いします。


そして、私たちがしている：
古典的な形でMVCのフォームを構築することは、ユーザーのために便利ではないページ全体をリロードするようサーバとの対話の別の方法を意味するものではありません。

        private static string   GetErrorsAndChanged(IEnumerable<DbEntityValidationResult> errors,bool changed)
        {
            dynamic dynamic = new ExpandoObject();
            dynamic.IsChanged = changed;//  IsChanged
            var errProperty = new Dictionary<string, object>();//     
            dynamic.Errors = new DynObject(errProperty);//       
            foreach (DbEntityValidationResult validationError in errors)//  
            {
                foreach (DbValidationError err in validationError.ValidationErrors)//  
                {
                    errProperty.Add(err.PropertyName,err.ErrorMessage);
                }
            }
            var json = JsonConvert.SerializeObject(dynamic); return json;
        }

    DynObject

 public sealed class DynObject : DynamicObject
    {
        private readonly Dictionary<string, object> _properties;

        public DynObject(Dictionary<string, object> properties)
        {
            _properties = properties;
        }

        public override IEnumerable<string> GetDynamicMemberNames()
        {
            return _properties.Keys;
        }

        public override bool TryGetMember(GetMemberBinder binder, out object result)
        {
            if (_properties.ContainsKey(binder.Name))
            {
                result = _properties[binder.Name];
                return true;
            }
            else
            {
                result = null;
                return false;
            }
        }

        public override bool TrySetMember(SetMemberBinder binder, object value)
        {
            if (_properties.ContainsKey(binder.Name))
            {
                _properties[binder.Name] = value;
                return true;
            }
            else
            {
                return false;
            }
        }
    }
