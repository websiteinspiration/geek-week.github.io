To receive and update data, you can write two functions for each of the fields, but I propose to do it differently. The structure of such functions will be identical, with the exception of the column name, so we will pass the field to the function. 







    #     
    def get_field(self, user_id, field):
        self.connect_to_db()
        request = f"SELECT {field} FROM user WHERE id=?"
        result = self.cursor.execute(request, (user_id,)).fetchone()
        self.close()
        return result[0]

    #   
    def set_field(self, user_id, field, value):
        self.connect_to_db()
        request = f"UPDATE user SET {field}=? WHERE id=?"
        self.cursor.execute(request, (value, user_id))
        self.connect.commit()
        self.close()
