図1木の例。

では、なぜ複数のスレッドでツリーをトラバースする必要があるのでしょうか。私の場合、ディスク上のファイルを検索する必要がありました。物理的にディスクが1つのストリームで機能することは明らかですが、単一のストリームがマルチレベルの階層でファイルを検索し、検索対象のファイルが1つのレベルの隣接するフォルダーにある場合、マルチスレッドファイル検索により高速化できます。また、マルチスレッド検索を使用することの妥当性の証明は、多くの成功した商用製品への実装です。私はアルゴリズムのアプリケーションの他の変形が可能であることを除外しません、コメントに書いてください。

ます。ここで何が起こっているのですか？アルゴリズムの本質は次のとおりです。

  1. 最初のストリームは、「極端な左」のパスに沿って、ルートから深さ全体までツリーをバイパスします。内側に移動すると常に左に移動します。つまり、常に最後の子ノードを選択します。
  2. 並行して、最初のスレッドが不足している子ノードをすべて収集し、それらをキュー（またはスタック）に送信します。そこで、別のスレッドがそれらを取得します。キューまたはスタックはマルチスレッドアプローチを実装する必要があり、アルゴリズムは繰り返され、ルートの代わりに取得したばかりのノードを置き換えます。

import java.io.File;
import java.util.concurrent.BlockingQueue;
import java.util.concurrent.Executor;
import java.util.concurrent.Executors;
import java.util.concurrent.LinkedBlockingDeque;

public class MultiThreadTreeWalker extends Thread {

    private static BlockingQueue<File> nodesToReview = new LinkedBlockingDeque<>();
    private File f;

    public MultiThreadTreeWalker(File f) {
        this.f = f;
    }

    public MultiThreadTreeWalker() {
    }

    public static void main(String[] args) {
        Executor ex = Executors.newFixedThreadPool(2);
        MultiThreadTreeWalker mw =  new MultiThreadTreeWalker(new File("C:\\"));
        mw.run();
        for (int i = 0;i<2;i++) {
            ex.execute(new MultiThreadTreeWalker());
        }
    }

    @Override
    public void run() {
        if (f != null) { //   
            reviewFileSystem(f);
        } else {
            try {
                while (true) {


                    f = nodesToReview.take();
                    reviewFileSystem(f);
                }
            } catch (InterruptedException e) {
                e.printStackTrace();
            }

        }
    }

    void reviewFileSystem(File f) {
        if (f == null) {
            return;
        }
        if (f.isFile()) {

            //  (+ )
            System.out.println(" " + f.getName() + "   " + Thread.currentThread());
            return;
        }
        File[] files = f.listFiles();

        if (files.length != 0) {

            for (int i = 0; i < files.length - 1; i++) {

                nodesToReview.add(files[i]); //    

            }
            //      

            File last = files[files.length - 1];

            reviewFileSystem(last);

        }

    }

}
