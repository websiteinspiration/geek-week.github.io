static member sizes matrix =

    let cols = matrix.values.[0,*].Length
    (rows, cols)

static member isEquallySized matrix1 matrix2 =
    let dim1 = Matrix.sizes matrix1
    let dim2 = Matrix.sizes matrix2
    (dim1 = dim2)

static member (==) (matrix1, matrix2) =
    if not (Matrix.isEquallySized matrix1 matrix2) then false
    else
        not (matrix1.values
               |> Array2D.mapi (fun x y v -> if matrix2.values.[x, y] <> v then false else true)
               |> Seq.cast<bool>
               |> Seq.contains false)

    //

行列とべき乗の積のF＃のコードを考えます。大きなパワーでスタックをオーバーフローさせないために、ここではテール再帰が使用されます。末尾再帰は、コンパイラが最終的にループに変換する再帰です。可能であれば、通常のテール再帰ではなく常にテール再帰を使用することをお勧めしますが、そのためには、各再帰フレームが最終的な計算値を返す必要があります。この値は通常アキュムレータと呼ばれ、次の再帰フレームに渡されます。つまり、スタックの上に計算値を返す通常の再帰とは異なり、テール再帰は計算値をスタックの下に渡します。新しい再帰フレームはそれぞれ独自の計算を行い、それらを以前に計算された値に追加します。これはバッテリーに保存されています。その後、再帰の最後のフレームが機能したため、アキュムレータにはすでに計算された値があり、単に結果として返されます。
