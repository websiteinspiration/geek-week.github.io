How was the selection

The second stage was the task - we need to check not only the level of knowledge, but also the willingness to invest time and effort in training. We took into account the successful solution of 4 problems out of 5 possible, the number and quality of attempts, implementation time. The testing platform made it possible to check the participants' decisions in real time, as well as immediately record all the results in the standings.

func Merge2Channels(f func(int) int, in1 <-chan int, in2 <-chan int, out chan<- int, n int) {
    line := make(chan int, n)
    out2 := make(chan int, n)
    go func() {
        for i := 0; i < n; i++ {
            var count = i
            var x1 = <-in1
            var x2 = <-in2
            go func() {
                var result = f(x1) + f(x2)
                out2 <- result
                line <- count
            }()
        }
    }()
    go func() {
        answer := make(map[int]int)
        for i := 0; i < n; i++ {
            var xL = <-line
            var xO = <-out2
            answer[xL] = xO
        }
        for i := 0; i < n; i++ {
            out <- answer[i]
        }
    }()
}
