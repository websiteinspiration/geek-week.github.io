4年前、私はすでにiOSでWebソケットを扱っていました。それから、cocoapodsライブラリの1つを使用して問題を解決しました。記事はHabréにあります。今日は、ココアポッドを使用しないネイティブの別のソリューションを示したいと思います。

Webソケットを操作するためのマネージャーを作成しました（この場合のマネージャーは、アプリケーション全体で一度しか作成されないため、アプリケーションのどこからでも呼び出すことができるシングルトンクラスです）。

私の目的にはネイティブの手段を使用します。

import Foundation

class WSManager {
    public static let shared = WSManager() //  
    private init(){}
    
    private var dataArray = [__]()
    
    let webSocketTask = URLSession(configuration: .default).webSocketTask(with: URL(string: "wss://__"*))
    
   //  
    public func connectToWebSocket() {
        webSocketTask.resume()
        self.receiveData() { _ in }
    }
    
//    
    public func subscribeBtcUsd() {
        let message = URLSessionWebSocketTask.Message.string("SUBSCRIBE: __")
        webSocketTask.send(message) { error in
            if let error = error {
                print("WebSocket couldn’t send message because: \(error)")
            }
        }
    }

//        
    public func unSubscribeBtcUsd() {
           let message = URLSessionWebSocketTask.Message.string("UNSUBSCRIBE: __ ")
           webSocketTask.send(message) { error in
               if let error = error {
                   print("WebSocket couldn’t send message because: \(error)")
               }
           }
       }
    
//  ,      
    func receiveData(completion: @escaping ([__]?) -> Void) {
      webSocketTask.receive { result in
        switch result {
            case .failure(let error):
              print("Error in receiving message: \(error)")
            case .success(let message):
              switch message {
                case .string(let text):
                    let data: Data? = text.data(using: .utf8)
                    let srvData = try? CODABLE_____.decode(from: data ?? Data())
                    for singleData in srvData ?? [] {
                        self.dataArray.append(__(1:  singleData.parametr1, 2:  singleData.parametr2, 3:  singleData.parametr3))  
                    }

                case .data(let data):
//        
                    print("Received data: \(data)")
              @unknown default:
                debugPrint("Unknown message")
              }
              
              self.receiveData() {_ in } // 
        }
      }
        completion(self.dataArray) //         
    }
}

ここにそのようなマネージャー、呼び出しの例があります

import UIKit

class MainViewController: UIViewController {
    
    private var dataArray = [__]()

    override func viewDidLoad() {
        super.viewDidLoad()
        WSManager.shared.connectToWebSocket() //  
        WSManager.shared.subscribeBtcUsd() //   
        self.getData() // 
    }

    private func getData() {
        // 
        WSManager.shared.receiveData() { [weak self] (data) in
            guard let self = self else { return }
            guard let data = data else { return }
             self.dataArray = data //             
        }
    }

}
