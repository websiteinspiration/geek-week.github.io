function SplitPane(props) {
  return (
    <div className="SplitPane">
      {
      props.left &&
      <div className="SplitPane-left">
        {props.left}
      </div>
     }
     {
      props.right &&
       <div className="SplitPane-right">
        {props.right}
       </div>
     }
    </div>
  );
}

function App() {
  return (

           }
           phone={
             <ContactsPhone name={el.phone} />
           }
         />
        )
      }
      right={
        <Chat />
      } />
  );
}
