def main():
    #   N, M  K

    threshold = int(sys.stdin.readline().strip())

    #    ยซยป ,

    #   
    bad_ips = set()
    last_ips = deque(maxlen=window_size)
    counter = Counter()

        #  IP-

        # ,

            #      
            #      
            oldest_ip = last_ips.pop()
            counter[oldest_ip] -= 1

                del counter[oldest_ip]

        #     
        last_ips.appendleft(current_ip)

            continue

        #

            bad_ips.add(current_ip)
            #  ยซยป   ,
