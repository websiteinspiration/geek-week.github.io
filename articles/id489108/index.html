Komponen berikut ini diperlukan untuk pengoperasian:

Isi dari conftest.py:
import pytest
from py.xml import html

#   
def pytest_html_results_table_header(cells):
	cells.insert(1, html.th('description'))		#  1- 
	cells.insert(2, html.th('a'))			#  2- 
	cells.insert(3, html.th('b'))			#  3- 
	cells.insert(4, html.th('sum'))			#  4- 
	cells.pop()

def pytest_html_results_table_row(report, cells):
	cells.insert(1, html.td( report.description ))	#  1-    
	cells.insert(2, html.td( report.a ))		#  2-    
	cells.insert(3, html.td( report.b ))		#  3-    
	cells.insert(4, html.td( report.sum ))		#  4-    
	cells.pop()

# hook       
@pytest.hookimpl(hookwrapper=True)
def pytest_runtest_makereport(item, call):
	pytest_html = item.config.pluginmanager.getplugin('html')
	outcome = yield
	report = outcome.get_result()
	#     -     __doc__    
	report.description = str( item.function.__doc__ )
	report.a = str( item.module.log['a'] )
	report.b = str( item.module.log['b'] )
	report.sum = str( item.module.log['sum'] )
	#  html  image    -     
	extra = getattr(report, 'extra', [])
	if report.when == 'call':
		#  html-
		if item.module.log_html != 'none':
			extra.append(pytest_html.extras.html( item.module.log_html ))
		#  img- (    )
		if item.module.log_img != 'none':
			extra.append(pytest_html.extras.image( item.module.log_img, mime_type='image/jpg', extension='jpg' ))
		#  img-  url-
		if item.module.log_img_url != 'none':
			extra.append(pytest_html.extras.image( item.module.log_img_url ))
		report.extra = extra

Intersepsi pelaksanaan uji coba dan pengisian laporan uji dilakukan melalui kait fungsi:
def pytest_runtest_makereport (item, panggilan) dengan dekorator: @ pytest.hookimpl (hookwrapper = True) .

			extra.append(pytest_html.extras.html( item.module.log_html ))

Itu ketika skrip uji dieksekusi, fungsi pytest_runtest_makereport (item, panggilan) akan dieksekusi , yang pada gilirannya:

  • report.description = str (item.function .__ doc__) - kolom report.description akan diberi nilai atribut __doc__ dari fungsi tes;
  • report.a = str (item.module.log ['a']) - kolom report.a akan diberi nilai log variabel global ['a'];
  • report.b = str (item.module.log ['b']) - kolom report.b akan diberi nilai log variabel global ['b'];
  • report.sum = str (item.module.log ['sum']) - kolom report.sum akan diberi nilai log variabel global ['sum'];
  • extra.append(pytest_html.extras.html( item.module.log_html )) ‚Äî html-, log_html;
  • extra.append(pytest_html.extras.image( item.module.log_img, mime_type='image/jpg', extension='jpg' )) ‚Äî img-, log_img;
  • extra.append(pytest_html.extras.image( item.module.log_img_url )) ‚Äî img-, log_img_url.

Parameter ditransfer ke laporan melalui variabel global dan atribut fungsi __doc__; selama setiap tes, kita harus mendefinisikan kembali variabel global.

Kode
#  
log = { 'a':'none', 'b':'none', 'sum':'none' }
log_html = 'none'
log_img = 'none'
log_img_url = 'none'


Kami akan memberikan nilai ke variabel global melalui fungsi:

Kode
#    
def set_global_var(a='none',b='none',sum_='none',html='none',img='none',img_url='none'):
	global log, log_html, log_img, log_img_url
	log = { 'a': a, 'b': b, 'sum': sum_ }
	log_html = html
	log_img = img
	log_img_url = img_url


Untuk mereset nilai variabel global secara otomatis dan tidak mengontrolnya di setiap pengujian, kami menggunakan fixture.

Kode
@pytest.fixture(scope="function", autouse=True)
def default_global_var():
	set_global_var()


Untuk menjalankan tes, buka direktori dengan skrip tes dan jalankan perintah:

pytest test.py -v --html=report.html --self-contained-html

Sebagai hasil dari pengujian, file report.html akan dibuat .
