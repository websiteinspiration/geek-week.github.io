module JustRead(
input	 logic	clk,
input  logic	rxf_n,
output logic	oe_n,
output logic   rd_n
);

enum {IDLE,TRANSFER} state = IDLE;

always @ (posedge clk)
begin
	oe_n <= 1;
	rd_n <= 1;
	case (state)
		IDLE: begin
			if (rxf_n == 0)
			begin
				oe_n <= 0;
				state <= TRANSFER;
			end
		end
		TRANSFER: begin
		   if (rxf_n == 0)
			begin
				oe_n <= 0;
				rd_n <= 0;
			end else
			begin
				state <= IDLE;
			end
		end
	endcase
end

endmodule

{
	if (blockSize > maxBlockSize)

	timespec before;

	for (int i = 0; i < totalSize; i += blockSize)
	{
		DWORD dwWritten;
		FT_Write(ftHandle0, buf, blockSize, &dwWritten);
		//    
		dwWrittenTotal += dwWritten;
	}
	timespec after;
	clock_gettime(CLOCK_REALTIME, &after);
	if (dwWrittenTotal < (DWORD)totalSize)
	{ 
		return -2;
	}
	//    
	uint64_t nsBefore = before.tv_nsec;
	uint64_t nsAfter = after.tv_nsec;

	//     
	nsAfter += (after.tv_sec - before.tv_sec) * 1000000000;
	
	//  
	nsAfter -= nsBefore;

	//     
	double res = ((double)nsAfter)/((double)1000000000);

	//   -   
	return ((double)dwWrittenTotal) / res;
}
