function addRoad(enter, hold, leave, curve, y) {
  var startY   = lastY();
  var endY     = startY + (Util.toInt(y, 0) * segmentLength);
  var n, total = enter + hold + leave;
  for(n = 0 ; n < enter ; n++)
    addSegment(Util.easeIn(0, curve, n/enter), Util.easeInOut(startY, endY, n/total));
  for(n = 0 ; n < hold  ; n++)
    addSegment(curve, Util.easeInOut(startY, endY, (enter+n)/total));
  for(n = 0 ; n < leave ; n++)
    addSegment(Util.easeInOut(curve, 0, n/leave), Util.easeInOut(startY, endY, (enter+hold+n)/total));
}

看板や木との衝突


道路の端に沿ってオブジェクトスプライトを追加してレンダリングできるようになったので、メソッドを変更してupdate()、プレーヤーが現在のセグメントでこれらのスプライトに遭遇したかどうかを判断する必要があります。四角形の交差の一般化された認識を実装

するには、補助メソッドを使用しUtil.overlap()ます。交差点が検出された場合、車を停止します。

if ((playerX < -1) || (playerX > 1)) {
  for(n = 0 ; n < playerSegment.sprites.length ; n++) {
    sprite  = playerSegment.sprites[n];
    spriteW = sprite.source.w * SPRITES.SCALE;
    if (Util.overlap(playerX, playerW, sprite.offset + spriteW/2 * (sprite.offset > 0 ? 1 : -1), spriteW)) {
      // stop the car
      break;
    }
  }
}

注：実際のコードを調べると、障害物を回避するために横に移動することができないため、実際には車を停止していないことがわかります。単純なハックとして、私たちはそれらの位置を修正し、スプライトの周りの側面に車を「滑らせ」ます。

マシンの更新


他の車が道路に沿って移動するように、最も単純なAIを与えます。

  • 一定速度で乗る
  • 追い越し時に自動的にプレイヤーを一周する
  • 追い越し時に他の車を自動的に回ります

注：カーブは実際のものではないため、道路上のカーブに沿って他の車を方向転換することを心配する必要はありません。車を道路のセグメントに沿って移動させるだけで、カーブに沿って自動的に通過します。

これはすべて、ゲームupdate()中、updateCars()各車を一定の速度で前進させ、このフレーム中に十分な距離を移動した場合に1つのセグメントから次のセグメントに切り替える通話中に発生します。

function updateCars(dt, playerSegment, playerW) {
  var n, car, oldSegment, newSegment;
  for(n = 0 ; n < cars.length ; n++) {
    car         = cars[n];
    oldSegment  = findSegment(car.z);
    car.offset  = car.offset + updateCarOffset(car, oldSegment, playerSegment, playerW);
    car.z       = Util.increase(car.z, dt * car.speed, trackLength);
    car.percent = Util.percentRemaining(car.z, segmentLength); // useful for interpolation during rendering phase
    newSegment  = findSegment(car.z);
    if (oldSegment != newSegment) {
      index = oldSegment.cars.indexOf(car);
      oldSegment.cars.splice(index, 1);
      newSegment.cars.push(car);
    }
  }
}

このメソッドupdateCarOffset()は、「人工知能」の実装を提供し、マシンがプレーヤーまたは他のマシンを移動できるようにします。これはコードベースで最も複雑な方法の1つであり、実際のゲームでは、マシンが単純なデモよりもはるかに現実的に見えるように、はるかに複雑にする必要があります。

このプロジェクトでは、単純なAIのブルートフォースを使用して、各マシンを強制します。

  • 20セグメントを楽しみにしています
  • 彼女の前で、道を横切る遅い車を見つけたら、彼女の周りを回ります
  • 道路左側の障害物から右折
  • 道路の右側にある障害物を左折
  • 残りの距離内で前方の障害物を回避するのに十分な方向を向く

また、プレイヤーには見えない車で不正行為をすることもできます。これにより、車がお互いに回って通り抜けることができなくなります。プレーヤーの視界内でのみ「スマート」に見えるはずです。

function updateCarOffset(car, carSegment, playerSegment, playerW) {

  var i, j, dir, segment, otherCar, otherCarW, lookahead = 20, carW = car.sprite.w * SPRITES.SCALE;

  // optimization, dont bother steering around other cars when 'out of sight' of the player
  if ((carSegment.index - playerSegment.index) > drawDistance)
    return 0;

  for(i = 1 ; i < lookahead ; i++) {
    segment = segments[(carSegment.index+i)%segments.length];

    if ((segment === playerSegment) && (car.speed > speed) && (Util.overlap(playerX, playerW, car.offset, carW, 1.2))) {
      if (playerX > 0.5)
        dir = -1;
      else if (playerX < -0.5)
        dir = 1;
      else
        dir = (car.offset > playerX) ? 1 : -1;
      return dir * 1/i * (car.speed-speed)/maxSpeed; // the closer the cars (smaller i) and the greater the speed ratio, the larger the offset
    }

    for(j = 0 ; j < segment.cars.length ; j++) {
      otherCar  = segment.cars[j];
      otherCarW = otherCar.sprite.w * SPRITES.SCALE;
      if ((car.speed > otherCar.speed) && Util.overlap(car.offset, carW, otherCar.offset, otherCarW, 1.2)) {
        if (otherCar.offset > 0.5)
          dir = -1;
        else if (otherCar.offset < -0.5)
          dir = 1;
        else
          dir = (car.offset > otherCar.offset) ? 1 : -1;
        return dir * 1/i * (car.speed-otherCar.speed)/maxSpeed;
      }
    }
  }
}

ほとんどの場合、このアルゴリズムは非常にうまく機能しますが、前に車がたくさん集まっていると、車が左から右、そして後ろに移動し、他の2台のマシン間の隙間に押し込もうとしていることがわかります。AIの信頼性を向上させるには多くの方法があります。たとえば、障害物を回避するのに十分なスペースがないと車が減速すると、自動車を減速させることができます。

ヘルパーメソッドをupdateHud()使用すると、値が変更された場合にのみDOM要素を更新できます。これは、更新に時間がかかる可能性があるため、値自体が変更されない場合は60fpsで実行しないでください。

function updateHud(key, value) { // accessing DOM can be slow, so only do it if value has changed
  if (hud[key].value !== value) {
    hud[key].value = value;
    Dom.set(hud[key].dom, value);
  }
}
