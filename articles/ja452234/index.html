ATMEGA 328P-MUの温度計と湿度計-Arduinoの開発レベルを上げる

今日、Arduinoプロジェクトの1つを共有したいと思います。昔、インターネットのどこかで、Arduinoについて知りました。私は非常に早くこのビジネスに参加しました、そこのエントリーのレベルは高くありません。しばらくして、センサーの束をすでに集めていたので、スマートホーム用のセンサーは、どういうわけかそれはまったく気になっていないと思っていました。モジュール、大きなプレーンボックス、ワイヤーの束、ホットグルー:)。私のボックスを温度センサーで、そしてたとえば同じXiaomiの温度センサーで見て、私はXiaomiのようになりたいと思ったが、同時にワイヤーとホットメルト接着剤を使用して10cm x 6cmのボックスで私のように再プログラムできることに気づきました。そしておそらく、PCBボード上の私のDIY arduinoプロジェクトの始まりが築かれました。

今日の記事では、atmega328p-muプロセッサをベースにした温度および湿度センサーについて説明します。これは、すべてのArduino開発者に知られているatmega328p-auプロセッサ（Arduino Uno、Pro Mini、Nano）の小型バージョン（絶対的な類似物）です。誰かが私の記事を以前に読んだことがあるなら、彼らは私がマイセンサーを好むことを知っています。それは何ですか？これは非常にシンプルで適切に設計されており、2.4 GHz、915、868、433 MHzの周波数でIOT無線ネットワークを作成するためのArduino IDEの（そしてそれだけでなく）重要でよく説明されているライブラリ、および485インターフェイスの有線ネットワークについては、すべてが言及されているわけではありません。プロトコルは常に進化しており、常に何かが追加されています。

最初に行われたのは、PCBボード上のセンサー自体です。私はケースを見ずにそれをしました、原則によれば、主なことはセンサーを作ることです、そして私は何とかケースを印刷します...ええ、それをしないでください:)実際、センサー自体はarduinka Pro Mini、nRF24l01無線モジュール、SHT20温度および湿度センサーと同じですが、ワイヤーとホットグルーがありません。 「ベルとホイッスル」の中で、これは無線でフラッシュするための外部SPIフラッシュドライブです（操作にはDualOptibutブートローダーが必要です。後で無線（バッテリーと半分のバッテリーにファームウェアがないため、回路ドライブにフラッシュドライブ）を配置するのをやめました）と、「crypto mic」ATSHA204A、いわば完全ハードウェアキット（Mysensorsでは、署名、暗号化などをアクティブ化するために、スケッチの最初に必要な#defを指定するだけです）。



掲示板自体はDiptraceプログラムで作成され、YouTubeのビデオチュートリアルを見た後、最初は「地獄」のように見えましたが、実際にはそれほど難しくはありませんでした。私はjlcpcb.comで中国でボードを注文しました。2ドル、どの色でも、2週間後にはすでに「自分の作品」が10枚届きます。





次のステップは身体の発達でした。ああ、これは同じ問題であることがわかりました。一般的に、私は簡単な方法を探すのではなく、Solid Worksを習得することにしました。結局のところ、これはDeeptraceとはまったく異なります。それにもかかわらず、私はこのエディターを勉強することをお勧めします。学習プロセスは、YouTubeでビデオレッスンをゆっくりと視聴し、エディターでレッスンを繰り返す1か月間続きました。ケースを開発する過程で、将来のケースのパラメータを考慮せずにデバイスボードを作成することは、シリーズからスティックを自分のホイールに挿入するという誤った判断であることが明らかになりました。ボードのバージョンの結果、ケースのボードの取り付けを考慮して3つが出てきました。これは最後のオプションではないと思います。

建物の開発当初は、FDM 3Dプリンターで印刷することを考えていましたが、森の奥に行くほど、すべてのウィッシュリストを再現できないことが明らかになりました。この理解が得られるまでに、もう1つの3DプリントテクノロジーであるSLAについてすでに学びました。二度考えずに印刷品質に感銘を受けずに、Ali- ANYCUBIC Photonのウィッシュリストが発行されました。 （リンクアナアリ、広告ではなく、アフィリエイトではない...単なるリンク）。

この記事を書いているときの私の経験に基づいて、すぐに書こうと思います...おお、このクールなものはクールなものです!!!もちろんケースはエディタで設計され、初めて印刷されたものではなく、多くの改善がありました。まあ、それはおそらく逆ではありません。その結果、思い通りの結果が得られました。かなり小型のデバイス、非常に正確なディテール、ボタン、フォント、頭に表されているすべてのものを備えた優れたDIYケース。マグネットがバックカバーに追加され、鉄の表面に簡単に取り付けられるようになりました。

void preHwInit()
{
  pinMode(led_pin, OUTPUT);
  digitalWrite(led_pin, OFF);
  pinMode(mode_pin, INPUT_PULLUP);
}

void before()
{
  set_led = loadState(100);
  if (set_led > 1) {
    set_led = 1;
    saveState(100, set_led);
  }
  delayTime = loadState(101);
  if (delayTime > 60) {
    delayTime = 3;
    saveState(101, delayTime);
  }
  send_batteryTime = loadState(102);
  if (send_batteryTime > 48) {
    send_batteryTime = 6;
    saveState(102, send_batteryTime);
  }


  digitalWrite(led_pin, ON);
}

void loop()
{
  if (configMode == 0) {



    if (sleep_flag == 0) {
      timer_status = sleep(digitalPinToInterrupt(mode_pin), FALLING, delayTime * 60 * 1000, false);
      //timer_status = sleep(digitalPinToInterrupt(mode_pin), RISING, delayTime * 60 * 1000, false);
      sleep_flag = 1;
    }
    if (timer_status == -1) {

      w_battetyTime = w_battetyTime + (delayTime * 60 * 1000);

      if (w_battetyTime >= send_batteryTime * 60 * 60 * 1000) {
        readBatLev();
        w_battetyTime = 0;
      }
      send_data();
      sleep_flag = 0;
    }
    if (timer_status == 0) {
      if (digitalRead(2) == LOW && flag_mode_button == 0) //  
      {
        flag_mode_button = 1;
        previousMillis = millis();
        wait(50);
      }
      if (digitalRead(2) == LOW && flag_mode_button == 1) {
        if ((millis() - previousMillis > 0) && (millis() - previousMillis <= 2000)) {
          if (millis() - lightMillis > 50)    {
            lightMillis = millis();
            digitalWrite(led_pin, !digitalRead(led_pin));
          }
        }
        if ((millis() - previousMillis > 2000) && (millis() - previousMillis <= 2500)) {
          digitalWrite(led_pin, OFF);
        }
        if ((millis() - previousMillis > 2500) && (millis() - previousMillis <= 4500)) {
          if (millis() - lightMillis > 25)    {
            lightMillis = millis();
            digitalWrite(led_pin, !digitalRead(led_pin));
          }
        }
        if (millis() - previousMillis > 4500) {
          digitalWrite(led_pin, OFF);
        }
      }
      if (digitalRead(2) == HIGH && flag_mode_button == 1) //   
      {
        if ((millis() - previousMillis > 0) && (millis() - previousMillis <= 2000)) {
          configMode = !configMode;
          flag_mode_button = 0;
          TRANSPORT_DEBUG(PSTR("MyS: CONFIGURATION MODE\n"));
          sleep_flag = 0;
          digitalWrite(led_pin, OFF);
        }
        if ((millis() - previousMillis > 2000) && (millis() - previousMillis <= 2500)) {
          flag_mode_button = 0;
          sleep_flag = 0;
        }
        if ((millis() - previousMillis > 2500) && (millis() - previousMillis <= 4500))
        {
          flag_mode_button = 0;
          sleep_flag = 0;
          digitalWrite(led_pin, OFF);
        }
        if (millis() - previousMillis > 4500) {
          flag_mode_button = 0;
          sleep_flag = 0;
          wait(50);
        }
      }
    }
  } else {
    if (listen_flag == 0) {
      RF24_startListening();
      listen_flag = 1;
    }
    if (millis() - lightMillis > 1000) {
      lightMillis = millis();
      digitalWrite(led_pin, !digitalRead(led_pin));
    }
    if (digitalRead(2) == LOW && flag_mode_button == 0) //  
    {
      flag_mode_button = 1;
      //previousMillis = millis();
      wait(50);
    }
    if (digitalRead(2) == LOW && flag_mode_button == 1) {
      
    }
    if (digitalRead(2) == HIGH && flag_mode_button == 1) //   
    {
       
      configMode = !configMode;
      listen_flag = 0;
      flag_mode_button = 0;
      TRANSPORT_DEBUG(PSTR("MyS: OPERATING MODE\n"));
      digitalWrite(led_pin, OFF);
      wait(50);
    }
  }
}



void receive(const MyMessage & message)
{
  if (message.sensor == SETTING_LED_SENS_ID) {
    if (message.type == V_VAR1) {
      if (message.getByte() <= 1) {
        set_led = message.getBool();
        saveState(100, set_led);
        send(msg_setting_led.set(set_led));
        if (set_led == 0) {
          TRANSPORT_DEBUG(PSTR("MyS: STATUS LED: OFF\n"));
        }
        if (set_led == 1) {
          TRANSPORT_DEBUG(PSTR("MyS: STATUS LED: ON\n"));
          if (set_led == 1) {
            digitalWrite(led_pin, ON);
            wait(50);
            digitalWrite(led_pin, OFF);
          }
        }
      }
    }
  }
  if (message.sensor == DELAY_TIME_SENS_ID) {
    if (message.type == V_VAR1) {
      if (message.getULong() <= 60 && message.getULong() != 0) {
        delayTime = message.getULong();
        saveState(101, delayTime);
        send(msg_delay_time.set(delayTime));
        TRANSPORT_DEBUG(PSTR("MyS: THE NEW INTERVAL TEMP&HUM SEND VALUE IS SET: %d MIN.\n"), delayTime);
        if (set_led == 1) {
          digitalWrite(led_pin, ON);
          wait(50);
          digitalWrite(led_pin, OFF);
        }
      } else if (message.getULong() > 60) {
        delayTime = 60;
        saveState(101, delayTime);
        send(msg_delay_time.set(delayTime));
        TRANSPORT_DEBUG(PSTR("MyS: THE NEW INTERVAL TEMP&HUM SEND VALUE IS SET: %d MIN.\n"), delayTime);
        if (set_led == 1) {
          digitalWrite(led_pin, ON);
          wait(50);
          digitalWrite(led_pin, OFF);
        }
      } else if (message.getULong() == 0) {
        delayTime = 1;
        saveState(101, delayTime);
        send(msg_delay_time.set(delayTime));
        TRANSPORT_DEBUG(PSTR("MyS: THE NEW INTERVAL TEMP&HUM SEND VALUE IS SET: %d MIN.\n"), delayTime);
        if (set_led == 1) {
          digitalWrite(led_pin, ON);
          wait(50);
          digitalWrite(led_pin, OFF);
        }
      }
    }
  }
  if (message.sensor == BATTARY_SEND_SENS_ID) {
    if (message.type == V_VAR1) {
      if (message.getULong() <= 168) {
        send_batteryTime = message.getULong();
        saveState(102, send_batteryTime);
        send(msg_battary_send.set(send_batteryTime));
        TRANSPORT_DEBUG(PSTR("MyS: THE NEW INTERVAL BATTERY SEND IS SET: %d HOUR\n"), send_batteryTime);
        if (set_led == 1) {
          digitalWrite(led_pin, ON);
          wait(50);
          digitalWrite(led_pin, OFF);
        }
      }
    }
  }
}

void send_data()
{
  humd = sht20.readHumidity();
  temp = sht20.readTemperature();
  int t_humd = (int)humd;
  int t_temp = (int)temp;
  if (abs(temp - oldtemp) >= tempThreshold) {
    send(msg_temp.set(temp, 1));
    oldtemp = temp;
    if (set_led == 1) {
      digitalWrite(led_pin, ON);
      wait(50);
      digitalWrite(led_pin, OFF);
    }
  }
  wait(100);
  if (abs(humd - oldhumd) >= humThreshold) {
    send(msg_hum.set(humd, 1));
    oldhumd = humd;
    if (set_led == 1) {
      digitalWrite(led_pin, ON);
      wait(50);
      digitalWrite(led_pin, OFF);
    }
  }
  TRANSPORT_DEBUG(PSTR("MyS: DATA - TEMPERATURE: %d, HUMIDITY %d\n"), t_temp, t_humd);
}


void readBatLev() {
  ADMUX = _BV(REFS1) | _BV(REFS0) | _BV(MUX0);
  wait(100);
  RF24_startListening();
  wait(200);
  ADCSRA |= _BV(ADSC);
  while (bit_is_set(ADCSRA, ADSC));
  uint8_t low  = ADCL;
  uint8_t high = ADCH;
  long temp = (high << 8) | low;
  float vcc = temp * 1.1 / 1023 * BAT_COOF * 100;
  battery = map((int)vcc, BAT_MIN, BAT_MAX, 0, 100);
  if (battery < 0) {
    battery = 0;
  }
  if (battery > 100) {
    battery = 100;
  }
  TRANSPORT_DEBUG(PSTR("MyS: BATTERY LEVEL: %d, PREVIUS BATTERY LEVEL: %d\n"), battery, old_battery);
  TRANSPORT_DEBUG(PSTR("MyS: BATTERY LEVEL ADC: %d\n"), temp);

  /*

        wait(100);

        TRANSPORT_DEBUG(PSTR("MyS: SEND BATTERY LEVEL\n"));

  wait(100);
