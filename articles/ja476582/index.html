DWTカウンター

stm32から入手できるDWTモジュール（データウォッチポイントとトレースユニット）に含まれている非常に役立つものについての小さなメモ。

DWTモジュール自体はかなり複雑なもの（あなたが読むことができるでここ 75ページ）、そして彼は、デバッグのために責任があります。ただし、モジュール全体についてではなく、そのコンポーネントの1つであるクロックカウンター（以下、DWTカウンター）について話しています。

実際、DWTカウンターは32ビットのレジスターであり、その値は後続の各クロックサイクルで1ずつ増加します。このレジスタに読み書きできるので、プログラムの一部の実行時間を測定して測定し、マイクロ秒の遅延を整理できます。このカウンターは完全に独立しています。

プログラムまたは関数の任意の部分の実行時間を測定するには、測定する必要があります...

  #define    DWT_CYCCNT    *(volatile uint32_t*)0xE0001004
  #define    DWT_CONTROL   *(volatile uint32_t*)0xE0001000
  #define    SCB_DEMCR     *(volatile uint32_t*)0xE000EDFC

  char str[16] = {0,};
  uint32_t count_tic = 0;

  SCB_DEMCR |= CoreDebug_DEMCR_TRCENA_Msk;//   DWT
  DWT_CONTROL|= DWT_CTRL_CYCCNTENA_Msk; //  
  DWT_CYCCNT = 0;//  

  //     

  count_tic = DWT_CYCCNT; // - 
  snprintf(str, 16, "Takt %lu\n", count_tic);
  HAL_UART_Transmit(&huart1, (uint8_t*)str, strlen(str), 1000);


72 MHzの周波数で測定されたHAL_Delay（1000）。

マイクロ秒のポーズを整理するには、ファイルdelay_micros.hを作成する必要があります...

#ifndef __DELAY_US_H__
#define __DELAY_US_H__

#ifdef __cplusplus
 extern "C" {
#endif

/******************************************************************************/
/*      */
#include "main.h"

//#define DWT_CONTROL *(volatile unsigned long *)0xE0001000
//#define SCB_DEMCR   *(volatile unsigned long *)0xE000EDFC

/******************************************************************************/
/* inline func */
__STATIC_INLINE void DWT_Init(void)
{
	CoreDebug->DEMCR |= CoreDebug_DEMCR_TRCENA_Msk; //   
	DWT->CTRL |= DWT_CTRL_CYCCNTENA_Msk;   //  
}

__STATIC_INLINE void delay_us(uint32_t us)
{
	uint32_t us_count_tic =  us * (SystemCoreClock / 1000000U);
	DWT->CYCCNT = 0U;
	while(DWT->CYCCNT < us_count_tic);
}

#ifdef __cplusplus
}
#endif
#endif //__DELAY_US_H__

...そしてそれをプロジェクトに追加します。

main.cにインクルードを作成します...

#include "delay_micros.h"

無限ループの前に、カウンターを初期化して......

DWT_Init();

そして遅れはそうしています...

delay_us(100); // 100
