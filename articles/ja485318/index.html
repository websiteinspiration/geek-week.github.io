from IPython.display import HTML, display

def dataToTable(data, columns = None):
    if len(data) == 0 :  
        display(HTML('<b> </b>'))
        return
    
    hdr = ''
    if columns != None:
        for col in columns: #   
            hdr = hdr + '<th>' + col + '</th>'
            
        hdr = '<tr>' + hdr + '</tr>'

    dt = ''
    for row in data: #     
        dt = dt + '<tr>'
        for cell in row: #    
            dt = dt + '<td>' + str(cell) + '</td>'
        dt = dt + '</tr>'
            
    display(HTML('<table>' + hdr + dt + '</table>')) #
