impl Component {
    fn new<Model: 'static>(node: Node<Model>, model: Model) -> Self {
        let use_model_closure = |comp: &Component| {
            comp.model.downcast_ref::<Model>().unwrap();
        };
        Self {
            node: Box::new(node),
            model: Box::new(model),
            use_model_closure,
        }
    }

    fn use_model(&self) {
        (self.use_model_closure)(self);
    }
