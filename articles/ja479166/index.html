tarantoolの制限付き有効期限付きモジュールを作成する


しばらく前に、私たちはtarantool スペースでタプルをクリーニングする問題に直面しました。 tarantoolがすでにメモリ不足になっているときではなく、事前に一定の頻度でクリーニングを開始する必要がありました。 Tarantoolと呼ばれるこのタスクのためのLuaで書かれたモジュールがあるexpirationdを。このモジュールを短時間使用したところ、このモジュールは適切ではないことに気付きました。大量のデータを定期的にクリーニングすると、LuaがGCにぶら下がっていました。したがって、ネイティブプログラミング言語で記述されたコードが問題を最善の方法で解決することを期待して、上限付きの有効期限付きモジュールの開発を検討しました。

expirationd_iterate:


static bool
expirationd_iterate(struct expirationd_task *task, box_iterator_t **iterp)
{
  box_iterator_t *iter = *iterp;
  box_txn_begin();
  for (uint32_t i = 0; i < task->scan_size; ++i) {
    box_tuple_t *tuple = NULL;
    if (box_iterator_next(iter, &tuple) < 0) {
      box_iterator_free(iter);
      *iterp = NULL;
      box_txn_rollback();
      return false;
    }
    if (!tuple) {
      box_iterator_free(iter);
      *iterp = NULL;
      box_txn_commit();
      return true;
    }
    if (expirationd_expired(task, tuple))
      expirationd_delete(task, tuple);
    else if (expirationd_breakable(task))
      break;
  }
  box_txn_commit();
  return true;
}

expirationd_expired:


static bool
expirationd_expired(struct expirationd_task *task, box_tuple_t *tuple)
{
  const char *buf = box_tuple_field(tuple, task->field_no - 1);
  if (!buf || mp_typeof(*buf) != MP_UINT)
    return false;
  uint64_t val = mp_decode_uint(&buf);
  if (val > fiber_time64() / 1000000)
    return false;
  return true;
}

expirationd_delete:


static void
expirationd_delete(struct expirationd_task *task, box_tuple_t *tuple)
{
  uint32_t len;
  const char *str = box_tuple_extract_key(tuple, task->space_id, task->rm_index_id, &len);
  box_delete(task->space_id, task->rm_index_id, str, str + len, NULL);
}

expirationd_breakable:


static bool
expirationd_breakable(struct expirationd_task *task)
{
  return task->it_index_id != task->rm_index_id && task->it_index_type == ITER_GT;
}
