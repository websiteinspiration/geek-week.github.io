Tic Tac Toe（PixiJS）



脆弱で愚かな、夢のようなタイプセッターがプログラマーになることを決めたが、何も起こらなかった...しかし、彼はプログラミングをあきらめず、小さなプログラムから始めることを決めた...

これは私が思いつく最高の説明だ。この目的のために、スキルを磨くための簡単なプログラムを書き始め、いつもの言語で新しいデザインに精通し、正直に言うと、喜びさえ感じ始めました。

開発経験がほとんどない場合は記事が役立ちます。すでに開発経験がある場合は、もっと価値のあるものに時間を費やしてください。

これはトレーニングではありません。もっとブログみたい。

目標は、ゲームの三目並べの3つのバージョンを作成することでした。

1-最もシンプル（美しいビジュアルなし、DOMを使用）
2-一緒にプレイする機会を与える（1台のコンピュータ）
3-これをすべてキャンバス

に転送します。三目並べについては説明しません。ゲームの原理を誰もが知っていることを願っています。すべての便利なリンク（リポジトリ、ドキュメント）は記事の最後にあります。

function botMove() {
	var items = document.querySelectorAll(".app_block:not(.active)");

	var step = getRandomInt(items.length);

	items[ step ].innerHTML = "0";
	items[ step ].classList.add("active");
	items[ step ].classList.add("active_o");

	var result = checkMap();
	if( result.val) {
		setTimeout(function() {
			exit(result);
		}, 1);
	}

	movePlayer = !movePlayer;
}

function getRandomInt(max) {
	return Math.floor(Math.random() * Math.floor(max));
}

2番目のバージョン


同じコンピュータの2人のプレーヤー。

クリックハンドラーのロジックの一部を別の関数に取り込み、クリックされたボタンを特定する必要があるため、呼び出しコンテキストを関数に渡しました。

var items = document.getElementsByClassName("app_block");
var movePlayer = true;
var game = true;

for (var i = 0; i < items.length; i++) {
	items[i].addEventListener("click", function() {
		var collecion = document.querySelectorAll(".app_block:not(.active)");
		if(collecion.length == 1) {
			exit({win: "other"});
		}

		if( !this.classList.contains("active") ){
			if( movePlayer) {
				firstPlayer(this);
			} else {
				secondPlayer(this);
			}
		}
	});
}

2つの関数に分割しましたが、コードが重複しています。理想的には、3で割ります。1つはメイン、2つはコンテキストで動作します。

function firstPlayer(that) {
	if(that.innerHTML == "") {
		that.classList.add("active");
		that.classList.add("active_x");
		that.innerHTML = "x"
	}

	var result = checkMap();
	if( result.val) {
		game = false;
		setTimeout(function() {
			exit(result);
		}, 10);
	}

	movePlayer = !movePlayer;
}

function secondPlayer(that) {
	if(that.innerHTML == "") {
		that.classList.add("active");
		that.classList.add("active_o");
		that.innerHTML = "0"
	}

	var result = checkMap();
	if( result.val) {
		game = false;
		setTimeout(function() {
			exit(result);
		}, 10);
	}

	movePlayer = !movePlayer;
}

おそらくこれが最も興味深い点ですこのゲームは、DOM要素の相互作用ではなく、ゲームのように見えます。

PixiJSを選択しました。このライブラリの+と-に​​ついては何も言えませんが、60,000の要素があり、それらがすべてアニメーション化されている1つの例を見ました。アニメーションはシンプルですが、FPSは50〜60のままです。私はそれが好きで、私はドキュメントを読み始めました。英語についての私の知識は最小限で難しいとすぐに言いますが、ロシア語の記事はほとんどありません（または私は見た目がよくありません）。いばらを通り抜けるには、Google翻訳の助けを借りて突く必要がありました。Yulia Pucnina

によるこのトピックに関する1つのレポート「Pixi jsを使用した脂肪アニメーション」のみを視聴しました。

レポートは2014年のものであり、APIが変更される可能性があることを理解する必要があります。ドキュメンテーションに1つ目、ビデオに2つ目。だから私は勉強しました。このような単純なプロトタイプを書くのに4時間かかりました。コードにより近い。

for (let i = 0; i < 9; i++) {
    let container = new PIXI.Container();
	let block = new PIXI.TilingSprite( PIXI.Texture.from("images/bg.png") , 240, 130);
    
	container.x = (i % 3) * 240;
    container.y = Math.floor(i / 3) * 130;
    container.addChild(block);
    
    let text = new PIXI.Text("");
    text.anchor.set(0.5);
    text.x = container.width / 2;
    text.y = container.height / 2;
    container.addChild(text);
    
    container.interactive = true;    
    container.on("mousedown", function () {
        addValueInBlock(this);
    });
    
    wrapper.addChild(container);
}
