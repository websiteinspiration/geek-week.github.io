試験の結果をチェックするブラウザ拡張を作成します


私は、他の卒業生と同様に、試験について心配しています。USEで得たポイントに依存しすぎているので、今では他のことを考えるのは難しいです。2分ごとにサイトcheck.ege.edu.ruを更新しないようにするために、これを行う拡張機能を作成し、同時にいずれかの試験がチェックされた場合に通知を送信することにしました。

タスクはそれほど難しくありませんが、不快な瞬間が1つあります。結果が公開されるサイトでは、ブラウザを閉じるたびに、参加者に関する情報を再入力する必要があります。これはあまり便利ではないので、ちょっと即興する必要がありました。必要な情報はすべてCookieに保存されるため、データを再入力することなく、必要なときに保存して更新することができます。したがって、拡張のロジックは次のとおりです。


ブラウザの拡張機能はjsで記述され、htmlとcssを使用して発行されます。一般に、拡張機能の開発は、サイトの作成とそれほど変わりません。通常、拡張機能には次の「スケルトン」があります。


Manifest.json


このファイルには、バージョン、名前、説明、接続ファイルなどの基本情報が保存されています。

Background.js


このファイルのコードは、ブラウザーを開いた直後に起動されます。これは、拡張機能のメインロジックが配置される場所です。

background.js
chrome.alarms.create("1min", {  //     
  delayInMinutes: 1,
  periodInMinutes: 1,
});

chrome.alarms.onAlarm.addListener(function(alarm) {
  if (alarm.name === "1min") {
    chrome.cookies.getAll({"url": 'http://check.ege.edu.ru'}, function(cookie) { 
    if (cookie.length){  
      chrome.storage.local.set({'sCookie': cookie}); 
//     cookie ,   
      checkUpdates(); //   
    }else{
      chrome.storage.local.get(['sCookie'], function(result) {
        if (!jQuery.isEmptyObject(result)){ 
//  ,    
          chrome.cookies.set({
              "url": 'http://check.ege.edu.ru',
              "name": result["sCookie"][0]["name"],
              "value": result["sCookie"][0]["value"]
          }, function(){
            checkUpdates(); //    
          });
        }
      });
    }
  });
  }
});

function checkUpdates(){
  var myInit = { 
      method: 'GET',
      credentials: 'include'};
  fetch('http://check.ege.edu.ru/api/exam', myInit).then(r => r.text()).then(result => { //   
    var examRes = {};
    jQuery.parseJSON(result)['Result']['Exams'].forEach(function(element) {
      examRes[element['Subject']] = element['TestMark']; 
//     examRes
    });
    chrome.storage.local.get(['examRes'], function(result) {
      for (var element in result['examRes']){
        if (result['examRes'][element] != examRes[element]){ 
            showNotification(element, examRes[element]);  
            chrome.storage.local.set({'examRes': examRes});
//    ,   ,  
//  
//    
        }
      }
    });
  })
}

function showNotification(subName, mark){ 
//  ,      
  chrome.notifications.create('reminder', {
        type: 'basic',
        iconUrl: 'icon.png',
        title: ' !',
        message: subName + ' - ' + mark
     });
}

ヘッダーには、拡張アイコンがクリックされるたびに実行されるpopup.jsファイルを含めました。その内容は次のとおりです。

popup.js
chrome.cookies.getAll({"url": 'http://check.ege.edu.ru'}, function(cookie) {
// ,      cookie
  if (cookie.length){
    chrome.storage.local.set({'sCookie': cookie});
    createTable();
//   ,      
  }else{
    chrome.storage.local.get(['sCookie'], function(result) {
//  ,  ,       cookie 
      if (!jQuery.isEmptyObject(result)){
        chrome.cookies.set({
          "url": 'http://check.ege.edu.ru',
          "name": result["sCookie"][0]["name"],
          "value": result["sCookie"][0]["value"]
        }, function(){
//  ,      
          createTable();
        });
      }else{
//   ,    check.ege.edu.ru   
        chrome.tabs.create({url : "http://check.ege.edu.ru"}); 
      }
    });
  }
});

function createTable(){
  var myInit = { 
      method: 'GET',
      credentials: 'include'};
  fetch('http://check.ege.edu.ru/api/exam', myInit).then(r => r.text()).then(result => {
//     
    jQuery.parseJSON(result)['Result']['Exams'].forEach(function(element) {
      if (element['HasResult'] == false){
//    ,       " "
        $("#mainTable").append('<tr><td>'+element['Subject']+'</td><td> </td></tr>');
      }else{
//  ,      
        $("#mainTable").append('<tr><td>'+element['Subject']+'</td><td>'+element['TestMark']+'</td></tr>');
      }
    });
  })
}
