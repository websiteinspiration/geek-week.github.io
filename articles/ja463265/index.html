ArduinoからI2CへのPCF8563リアルタイムクロックの友

こんにちは、ハブル。PCF 8563チップでI2Cを使用してRTCをたまたま使用しましたが、Arduinoを使用して作業するための通常の機能が見つからなかったため、I2Cを介して直接作業する必要がありました。私は英語のインターネットで説明のチュートリアルを見つけ、プロセスの詳細を説明しました。この記事の一部は、このRTCに関する独自の経験の一部を翻訳したものです。



日時を設定する


マニュアル（ここにあります）に最初に登り、そこでレジスター編成の表を探します。



日付と時刻を設定するには、02hから08hまでのレジスタが必要です。これらのレジスターのデータは、2進10進形式（BCD）で保管されるため、秒、分などを記録します。最初に、変換関数（10進形式からBCDへ、またはその逆）を実装し、Wire.write（）コマンドを使用して、それらを02hから始まるレジスタに書き込みます。同じ方法で、BCDから10進法に転送し、同じレジスタ02hから読み取ります。表からわかるように、一部のレジスタではビットが使用されていません（文字xで示されています）。読み取り時に情報が過負荷にならないようにするため、不要なビットをリセットするビット単位のAND演算（論理「AND」）を使用します。たとえば、月の日（Days）、ビット0から5が必要です。操作（dayOfMonth＆B00111111）を使用して、6ビットと7ビットをリセットします。それ以外はすべて変更されません。

シリアルモニターに情報を出力して、コード自体を直接：

#include "Wire.h"
#define PCF8563address 0x51 //     

byte second, minute, hour, dayOfWeek, dayOfMonth, month, year;
String days[] = {"Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday" };

//     -    .
byte bcdToDec(byte value)
{
 return ((value / 16) * 10 + value % 16);
}
//  
byte decToBcd(byte value){
 return (value / 10 * 16 + value % 10);
}

//       PCF8563
void setPCF8563()
{
 Wire.beginTransmission(PCF8563address);
 Wire.write(0x02);
 Wire.write(decToBcd(second)); 
 Wire.write(decToBcd(minute));
 Wire.write(decToBcd(hour));   
 Wire.write(decToBcd(dayOfMonth));
 Wire.write(decToBcd(dayOfWeek)); 
 Wire.write(decToBcd(month));
 Wire.write(decToBcd(year));
 Wire.endTransmission();
}

//       PCF8563
void readPCF8563()
{
 Wire.beginTransmission(PCF8563address);
 Wire.write(0x02);
 Wire.endTransmission();
 Wire.requestFrom(PCF8563address, 7);
 second   = bcdToDec(Wire.read() & B01111111); //      
 minute   = bcdToDec(Wire.read() & B01111111); 
 hour    = bcdToDec(Wire.read() & B00111111); 
 dayOfMonth = bcdToDec(Wire.read() & B00111111);
 dayOfWeek = bcdToDec(Wire.read() & B00000111); 
 month   = bcdToDec(Wire.read() & B00011111); 
 year    = bcdToDec(Wire.read());
}
void setup()
{
 Wire.begin();
 Serial.begin(9600);
 //     
 second = 0;
 minute = 28;
 hour = 9;
 dayOfWeek = 2;
 dayOfMonth = 13;
 month = 8;
 year = 13;
 setPCF8563();
}
void loop()
{
 readPCF8563();
 Serial.print(days[dayOfWeek]); 
 Serial.print(" "); 
 Serial.print(dayOfMonth, DEC);
 Serial.print("/");
 Serial.print(month, DEC);
 Serial.print("/20");
 Serial.print(year, DEC);
 Serial.print(" - ");
 Serial.print(hour, DEC);
 Serial.print(":");
 if (minute < 10)
 {
  Serial.print("0");
 }
 Serial.print(minute, DEC);
 Serial.print(":"); 
 if (second < 10)
 {
  Serial.print("0");
 } 
 Serial.println(second, DEC); 
 delay(1000);
}


アラームを設定する


PCF 8563では、アラームを有効にして、特定の時間、曜日、または月に設定できます。アラーム設定は、レジスタに09hから0hまでのバイナリ-10進形式で配置されています。



アラームを分や時間などの一部の設定に従って機能させるには、7番目のビット（有効ビット）を1に設定する必要があります。これには、論理演算OR（OR）と値B10000000を使用します。

アラームのチェックは、ハードとソフトの2つの方法で実行できます。ソフトメソッドを使用して、レジスタ0x01の3番目のビット（AFアラームフラグビット）をチェックし、トリガーされると1に等しくなり、0に設定すると、アラームが発生します。



ハード検証では、同じレジスタ（AIE）の1ビットを1に変換する必要があります。アラームが鳴ると、ボードのINT（割り込み）ピンが導通位置になり、オープンドレイン出力になるため、たとえば、LEDを抵抗で安全にはんだ付けし、すべてを5ボルトに接続できます。

ジェネレータとして使用


多くのRTCと同様に、PCF8563は信号発生器として使用できます。COTピン（マイクロ回路のピン7）はオープンドレインであるため、このピンを介してLEDを異なる周波数で点滅させることができますレジスタ0x0Dは周波数パラメータに対応しています。



このレジスタの7番目のビット（FE）は、ジェネレータをオンにします。ビット0および1は、目的の周波数を設定します。ビット6〜2は使用されません。

以下の値をレジスタに書き込むと、目的の周波数を取得できます。

  • 10000000-32.768 kHz;
  • 10000001-1.024 kHz;
  • 10000010-32 Hz;
  • 10000011-1 Hz;
  • 0-出力を無効にします。

スケッチの例：

#include "Wire.h"
#define PCF8563address 0x51

void PCF8563oscOFF()
//  
{
 Wire.beginTransmission(PCF8563address);
 Wire.write(0x0D);
 Wire.write(0);
 Wire.endTransmission();
}

void PCF8563osc1Hz()
//    1 Hz
{
 Wire.beginTransmission(PCF8563address);
 Wire.write(0x0D);
 Wire.write(B10000011);
 Wire.endTransmission();
}

void PCF8563osc32Hz()
//    32 Hz
{
 Wire.beginTransmission(PCF8563address);
 Wire.write(0x0D);
 Wire.write(B10000010);
 Wire.endTransmission();
}

void PCF8563osc1024kHz()
//    1.024 kHz
{
 Wire.beginTransmission(PCF8563address);
 Wire.write(0x0D);
 Wire.write(B10000001);
 Wire.endTransmission();
}

void PCF8563osc32768kHz()
//    32.768 kHz
{
 Wire.beginTransmission(PCF8563address);
 Wire.write(0x0D);
 Wire.write(B10000000);
 Wire.endTransmission();
}

void setup()
{
 Wire.begin();
}

void loop()
{
 PCF8563osc1Hz();
 delay(2000);
 PCF8563osc32Hz();
 delay(2000);
 PCF8563osc1024kHz();
 delay(2000);
 PCF8563osc32768kHz();
 delay(2000);
 PCF8563oscOFF();
 delay(2000);
}
