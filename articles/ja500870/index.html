魔法の速さ


これで実装を開始できます。簡単なビューから始めましょう。すべてのIBOutletとビューの設定を作成する必要があります。

//    ,   view     UIKit. 
import UIKit

class ShopView: UIView{
    //       storyboard. 
    @IBOutlet weak var tableView: UITableView!
    //      view. 
    func configure(){
        //     ,    ,      ...
        tableView.estimatedRowHeight = 50
        tableView.rowHeight = 100
    }
}

これがビューの外観です。configure（）関数では、設定を行いますが、ビューはアニメーションやボタンのタイトルの変更などに使用する必要があります。しかし、なぜこの利点をそれほど使用しないのですか？したがって、たとえば、コントローラーでアニメーションを使用します。view.hideBorders（self）で、これで完了です。viewは、すでに美しいアニメーションを作成します。

//    view  . 
func hideBorders(view: UIView){
    UIView.animate(withDuration: 0.3,
                   delay: 0,
                   options: .curveEaseInOut,
                   animations: {
           //         constraint. 
           self.createViewHeighConstraint.constant = view.bounds.height * 0.6
           //   ,   ,    . 
           self.editButton.layer.cornerRadius = 20
           self.postView.layer.shadowOpacity = 0
           self.postView.layer.cornerRadius = 0
           view.layoutIfNeeded()
    }, completion: nil)
}

火！しかし、私たちのプロジェクトに戻って、モデルを実装する時間です。そして、私たちは商品を持っているでしょう、そして彼らのグループはそうかもしれません。

//   ,   model       View.
// ,    ,       Foundation.
import Foundation
//      ,  . 
public struct Product {
  public let name: String
}
//    ,      ,   . 
public struct ProductGroup {
  public let products: [Product]
  public let title: String
}
//  swift   ()    extension. 
extension ProductGroup{
    //  ,     ( Product). 
    public static func vegetables() -> ProductGroup{
       //   .
        let products = [
            Product(name: ""),
            Product(name: ""),
            Product(name: "")
        ]
        //      "". 
        return ProductGroup(products: products, title: "")
    }
}

すばらしい、私たちのモデルは準備ができています。コントローラを組み立てるために残っています。まず最初に、2つのメインエンティティ、モデルとビューを作成する必要があります。次に、プライベート拡張機能のコントローラー設定を取り出し、モデルからテーブルにデータを転送します。

//  controller'e      ,      UIKit.
import UIKit

class ShopController: UIViewController {
    //  -     .
    private var shopModel = ProductGroup.vegetables()
    
    //     (       )
    private var shopView: ShopView! {
        guard isViewLoaded else { return nil }
        return (view as! ShopView)
    }
    //      view,         controller'a  view  . 
    override func viewDidLoad() {
        super.viewDidLoad()
        //    view      .
        configure()
    }
}

private extension ShopController{
    //          .
    func configure(){
        shopView.tableView.delegate = self
        shopView.tableView.dataSource = self
    }
}

//      . 
extension ShopController: UITableViewDataSource, UITableViewDelegate{

    func tableView(_ tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
        //  -   -   
        return shopModel.products.count
    }
    
    func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) -> UITableViewCell {
        //   ,   reuse Id ,      .
        let cell = tableView.dequeueReusableCell(withIdentifier: "ProductCell")
        //    label  cell     . 
        cell?.textLabel?.text = shopModel.products[indexPath.row].name
        return cell!
    }
}

彼はビューについて説明することを約束しました、忘れていませんか？）私たちのshopViewは計算されたプロパティです。ここでは、このプロパティにアクセスしたときにビューが意図せずにロードされないように、isViewLoadedをチェックします。ビューがすでにロードされている場合は、強制的にShopViewに変換します。

以上で、MVCの準備は完了です！command + Rを押して3つのセルを見るだけです。
