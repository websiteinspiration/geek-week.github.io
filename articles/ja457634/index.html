Verilog上のDDS Synthesizer


この投稿では、VerilogでDDSシンセサイザーを作成する方法について説明します。これは、正弦波発振を生成するために使用され、その周波数と初期位相は、8ビットのユニポーラDACで使用するために調整および計算できます。シンセサイザーがどのように機能するかは、Components and Technologies誌の記事に詳しく書かれています。サインテーブルの使用メモリ量を減らすには、対称性を使用します。


LinuxでのコンパイルにはIverilogを使用し、GTKWaveを表示しました。便宜上、簡単なMakefileが作成されました。多分誰かのために役立つでしょう。最初に、iverilogコンパイラーを使用してtb.outファイルを取得し、それをiverilogと共にインストールされるvvpシミュレーターに送信します。その結果、vvpはプロジェクトで使用されるすべての変数（信号）を含むout.vcdを生成します。上記に加えて、表示ターゲットは可変ファイルを使用してGTKWaveを起動し、信号の波形を確認できます。


SRC = nco.v
TB = nco_tb.v

all:
    iverilog -o tb.out $(TB)
    vvp -lxt tb.out

check:
    iverilog -v $(TB) 

display:
    iverilog -o tb.out $(TB)
    vvp -lxt tb.out
    gtkwave out.vcd &

clean:
    rm -rf *.out *.vcd *.vvp

module nco(clk,
    rst,
    out
    );

input clk, rst;
output reg [7:0] out;

reg [5:0] phase_inc = 6'h1;
reg [7:0] phase_acc = 0;

parameter LUT_SIZE = 64;
reg [6:0] lut [0:LUT_SIZE-1];

always @(posedge clk) begin
    if (rst) begin
        phase_inc = 6'h1;
        phase_acc = 0;
        out = 0;
        lut[0] = 7'h00;
        //    
        lut[63] = 7'h7F;
    end
    else begin
        //      1   
        if (phase_acc[7:6] == 2'b00) begin
        //        LUT
            out <= {1'b1,lut[phase_acc[5:0]]};
        end
        if (phase_acc[7:6] == 2'b01) begin
            out <= {1'b1,lut[~phase_acc[5:0]]};
        end
        if (phase_acc[7:6] == 2'b10) begin
            out <= {1'b0,~lut[phase_acc[5:0]]};
        end
        if (phase_acc[7:6] == 2'b11) begin
            out <= {1'b0,~lut[~phase_acc[5:0]]};
        end
        phase_acc <= phase_acc + {2'b0,phase_inc};
    end
end

endmodule
