if __name__ == "__main__":
    ffi = cffi.FFI()
    #   
    PATH = os.getcwd()

    # test.h    
    #      build.py
    with open(os.path.join(PATH, "src/c/test.h")) as f:
        ffi.cdef(f.read())

    ffi.set_source("_test", #    cffi,   _
        #  test.h,     _test
        '#include "../src/c/test.h"',
        #   libtest.so (  ) 
        #  _test.cpython-36m-x86_64-linux-gnu.so ( CFFI)
        libraries=[os.path.join(PATH, "lib/test"), "./test"],
        library_dirs=[PATH, 'objs/'],
    )

    #  _test   lib
    ffi.compile(tmpdir='./lib')
