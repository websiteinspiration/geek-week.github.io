仮想メモリ


各プロセスを別のプロセスから分離できるように、仮想メモリが必要です。彼を止めることができませんでした。仮想メモリがない場合、毎回メモリ内の異なるアドレスにelfファイルをロードする必要があります。ただし、ご存じのとおり、実行可能ファイルには特定のアドレス（絶対）へのリンクが含まれる場合があります。したがって、elfをコンパイルするときに、メモリのどのアドレスにロードされるかはすでにわかっています（リンカースクリプトを参照）。したがって、仮想メモリがないと2つのelfファイルをロードできません。しかし、仮想メモリがオンになっていても、任意のアドレスでロードできる動的ライブラリ（.soなど）があります。再配置セクションがあるため、どのアドレスからでもダウンロードできます。このセクションでは、絶対アドレッシングが使用されるすべての場所が登録されており、カーネルはそのようなelfファイルをロードするときに、これらのアドレスをペンで修正する必要があります。実際のダウンロードアドレスと目的のダウンロードアドレスの違いを追加します。

4キロバイトページでの作業を検討します。この状況では、最大4メガバイトのRAMをアドレス指定できます。それで十分です。アドレス分布マップは次のようになります：

0-1 mb：触れないでください。
1-2 mb：コードとカーネルデータ。
2-3 mb：カーネルの束。
3-4 mb：アップロードされたelfファイルのカスタムページ。

ページングが有効な場合の線形アドレス（フラットモデルから取得）は、物理アドレスと等しくありません。代わりに、アドレスは、オフセット（下位ビット）、ページテーブルのエントリのインデックス、およびページディレクトリのインデックス（上位ビット）で除算されます。各プロセスには、独自のページディレクトリがあり、それに応じてページテーブルがあります。これにより、仮想アドレス空間を整理できます。

ページディレクトリエントリは次のようになります。

struct page_directory_entry_t {
    u8 present : 1;
    u8 read_write : 1;
    u8 user_supervisor : 1;
    u8 write_through : 1;
    u8 cache_disabled : 1;
    u8 accessed : 1;
    u8 zero : 1;
    u8 page_size : 1;
    u8 ignored : 1;
    u8 available : 3;
    u32 page_table_addr : 20;
} attribute(packed);

ページテーブル要素は次のようになります。

struct page_table_entry_t {
    u8 present : 1;
    u8 read_write : 1;
    u8 user_supervisor : 1;
    u8 write_through : 1;
    u8 cache_disabled : 1;
    u8 accessed : 1;
    u8 dirty : 1;
    u8 zero : 1;
    u8 global : 1;
    u8 available : 3;
    u32 page_phys_addr : 20;
} attribute(packed);

カーネルについては、ページディレクトリとページテーブルを静的変数として記述します。確かに、ページの境界に揃える必要があります。

static struct page_directory_entry_t kpage_directory attribute(aligned(4096));
static struct page_table_entry_t kpage_table[MMU_PAGE_TABLE_ENTRIES_COUNT] attribute(aligned(4096));

簡単な方法で、物理アドレス空間全体をカーネルがアクセスできるようにします。カーネルページの特権レベルは、だれもそれらに登らないようにスーパーバイザーである必要があります。カーネルモードで実行する必要がある軽量プロセスは、カーネルと同じアドレス空間を共有します。このプロセスにより、保留中の割り込みを処理するための保留中の実行キューができます。しかし、それについては、キャラクターデバイスドライバーに関するレッスンで詳しく説明します。このトピックを検討する前に、マルチタスクをまだ実現していません。

カーネルページと対応するページテーブルのディレクトリを作成します。カーネルが初期化されると、アクティブになります。

/*
 * Api - init kernel page directory
 * Here assumed each entry addresses 4Kb
 */
extern void mmu_init()
{
    memset(&kpage_directory, 0, sizeof(struct page_directory_entry_t));

    /* set kernel page directory */
    kpage_directory.zero = 1;
    kpage_directory.accessed = 0;
    kpage_directory.available = 0;
    kpage_directory.cache_disabled = 0;
    kpage_directory.ignored = 0;
    kpage_directory.page_size = 0; /* 4KB */
    kpage_directory.present = 1; /* kernel pages always in memory */
    kpage_directory.read_write = 1; /* read & write */
    kpage_directory.user_supervisor = 1; /* kernel mode pages */
    kpage_directory.write_through = 1;
    kpage_directory.page_table_addr = (size_t)kpage_table >> 12;

    /* set kernel table */
    for (int i = 0; i < MMU_PAGE_TABLE_ENTRIES_COUNT; ++i) {
        kpage_table[i].zero = 0;
        kpage_table[i].accessed = 0;
        kpage_table[i].available = 0;
        kpage_table[i].cache_disabled = 0;
        kpage_table[i].dirty = 0;
        kpage_table[i].global = 1;
        kpage_table[i].present = 1; /* kernel pages always in memory */
        kpage_table[i].read_write = 1; /* read & write */
        kpage_table[i].user_supervisor = 1; /* kernel mode pages */
        kpage_table[i].write_through = 1;
        kpage_table[i].page_phys_addr = (i * 4096) >> 12; /* assume 4Kb pages */
    }
}

デフォルトでは、プロセスページテーブルには、カーネルページと、将来のプロセスページの空のエントリが含まれます。現在のフラグがクリアされ、物理ページアドレスが0のレコード。

/*
 * Api - Create user page table
 */
extern struct page_table_entry_t* mmu_create_user_page_table()
{
    struct page_table_entry_t* upage_table;
    upage_table = malloc_a(sizeof(struct page_table_entry_t) * MMU_PAGE_TABLE_ENTRIES_COUNT, 4096);

    /* share kernel pages */
    memcpy(upage_table, kpage_table, sizeof(struct page_table_entry_t) * MMU_KERNEL_PAGES_COUNT);
    /* fill user pages */
    for (int i = MMU_KERNEL_PAGES_COUNT; i < MMU_PAGE_TABLE_ENTRIES_COUNT; ++i) {
        struct page_table_entry_t* current;
        current = upage_table + i;

        current->zero = 0;
        current->accessed = 0;
        current->available = 0;
        current->cache_disabled = 0;
        current->dirty = 0;
        current->global = 1;
        current->present = 0; /* not present as so as there is no user pages yet */
        current->read_write = 1; /* read & write */
        current->user_supervisor = 0; /* user mode page */
        current->write_through = 1;
        current->page_phys_addr = 0; /* page is not present */
    }

    return upage_table;
}

デフォルトでは何もないため、プロセスページテーブルに新しい物理ページを追加する方法を学ぶ必要があります。これは、プログラムヘッダーに記述されているセグメントを読み込むときに、elfファイルをメモリに読み込むときに必要になります。関数はこれで私たちを助けます：

/*
 * Api - Occupy user page
 */
extern bool mmu_occupy_user_page(struct page_table_entry_t* upage_table, void* phys_addr)
{
    for (int i = MMU_KERNEL_PAGES_COUNT; i < MMU_PAGE_TABLE_ENTRIES_COUNT; ++i) {
        struct page_table_entry_t* current;
        current = upage_table + i;

        if (current->present) {
            /* page is buzy */
            continue;
        }

        current->zero = 0;
        current->accessed = 0;
        current->available = 0;
        current->cache_disabled = 0;
        current->dirty = 0;
        current->global = 1;
        current->present = 1;
        current->read_write = 1; /* read & write */
        current->user_supervisor = 0; /* user mode page */
        current->write_through = 1;
        current->page_phys_addr = (size_t)phys_addr >> 12; /* assume 4Kb pages */

        return true;
    }

    return false;
}

ページングモードは、プロセッサフ​​ラグレジスタのビットによってオン/オフされます。

/*
 * Enable paging
 * void asm_enable_paging(void *page_directory)
 */
asm_enable_paging:
    mov 4(%esp),%eax     # page_directory
    mov %eax,%cr3
    mov %cr0,%eax
    or $0x80000001,%eax  # set PE & PG bits
    mov %eax,%cr0
    ret

/*
 * Disable paging
 * void asm_disable_paging()
 */
asm_disable_paging:
    mov %eax,%cr3
    mov %cr0,%eax
    xor $0x80000000,%eax  # unset PG bit
    mov %eax,%cr0
    ret

プロセスのアドレス空間を作成する方法を学習した後、ビジー状態とフリー状態の物理ページをなんとかして管理する必要があります。これにはビットマップメカニズムがあり、ページごとに1ビットです。3メガバイトまでのページについては説明しません。ページはカーネルに属し、常にビジーであるためです。3メガバイトから4メガバイトまでのユーザーページを選択し始めます。

static u32 bitmap[MM_BITMAP_SIZE];


物理ページは、次の機能に従って割り当ておよび割り当て解除されます。実際、ページの物理アドレスでマップの目的のビットを見つけるだけで、その逆も同様です。不都合なのは、メモリセルのサイズが限られているため、バイト番号とビット番号の2つの座標を使用する必要があることです。

/*
 * Api - allocate pages
 */
extern void* mm_phys_alloc_pages(u_int count)
{
    /* find free pages */
    for (int i = 0; i < MM_DYNAMIC_PAGES_COUNT; ++i) {
        bool is_found = true;
        for (int j = 0; j < count; ++j) {
            is_found = is_found && !mm_get_bit(i + j);
        }

        if (is_found) {
            /* occupy */
            for (int j = 0; j < count; ++j) {
                assert(!mm_get_bit(i + j));
                mm_set_bit(i + j);
            }

            return (void *)mm_get_addr(i);
        }
    }

    return null;
}

/*
 * Api - free page
 */
extern bool mm_phys_free_pages(void* ptr, u_int count)
{
    size_t address = (size_t)ptr;
    assert(address >= MM_AREA_START);
    assert(address % MM_PAGE_SIZE == 0);

    /* find page */
    for (int i = 0; i < MM_DYNAMIC_PAGES_COUNT; ++i) {
        size_t addr = mm_get_addr(i);
        if (addr == address) {
            /* free pages */
            for (int j = 0; j < count; ++j) {
                assert(mm_get_bit(i + j));
                mm_clear_bit(i + j);
            }

            return true;
        }
    }

    return false;
}
