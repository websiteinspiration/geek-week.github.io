nRF52832のガラスパネル付きタッチミニスイッチ

今日の記事では、新しいプロジェクトを皆さんと共有したいと思います。今回はガラスパネル付きのタッチスイッチです。デバイスはコンパクトで、サイズは42x42mmです（標準のガラスパネルの寸法は80x80mm）。このデバイスの歴史は、はるか昔、約1年前に始まりました。



最初のオプションはatmega328マイクロコントローラーでしたが、最終的にはすべてnRF52832マイクロコントローラーで終わりました。



デバイスのセンサー部分はTTP223チップで動作します。両方のセンサーは、1つの割り込みによって処理されます。TPS610981チップのブーストコンバーターを介して、CR2477バッテリーで駆動します。Datashit。




このデバイスは、電界効果トランジスタの電源オフ回路を実装しています。ボタンを押すと、マイクロコントローラー自体が電源管理をインターセプトし、ボタンをサービスモードに使用できます（私の場合、他のデバイスとペアになっていて、電源をオフにして工場出荷時設定にリセットします（工場リセット））。

void disableNfc() {
  NRF_NFCT->TASKS_DISABLE = 1;
  NRF_NVMC->CONFIG = 1;
  NRF_UICR->NFCPINS = 0;
  NRF_NVMC->CONFIG = 0;
}

void turnOffAdc() {
  if (NRF_SAADC->ENABLE) {
    NRF_SAADC->TASKS_STOP = 1;
    while (NRF_SAADC->EVENTS_STOPPED) {}
    NRF_SAADC->ENABLE = 0;
    while (NRF_SAADC->ENABLE) {}
  }
}

void myTone(uint32_t j, uint32_t k) { //   myTone
  j = 500000 / j;                                //    j      
  k += millis();                                 //         
  while (k > millis()) {                         //  ,     
    digitalWrite(28, HIGH); delayMicroseconds(j); //    i    «1»   j
    digitalWrite(28, LOW ); delayMicroseconds(j); //    i   «0»   j
  }
}

void readBatLev() {
  //NRF5_ESB_startListening();
  wait(200);
  temp = analogRead(29);
  vcc = temp * 0.0033 * 100;
  battery = map((int)vcc, BAT_MIN, BAT_MAX, 0, 100);
  if (battery < 0) {
    battery = 0;
  }
  if (battery > 100) {
    battery = 100;
  }
  sendBatteryLevel(battery);
  wait(200);
  send(powerMsg.set(temp));
  wait(200);
  NrfRSSI = transportGetReceivingRSSI();
  NrfRSSI2 = map(NrfRSSI, -85, -40, 0, 100);
  if (NrfRSSI2 < 0) {
    NrfRSSI2 = 0;
  }
  if (NrfRSSI2 > 100) {
    NrfRSSI2 = 100;
  }
  send(msgRF52RssiReceiv.set(NrfRSSI2));
  wait(200);
}
