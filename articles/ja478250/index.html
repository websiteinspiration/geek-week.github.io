#include<bits/stdc++.h>
using namespace std;
int func2(int x, int k, int n)
{

    int ans = 0, r = 1;
    for (int i = 1; i <= k; i++)
    {
        r *= x - i + 1;
        r = r / i;
        ans = ans + r;
        if (ans >= n) break;
    }
    return ans;
}
int func(int k, int n)
{
    int low = 1, high = n;
    while (low < high)
    {
        int mid = (low + high) / 2;
        if (func2(mid, k, n) < n)
            low = mid + 1;
        else
            high = mid;
    }
    return low;
}

using namespace std;
void helper(int n, int source, vector<vector<int>>&M, vector<bool>&visited, int count, int &ans, int sum)
{
    if (count == n - 1)
    {
        ans = min(sum + M[source][0], ans);
        return;
    }

    for (int i = 1; i < n; i++)
    {
        if (!visited[i])
        {
            visited[i] = true;
            helper(n, i, M, visited, count + 1, ans, sum + M[source][i]);
            visited[i] = false;
        }
    }

}
