lint.sh
#!/bin/bash

PROJECT=$1
shift
SOURCES=$@
DESTINATIONS=""
DELIMITER=""

for src in $SOURCES
do
    DELIMITER=" --files "
    DESTINATIONS="$DESTINATIONS$DELIMITER${src}"
done

ng lint $PROJECT --tslint-config ./tslint.json $DESTINATIONS

. package.json :


  "husky": {
    "hooks": {
      "pre-commit": "lint-staged --relative"
    }
  },
  "lint-staged": {
    "*.{ts,js}": [
      "prettier --write --config .prettierr.yaml",
      "tslint --project tslint.ide_only.json --fix --force",
      "sh lint.sh partner-account",
      "git add"
    ],
    "*.{html,scss,css}": [
      "prettier --write --config .prettierr.yaml",
      "git add"
    ]
  },

に注意してくださいlint-staged --relative。--relativeそこではパラメータが必要です。コミットすると、それは私たちから始まりlint-stagedます。次に、ファイルを選択し、それらに応じてコマンドのリストを開始します。


残念ながら、これはコードレビューをキャンセルしませんが、はるかにきれいになりました。開発者にアクセス修飾子、メソッドとsvの説明について思い出させることは少なくなりましたが、彼らの仕事は同じスタイルで書かれ始めました（まあ、ほぼD）。
