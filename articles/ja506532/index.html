いばらから星へ：レーザーポインターを任意の天体に向けるデバイスを作成します

小型スポイラー
,

ハードウェア


したがって、最小限に機能するプロトタイプを作成するには、次のものが必要です。

  1. 一枚のボードまたはすべてのコンポーネントを接続できるもの
  2. ほとんどすべてのマイクロコントローラー。arduino unoを最も簡単なオプションとして選択しました
  3. GPSモジュール（日付、時刻、座標を取得します）。理論的には、代わりにリアルタイムクロックモジュールを使用できますが、手動で座標を入力する必要があり、デバイスはそのままでは機能しませんが、コールドスタートの時間が大幅に短縮されます。
  4. 2つのサーボ、またはより良いが2つのステッピングモーター
  5. レーザーポインター
  6. 別のささいなこと：はんだごて、ホットメルト接着剤、ブレッドボード、ワイヤー、ボタン、コンデンサー、そして直接の手。

組み立てるコツはありません。ベースのすべてのコンポーネントをホットメルト接着剤で固定するだけです。
サーボとレーザーの軸ができるだけ垂直になるようにしてください。これにより、ポインティングエラーが減少します。

接続


ここでも複雑なことはありません。

  • GPSはuartを介して接続されます（モジュールに何も送信する必要がないため、Rxのみが必要です）
  • サーボ-ピン10（方位軸）および11（高さ軸）
  • ボタン-2ピン
  • すべてのモジュールへの電力
  • オプション-パワーコンデンサ

私の実装の写真（19Mb）

#include <math.h>
#include <ServoSmooth.h>
#include <GyverButton.h>
#include "TinyGPS++.h"
TinyGPSPlus gps;
ServoSmooth yaw;
ServoSmooth pitch;
GButton but(2);
int yr = 0, mo = 0, d = 0, h = 0, m = 0, s = 0;
float phi = 0, lambda = 0;
float az = 0, height = 0;
int counter = 0;
float alpha = 0.0, delta = 0.0;
float sunalpha = 0, sundelta = 0;
float Coordinates[10][2] =
{
  {0, 0},
  {sunalpha * 360 / 2 / PI, sundelta * 360 / 2 / PI}, //sun
  {297.9458, 8.9233}, //altair
  {279.4083, 38.8038}, //vega
  {310.5333, 45.3538}, //deneb
  {79.55, 46.0163},//capella

int Days(int d, int m, int y)// ,        (21.03)
{
  int days = 0;
  int yearNotLeap[12] = { 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
  int yearIsLeap[12] = { 31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };

  if (((y % 4 == 0) && (y % 100 != 0)) || (y % 400 == 0))
  {
    for (int i = 0; i < m - 1; i++)
    {
      days += yearIsLeap[i];
    }
  }
  else
  {
    for (int i = 0; i < m - 1; i++)
    {
      days += yearNotLeap[i];
    }
  }
  if (m == 1) {

    return d - 81;
  }
  else
    return days + d - 81;
}
void GpsGetData()
{
  while (Serial.available() > 0)
    gps.encode(Serial.read());
  if (gps.location.isUpdated())
  {
    phi = gps.location.lat(); //   (double)
    lambda = gps.location.lng(); //    (double)
    yr = gps.date.year(); //  (2000+) (u16)
    mo = gps.date.month(); //  (1-12) (u8)
    d = gps.date.day(); //  (1-31) (u8)
    h = gps.time.hour(); //  (0-23) (u8)
    m = gps.time.minute(); //  (0-59) (u8)
    s = gps.time.second(); //  (0-59) (u8)
  }
}
void CalculateParams()
{
  float _time = (h + (float)m / 60 + (float)s / 3600) * 360.0 / 24.0;//   
  float sunlambda = (float)Days(d, mo, yr) * 360 / 365; //      
  sundelta = asin(0.398749 * sin(sunlambda * 2 * PI / 360));//    
  sunalpha = asin(tan(sundelta) / 0.434812); //      
  float tS = _time + lambda - 180 + sunalpha * 360 / (2 * PI) - alpha; //     
  height = 360 / 2 / PI * asin(sin(phi * 2 * PI / 360) * sin(delta * 2 * PI / 360) + cos(phi * 2 * PI / 360) * cos(delta * 2 * PI / 360) * cos(tS * 2 * PI / 360));// 
  az = 360 / 2 / PI * asin(sin(tS * 2 * PI / 360) * cos(delta * 2 * PI / 360) / cos(height * 2 * PI / 360)); //   (   )
  if (tS > 90 || tS < -90)
  {
    az = 180 - az;
  }
 // if (az > 180 && az < 270)
  //  az = az - 360;
}

void setup() {
  Serial.begin(9600);
  yaw.attach(10, 90);
  pitch.attach(11, 180);
  yaw.setSpeed(20);
  yaw.setAccel(0.1);
  pitch.setSpeed(20);
  pitch.setAccel(0.1);
  yaw.setAutoDetach(false);
  pitch.setAutoDetach(false);
  pinMode(LED_BUILTIN, OUTPUT);
}

void loop() {
  but.tick();
  yaw.tick();
  pitch.tick();
  if (but.isSingle())
  {
    counter += 1;
    if (counter == 9)
      counter = 0;
  }
  alpha = Coordinates[counter][0];
  delta = Coordinates[counter][1];
  GpsGetData();
  CalculateParams();
  digitalWrite(LED_BUILTIN, gps.location.isValid());
  if (az > 90 && az < 180)
  {
    yaw.setTargetDeg(270 - az);
    pitch.setTargetDeg(height);
  }
  if (az < -90 && az > -180)
  {
    yaw.setTargetDeg(-90 - az);
    pitch.setTargetDeg(height);
  }
  if (az < 90 && az > -90)
  {
    yaw.setTargetDeg(90.0 - az);
    pitch.setTargetDeg(180 - height);
  }
}

天文学レッスン


アストロテルミンなどについて

...そして、たった1つの瞬間があります-関数CalculateParams（）
この関数が行うべきこと：赤道系の星の座標（赤経と赤緯）、観測者の時間と座標を取り、オブジェクトの高さと方位角を与えます。つまり、本質的に、星の座標を赤道系（星が動かない）から水平（日中に星が動く）に変換します。

これは、球形三角法の公式と真空中の球体天文学を使用して実現されます。 アルゴリズムは次のとおりです。



  1. 太陽の黄道経度を計算する
  2. 太陽の赤緯を計算する
  3. 直接日の出を計算する
  4. 星の時角を計算する
  5. 高度と方位を計算する

実装方法は次のとおりです。

void CalculateParams()
{
  float _time = (h + (float)m / 60 + (float)s / 3600) * 360.0 / 24.0;//   
  float sunlambda = (float)Days(d, mo, yr) * 360 / 365; //

  az = 360 / 2 / PI * asin(sin(tS * 2 * PI / 360) * cos(delta * 2 * PI / 360) / cos(height * 2 * PI / 360)); //   (   )

  {
    az = 180 - az;
  }
}

エラー推定


エラーに対する各因子の寄与を評価してみましょう（不正確さへの寄与でソート）

  1. 手の湾曲-これは私にとって最も重要な要素です
  2. 不正確な地平線の位置
  3. 方位角ゼロの不正確な位置（天文学者は南から時計回りに方位角を数えることを思い出します）

    また、時間の方程式は考慮していません）。 -このため、小さな漏れ。
