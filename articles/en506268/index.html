Generative art: creating triangles after 3 hours of learning p5.js

I have little experience using a computer for creativity and art. When I started learning p5.js, I was inspired by geometric patterns and decided to write code to create something cool.

After about an hour and a half, I managed to get randomly formed triangles of various colors.


Random Triangles

As part of the lesson, we were asked not to stop until we get what we are really happy with - to play with shape, size and color.

During a long trip, I mentally scrolled all kinds of ideas in my head, and decided to make sure that all these triangles were connected to each other and not duplicated. Before starting the project, we discussed this concept, but came to the conclusion: it is too difficult to do, given that we only study p5.js. for only 3 hours. I looked at the libraries that implemented this concept, but decided to create my own solution for practice. It‚Äôs hard for me to draw while traveling, but as soon as I could stop, I sketched it: The


sketch I made after a long trip about how I could approach this concept

What if instead of random coordinates to create triangles I could :
  • set the grid to canvas (plane)
  • randomly arrange points in this grid
  • save coordinates in array
  • then systematically go through these points and start using them to create new triangles

Everything in my head looked as if getting rid of the randomness when forming a triangle could work.
Now, let's see how far I have come in this direction before I realize how stupid it was to think that it would work.

Meshing.

function setup() {
  createCanvas( windowWidth, windowHeight )
  colorMode( HSB )
  noLoop()
  noStroke()
}


function draw() {
  background( 20, 20, 20)
  
  // Setup a grid so i can figure out the uppper and lower bounds
  // of where the points should go
  
  var columns = 5
  var rows = 5
  var columnWidth = windowWidth / columns
  var columnHeight = windowHeight / rows
  fill( 30, 30, 30)
  stroke( 90, 90, 90 )
  rect( 0, 0, columnWidth, columnHeight )

  // Create a row

  for ( i = 0; i < rows; i++ ){

    // Create a column

    for ( j = 0; j < columns; j++ ){

      rect( j * columnWidth, i * columnHeight, columnWidth, columnHeight )
    }
  }
}

Final code 
function setup() {
  var canvas = createCanvas( 1600, 1600 )
  canvas.parent("canvasArea");
  colorMode( HSB )
  noLoop()
  noStroke()
}

function draw() {
  background( 197, 31, 65 )

  // Setup a grid so i can figure out the uppper and lower bounds
  // of where the points should go
  var columns = 20
  var rows = 20

  var columnWidth = 1600 / columns
  var columnHeight = 1600 / rows

  // Show the grid
  fill( 18, 11, 18 )
  //stroke( 90, 90, 90 )
  //rect( 0, 0, columnWidth, columnHeight )

  var allCoordinates = [];
  // Create a row
  for ( i = 0; i < rows; i++ ){
    // Create a column
    for ( j = 0; j < columns ; j++ ){
      rect( j * columnWidth, i * columnHeight, columnWidth, columnHeight )
      //create my point
      var pointPositionX = random( j * columnWidth + columnWidth, j * columnWidth )
      var pointPositionY = random( i * columnHeight + columnHeight, i * columnHeight )

      allCoordinates.push( [pointPositionX, pointPositionY] )
      circle( pointPositionX, pointPositionY, 10 )
    }
  }
  var whichRow = 1
  var l = 1
  while ( l < allCoordinates.length ){

    //firstTriangle
    if (whichRow == 1 || whichRow == rows){
      //don't do anything on the first and last row (thse need just one triangle)
    }
    else{
      if ( (l + 1) % rows == 0 || l == 0){
        console.log("it's the last item, do nothing")
      }
      else{
        //Triangle 1
        var colorChoices = [
          [45, 15, 75],
          [22, 42, 95],
          [5, 55, 75],
          [0, 61, 95]
        ]

        var randomColorChoice = Math.floor(random(0,4))
        fill( colorChoices[randomColorChoice][0], colorChoices[randomColorChoice][1], Math.floor(random(50,100)) )
        //stroke( 200, 90, 90 )

        var currentPoint = l
        var nextPoint = l + 1
        var bottomPoint = l + rows
        var firstPointX = allCoordinates[currentPoint][0]
        var firstPointY = allCoordinates[currentPoint][1]
        var secondPointX = allCoordinates[nextPoint][0]
        var secondPointY = allCoordinates[nextPoint][1]
        var thirdPointX = allCoordinates[bottomPoint][0]
        var thirdPointY = allCoordinates[bottomPoint][1]
        triangle(firstPointX, firstPointY, secondPointX, secondPointY, thirdPointX, thirdPointY)

        //Triangle 2
        randomColorChoice = Math.floor(random(0,4))
        fill( colorChoices[randomColorChoice][0], colorChoices[randomColorChoice][1], Math.floor(random(50,100)) )
        //stroke( 200, 90, 90 )

        var currentPoint = l
        var nextPoint = l - rows
        var bottomPoint = l + 1
        var firstPointX = allCoordinates[currentPoint][0]
        var firstPointY = allCoordinates[currentPoint][1]
        var secondPointX = allCoordinates[nextPoint][0]
        var secondPointY = allCoordinates[nextPoint][1]
        var thirdPointX = allCoordinates[bottomPoint][0]
        var thirdPointY = allCoordinates[bottomPoint][1]
        triangle( firstPointX, firstPointY, secondPointX, secondPointY, thirdPointX, thirdPointY )
      }
    }
    l++
    if ( l % rows == 0 ){ whichRow++ }
  }
}
