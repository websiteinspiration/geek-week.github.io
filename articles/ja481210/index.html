public boolean isPalindrome(String s) {
    int first = 0;
    int last = s.length() - 1;
    while (first < last) {
        if (s.charAt(first) != s.charAt(last)) {
            return false;
        }
        first++;
        last--;
    }
    return true;
}

public boolean isPalindrome(String s) {
    int len = s.length;
    for (int i = 0; i < len / 2; i++) {
        if ( s.charAt(i) != s.charAt(len - 1 - i)) {
            return false;
        }
    }
    return true;
}
