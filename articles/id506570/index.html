import cats.implicits._
import cats.Eq

implicit class DivisionSyntax[T](val value: T) extends AnyVal {
  def divisibleBy(n: T)(implicit I: Integral[T], ev: Eq[T]): Boolean = {
    import I._
    (value % n) === zero
  }
  def divisibleByInt(n: Int)(implicit I: Integral[T], ev: Eq[T]): Boolean =
    divisibleBy(I.fromInt(n))
}

10 divisibleBy 5 // => true
BigInt(10) divisibleBy BigInt(3) // => false
BigInt(10) divisibleByInt 3 // => false

Di sini digunakan:


  • ketik kelas " Integral" yang membutuhkan tipe " T" untuk dapat menghitung sisa divisi dan memiliki nilai " zero"
  • tipe " Eq" yang membutuhkan tipe " T" untuk dapat membandingkan elemen - elemennya (operator " ===" adalah sintaksnya)
  • "T" extension methods & value classes, - ( dotty, )
