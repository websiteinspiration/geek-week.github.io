持続する場合：
if isinstance(self.custom_handlers, property):
if self.custom_handlers and e.__class__ in self.custom_handlers:
if e.__class__ not in self.exclude:


これは制限ではありません。したがって、ifsの一部を削除し、何かを__init__に転送しました。一度呼び出される場所に。具体的には、コード内のプロパティのチェックは1回呼び出す必要があります。デコレータがメソッドに適用され、メソッドに割り当てられます。したがって、クラスのプロパティは変更されません。したがって、常にプロパティをチェックする必要はありません。

別のポイントはifです。プロファイラーは、そのようなそれぞれに個別の呼び出しがあることを示したため、すべてのハンドラーを1つのディクテーションに組み合わせることにしました。これにより、単純に使用する代わりに、一般的にifを回避することができました。
self.handlers.get(e.__class__, Exception)(e)


そのため、self.handlersにはdictがあり、デフォルトとして残りの例外を発生させる関数が含まれています。

もちろん、ラッパーには特別な注意が必要です。これは、デコレータが呼び出されるたびに呼び出される関数と同じです。それら。ここでは、不必要なチェックやあらゆる種類のロードを可能な限り最大化して__init__または__call__に配置しないことをお勧めします。これが以前のラッパーでした：
def wrapper(self, *args, **kwargs):
        if self.custom_handlers:
            if isinstance(self.custom_handlers, property):
                self.custom_handlers = self.custom_handlers.__get__(self, self.__class__)

        if asyncio.iscoroutinefunction(self.func):
            return self._coroutine_exception_handler(*args, **kwargs)
        else:
            return self._sync_exception_handler(*args, **kwargs)


チェックの数は屋根を通過します。これはすべて、デコレータへのすべての呼び出しで呼び出されます。したがって、ラッパーは次のようになります。
    def __call__(self, func):
        self.func = func

        if iscoroutinefunction(self.func):
            def wrapper(*args, **kwargs):
                return self._coroutine_exception_handler(*args, **kwargs)
        else:
            def wrapper(*args, **kwargs):
                return self._sync_exception_handler(*args, **kwargs)

        return wrapper


__call__は1回呼び出されます。__call__内では、関数の非同期の度合いに応じて、関数自体またはコルーチンを返します。また、asyncio.iscoroutinefunctionが追加の呼び出しを行うことにも注意したいので、inspect.iscoroutinefunctionに切り替えました。実際、asyncioと検査のベンチ（cProfile）：

完全なコード：
from inspect import iscoroutinefunction

from asyncio import QueueEmpty, QueueFull
from concurrent.futures import TimeoutError


class ProcessException(object):

    __slots__ = ('func', 'handlers')

    def __init__(self, custom_handlers=None):
        self.func = None

        if isinstance(custom_handlers, property):
            custom_handlers = custom_handlers.__get__(self, self.__class__)

        def raise_exception(e: Exception):
            raise e

        exclude = {
            QueueEmpty: lambda e: None,
            QueueFull: lambda e: None,
            TimeoutError: lambda e: None
        }

        self.handlers = {
            **exclude,
            **(custom_handlers or {}),
            Exception: raise_exception
        }

    def __call__(self, func):
        self.func = func

        if iscoroutinefunction(self.func):
            def wrapper(*args, **kwargs):
                return self._coroutine_exception_handler(*args, **kwargs)
        else:
            def wrapper(*args, **kwargs):
                return self._sync_exception_handler(*args, **kwargs)

        return wrapper

    async def _coroutine_exception_handler(self, *args, **kwargs):
        try:
            return await self.func(*args, **kwargs)
        except Exception as e:
            return self.handlers.get(e.__class__, Exception)(e)

    def _sync_exception_handler(self, *args, **kwargs):
        try:
            return self.func(*args, **kwargs)
        except Exception as e:
            return self.handlers.get(e.__class__, Exception)(e)

PSはHabrのユーザーのコメントのおかげで、さらに加速することができました。それが起こったのです。
from inspect import iscoroutinefunction

from asyncio import QueueEmpty, QueueFull
from concurrent.futures import TimeoutError


class ProcessException(object):

    __slots__ = ('handlers',)

    def __init__(self, custom_handlers=None):
        if isinstance(custom_handlers, property):
            custom_handlers = custom_handlers.__get__(self, self.__class__)

        raise_exception = ProcessException.raise_exception

        exclude = {
            QueueEmpty: lambda e: None,
            QueueFull: lambda e: None,
            TimeoutError: lambda e: None
        }

        self.handlers = {
            **exclude,
            **(custom_handlers or {}),
            Exception: raise_exception
        }

    def __call__(self, func):
        handlers = self.handlers

        if iscoroutinefunction(func):
            async def wrapper(*args, **kwargs):
                try:
                    return await func(*args, **kwargs)
                except Exception as e:
                    return handlers.get(e.__class__, handlers[Exception])(e)
        else:
            def wrapper(*args, **kwargs):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    return handlers.get(e.__class__, handlers[Exception])(e)

        return wrapper

    @staticmethod
    def raise_exception(e: Exception):
        raise e




timeit.timeit('divide(1, 0)', number=100000, setup='from __main__ import divide')
0.13714907199755544


平均で0.03加速します。感謝コスティアンティン そして イングビー。

PS更新！コメントからのコメントに基づいて、コードをさらに最適化しましたワングレイワンホワイト そして リセット。self.funcをfuncだけに置き換え、self.handlersを変数にした。実行はさらに加速され、特にゼロあたりの繰り返しが多い場合に顕著になります。私はtimeitを引用します：
