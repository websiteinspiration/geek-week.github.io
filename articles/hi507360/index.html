        }
        //  ,    
        const Animation = this.animations.find(item => item.name === animation);

    isPlay(animation) {
        return this.animations.find(item => item.name === animation).play
    }
    createAnimation(name, callback) {
        this.animations.push({
            play: false,
            name,
            callback,
        })
    }
}
