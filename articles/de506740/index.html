So verwenden Sie Microsoft SQL für Berichte in Power BI. Am Beispiel von Mindbox

Wenn Sie verstehen möchten, wie effektiv Werbeinvestitionen sind, müssen Sie alles messen: Briefe, Besuche, Bestellungen, Einnahmen. Die Geschwindigkeit, mit der diese Metriken abgerufen werden, ist ebenfalls wichtig. Aus Gründen der Übersichtlichkeit können Sie in Microsoft Power BI schöne und verständliche Berichte erstellen. Um den Empfang von Daten für Berichte zu vereinfachen, können diese in ihrer eigenen Datenbank gespeichert werden. Dies ist praktisch, insbesondere wenn viele Informationen vorhanden sind und es schwierig ist, diese direkt von Online-Diensten abzurufen. Wir speichern Daten in Microsoft SQL, Sie können jedoch auch einen anderen Datenbankserver verwenden.


Probleme beim Erstellen von Berichten mit Daten aus ESP


Moderne Mail-Plattformen (Email Sending Platform oder ESP) sammeln Statistiken zu Mailings: Senden, Zustellen, Öffnen, Klicken. Mit fortschrittlicheren Plattformen wie Mindbox können Sie hier Bestellungen und Einnahmen hinzufügen.


Darüber hinaus können und sollten Daten mit Google Analytics und / oder Yandex.Metrics erfasst werden.


Daten werden an einem Ort gespeichert, auf den über die E-Mail-Plattform-APIs zugegriffen werden kann. Es ist jedoch schwierig, Berichte zu diesen Rohdaten zu erstellen, da:


  1. Daten können in verschiedenen Formaten gespeichert werden, was oft unpraktisch ist.
  2. Der Zugriff auf Daten kann aufgrund komplexer Autorisierungsschemata und eines komplexen Datenerfassungsprotokolls schwierig sein.
  3. Daten können nur in den letzten Monaten im ESP gespeichert werden, ältere Plattformdaten werden normalerweise gelöscht.

function mbxChunk(operation, cb) {
console.log('New Chunk '+operation);
var sqlreq = new sql.Request();
//  mindboxId  ,  ,   ID  
sqlreq.query("SELECT MAX(mindboxId) AS last FROM "+config.mssql.database+".dbo."+operation+";", function(err, maxres) {
if (err || !maxres.recordset[0]) {
console.log('MsSQL SELECT MAX error: ', err);
cb(err);
return;
}
var chunk = parseInt(maxres.recordset[0].last || 0) + 1;

// ,   ID=chunk
mbxCall(operation, chunk, function(err, xml) {
console.log(operation + ' from #' + chunk.toString() + ' started');
if (err) {
console.log(operation + ' from #' + chunk.toString() + ': Mindbox API request error: ', err);
cb(err);
}
parseString(xml, function(err, res) {
if (err) {
console.log(operation + ' from #' + chunk.toString() + ': Mindbox returns invalid XML, parse error: ', err);
console.log(xml);
cb(err);
}
if (res && res.result && res.result.status == 'Success') {
var rows = res.result.customerActions ? res.result.customerActions[0].customerAction : [];
if (rows.length > 0) {
mbxHandle(operation, rows, function(err, res) {
console.log(operation + ' from #' + chunk.toString() + ' + '+rows.length.toString()+' done!');
mbxChunkCopy(operation, cb);
});
}
else {
console.log(operation + ' - data is over!');
cb();
}
}
else {
console.log(operation + ' from #' + chunk.toString() + ': Mindbox returns unsuccessfull result: ', res);
cb(err);
}
});
})
})
}

function mbxChunkCopy(operation, cb) {
mbxChunk(operation, cb);
}

function mbxCall(operation, startId, cb) {
var request = require("request");
var options = {
method: 'POST',
url: 'https://api.mindbox.ru/v3/operations/sync',
qs: {
endpointId: 'fskuniversalpoint', operation: operation
},
headers: {
"Cache-Control": "no-cache",
"Content-Type": "application/xml",
"Accept": "application/xml",
"Authorization": 'Mindbox secretKey="'+config.mindbox_api_key+'"'
},
body: '<operation>\r\n  <page>\r\n    <firstmindboxid>' + startId.toString() + '</firstmindboxid>\r\n    <pagenumber>1</pagenumber>\r\n    <itemsperpage>1000</itemsperpage>\r\n  </page>\r\n</operation>'
};

request(options, function (error, response, body) {
cb(error, body);
});
}
