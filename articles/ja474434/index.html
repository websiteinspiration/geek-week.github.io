脚本

from telegram import Bot
from pythonping import ping
import time
import yaml
from libs.host import address

def init():

    """
          ,
     userid      set_hosts
    """

    global bot, userid

    with open('/usr/src/app/config.yaml') as f:
        try:
            docs = yaml.load_all(f, Loader=yaml.FullLoader)

            for doc in docs:
                for k, v in doc.items():
                    if k == "botid":
                        bot = Bot(v)
                    elif k == "userid":
                        userid = v
                    elif k == "hosts":
                        set_hosts(v)

        except yaml.YAMLError as exc:
            print(exc)

def set_hosts(hosts):

    """
            
    """

    global hosts_list
    hosts_list = []

    for item in hosts:
        ac = item.split(":")
        hosts_list.append(address(ac[0], ac[1]))

def send_message(message):

    bot.send_message(userid, message, parse_mode='HTML', disable_web_page_preview=True)

def ping_url(url):

    """
     . Response list -    ping.   
      .    ,    .
    """

    i = 0;

    try:
        response_list = ping(url)

        for response in response_list:

            return False

Dockerイメージをまとめる


この記事で 書いたdockerとdocker-composeのインストール方法。

dockerfileを作成する必要があります。コードフォルダーの横に、ボットのソースコードを配置します。

FROM python:buster
RUN pip3 install python-telegram-bot pythonping pyyaml
WORKDIR /usr/src/app
COPY ./code/. .
ENTRYPOINT ["python"]
CMD ["main.py"]

公式のPythonリポジトリからdockerイメージを接続します。ライブラリをインストールします。現在のフォルダーを/ usr / src / appに設定します。このフォルダーにコードをコピーします。起動時に呼び出されるユーティリティはpythonになります。CMDは実行するパラメーターを渡します-main.py。

次に、イメージを収集してリポジトリに配置する必要があります

docker build -t alsoeast/pingbot:1.0 .
docker push alsoeast/pingbot:1.0

Dockerイメージを実行するには、docker-composeを使用します。

version: '3'

networks:
  default:
    external:
      name: network

services:

  pingbot:
    container_name: pingbot
    image: alsoeast/pingbot:1.0
    restart: always
    volumes:
      - ./config.yaml:/usr/src/app/config.yaml

イメージはリポジトリから取得され、スクリプトが引数を受け取ることができるようにconfig.yamlファイルがマウントされます。走るだけです

docker-compose up -d

誰かがそれを必要とする場合、ボットはDockerハブで利用できます。
