            }
        });
        })(scope);


また、リソースhttps://scadavis.io/learn.htmlの例とテンプレートノードの機能を使用して、SCADAオブジェクトのさまざまなイベントを処理し、次のように他のノードに渡すことができます。 
(function(scope) {
substationsynoptic.on("click", function (event, tag) {
    var v = substationsynoptic.getValue(tag);
    if (event.currentTarget.id === "TAPUP")
        substationsynoptic.setValue(tag, v + 1, false, false);
    else
        if (event.currentTarget.id === "TAPDOWN")
            substationsynoptic.setValue(tag, v - 1, false, false);
    if (event.currentTarget.id === "XCBROPEN")
        substationsynoptic.setValue(tag, false, false, false);
    else
        if (event.currentTarget.id === "XCBRCLOSE")
            substationsynoptic.setValue(tag, true, false, false);
    if (v === true)
        substationsynoptic.setValue(tag, false, false, false);
    else
        if (v === false)
            substationsynoptic.setValue(tag, true, false, false);
    scope.send( {'payload':{ 'tag': tag, 'value': v }} );
});
})(scope);




グラフィックエディター



SCADAシステム用のビジュアルページを構築するためのグラフィックエディターについて一言。上記のように、オフサイトではSCADAvis.io Synoptic Editorを購入することをお勧めします。単独で、再び-これは唯一のオプションではありません。説明自体からわかるように、エディターはInkscape SVGエディターに基づいています。実際には、これは、svgファイルのさまざまなグラフィックコンポーネントにタグ（変数）およびその他のプロパティを割り当てることができるプラグインを備えたInkscapeです。それら。 scadavisのコンテキストでページを簡単に表示するには、便利なエディタで編集できるsvgファイルがあれば十分です。しかし、それだけではありません。 ECAVA INTEGRAXOR www.integraxor.comのエディターには多くの類似点があります。Inkscape 0.91 SAGE 4.16は無料で配布され（サイトに登録する必要があるだけです）、Synoptic Editor。結局のところ、Inkscape 0.91 SAGE 4.16を使用してScadavisのsvgファイルを作成および編集することが実際に可能です。この違いは、スクリプトアニメーションや、SAGEエディターのオブジェクトプロパティにないその他の特定の事項を設計するときに見られます。SCADAvis.io Synoptic Editorの







説明によると、包含条件に従ってオブジェクトの色を割り当てるには、[制限]列に数値2を指定する必要があります。



  • 0-初期化されていない状態
