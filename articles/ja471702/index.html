  const [visible, setVisible] = React.useState(true);

    <React.Fragment>

      <div className={visible ? "visible" : "invisible"}>Hello</div>
    </React.Fragment>
  );

  animation: fadeIn 0.5s linear forwards;

  0% {
    opacity: 1;

    opacity: 0;
    display: none;
  }
}

  0% {
    opacity: 0;
  }
  100% {
    opacity: 1;
  }
