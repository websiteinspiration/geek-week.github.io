Um Daten zu empfangen und zu aktualisieren, können Sie zwei Funktionen für jedes der Felder schreiben, aber ich schlage vor, dies anders zu machen. Die Struktur solcher Funktionen ist bis auf den Spaltennamen identisch, daher übergeben wir das Feld an die Funktion. 







    #     
    def get_field(self, user_id, field):
        self.connect_to_db()
        request = f"SELECT {field} FROM user WHERE id=?"
        result = self.cursor.execute(request, (user_id,)).fetchone()
        self.close()
        return result[0]

    #   
    def set_field(self, user_id, field, value):
        self.connect_to_db()
        request = f"UPDATE user SET {field}=? WHERE id=?"
        self.cursor.execute(request, (value, user_id))
        self.connect.commit()
        self.close()
