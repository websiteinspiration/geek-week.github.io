ビットトーンで学ぶ

むかしむかし、非常に優れた有用なOpenSSLパッケージの「方法」を選択し、「方法」を研究しているときに、いつものように、1つの単純なアイデアが突然現れ、そのようなすべての非常に予想外のアイデアが忘れ去られました。

しかし、乾燥した残留物が残っていました。OpenSSLで、BN_ULONGと大きなプログラムを乗算して、平方根をビットごとに抽出する小さなプログラムでエラーが見つかりました。エラーメッセージはバグ追跡に移動して修正されました（OpenSSLでエラーを毎日見つけるのではなく、私はこの機会に私の過度の感情をお詫びします）。しかし、平方根を見つけるための非常に小さなプログラムは、ビットごとのモジュロ2 ^ nです。ここで、nはビット数\ビット深度とあなたの注意を喚起します。

列でビットごとに2つの数値を乗算するアルゴリズムを検討する場合、因子のiビットの値によって、転送を知ることで、結果のビットをすばやく決定できます-結果を知り、因子のビットの分布を仮定すると、これらのビットをすばやく計算できます。いつものように、良いことは何も起こりませんでしたが、要因の平等、すなわち平方根を抽出すると、対応する係数のビット（現在は1です）をすばやく取得できます。

明らかに、n / 2ビットを受信すると、それがルートかどうかが明らかになるか、2の累乗を法とすると、得られた2つの数値がルートになります。

小さな仮定-私たちは奇数から根を抽出すると信じています。最後のビットがゼロとそれらの偶数である場合、それらを破棄できます。

もちろん、主要な部分はコードです。

検証とテストのために、同じOpenSSLを使用して大きな素数を取得しました。その後、BN_mulを使用して数値がそれ自身で乗算され、square_root関数では、ルートが2回計算されます。因数11または01の最後のビットはこのアルゴリズムでは区別できないため、計算は2回実行されます。数値を保存するには、OpenSSLパッケージのBIGNUMまたは同じ長さのビットセットを使用します。

だからコード

#include <bitset>
#include <stdio.h>
#include <openssl/rsa.h>
#include <openssl/bn.h>

using namespace std;
#define DIM 512

int ret = 0;
RSA *r = NULL;
BIGNUM *bne = NULL;
BIO *bp_public = NULL, *bp_private = NULL;

int bits = DIM;
unsigned long e = RSA_F4;

bool generate_key() {
	r = RSA_new();
	ret = RSA_generate_key_ex(r, bits, bne, NULL);
	return (ret );
}

bitset<DIM> square_root(bitset<DIM> key, int prim_1) {
	bitset<DIM> prim;
	int carry = prim_1;
	int i, j, ie;
	prim[0] = 1;
	prim[1] = prim_1;

	ie = DIM / 2;
	for (i = 2; i < ie; i++) {
		for (j = 1; j < i; j++)
			carry = carry + (int) (prim[j] * prim[i - j]);

		bool i1 = i & 1;
		int q2 = (carry / 2) & 1;
		int key1 = (int) key[i + 1];
		if (!i1 && q2 != key1)
			prim[i] = 1;
		if (!i1 && q2 == key1)
			prim[i] = 0;
		if (i1 && q2 == key1)
			prim[i] = prim[(i + 1) / 2];
		if (i1 && q2 != key1)
			prim[i] = 1 - (int) prim[(i + 1) / 2];

		carry += 2 * (int) prim[i];
		carry /= 2;
	}
	return prim;
}

int main() {
	bitset<DIM> bit0_sqrt(0);
	bitset<DIM> bit1_sqrt(0);
	bitset<DIM> bit_p2(1);
	bne = BN_new();
	ret = BN_set_word(bne, e);
	char *pc, *qc, *rc;

	if (generate_key() == 1) {
		BIGNUM *rez = NULL;
		BIGNUM *p2 = NULL;
		BIGNUM *tmp = NULL;
		const BIGNUM *n = NULL;
		const BIGNUM *e = NULL;
		const BIGNUM *d = NULL;
		const BIGNUM *p = NULL;
		const BIGNUM *q = NULL;

		RSA_get0_key(r, &n, &e, &d);
		RSA_get0_factors(r, &p, &q);

		pc = BN_bn2hex(p);
		printf("  p = 0x%s\n", pc);

		p2 = BN_new();
		tmp = BN_new();
		BN_CTX *ctx;
		ctx = BN_CTX_new();
		BN_CTX_start(ctx);
		BN_mul(p2, p, p, ctx);
		pc = BN_bn2hex(p2);
		printf("p^2 = 0x%s\n", pc);

		bit_p2 = 0;
		for (int i = 0; i < BN_num_bits(p2); i++)
			if (BN_is_bit_set(p2, i))
				bit_p2[i] = 1;

		rez = BN_new();
		bit0_sqrt = square_root(bit_p2, 0);
		for (int i = 0; i < DIM; i++)
			if (bit0_sqrt[i])
				BN_set_bit(rez, i);
			else
				BN_clear_bit(rez, i);
		rc = BN_bn2hex(rez);
		printf(" 0r = 0x%s\n", rc);

		BN_sqr(tmp, rez, ctx);
		rc = BN_bn2hex(tmp);
		printf(" p2 = 0x%s\n", rc);

		bit1_sqrt = square_root(bit_p2, 1);
		for (int i = 0; i < DIM; i++)
			if (bit1_sqrt[i])
				BN_set_bit(rez, i);
			else
				BN_clear_bit(rez, i);
		rc = BN_bn2hex(rez);
		printf(" 1r = 0x%s\n", rc);

		BN_sqr(tmp, rez, ctx);
		rc = BN_bn2hex(tmp);
		printf(" p2 = 0x%s\n", rc);
	}

	BIO_free_all(bp_public);
	BIO_free_all(bp_private);
	RSA_free(r);
	BN_free(bne);
}

コードはシンプルで、1つにまとめて謝罪します。ただし、主要な部分のほとんどはOpenSSLユーティリティの呼び出しですが、ルート計算プログラムでは、ビットの評価と計算は10行しかないため、DIMがディメンションであることは明らかです。

結論は、数値と0rおよび1rの両方が、D2の累乗に対する2を法とするp2の平方根であることを明確に示しています。それら。0r * 0r == p2 mod 2 ^ DIMおよび1r * 1r == p2 mod 2 ^ DIM。さらに、1r * 1r == p2。
