lm_preds <- function(
	sd, by, n
)
{
	
	if(
		n < 100 | 
		!by[['a']] %in% head(letters, 4)
	   )
	{
		
		res <-
			list(
				low = NA
				, mean = NA
				, high = NA
				, coefs = NA
			)
		
	} else {

		lmm <- 
			lm(
				d ~ c + b
				, data = sd
			)
		
		preds <- 
			stats::predict.lm(
				lmm
				, sd
				, interval = "prediction"
				)
		
		res <-
			list(
				low = preds[, 2]
				, mean = preds[, 1]
				, high = preds[, 3]
				, coefs = coefficients(lmm)
			)
	}

	res
	
}

lm_preds <- function(
	sd, by, n
)
{
	
	if(
		n < 100 |

	   )
	{
		
		res <-
			list(
				low = NA
				, mean = NA

			)

		lmm <- 
			lm(
				d ~ c + b

			)

			stats::predict.lm(

			list(
				low = preds[, 2]

			)
	}

	res
	
}
