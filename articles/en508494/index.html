1. Frequent reference to the same elements



Sometimes when writing code, you have to refer to the same elements over and over again. When working with the DOM, for example, such elements are document and document.body.



I can offer three options for solving this problem: 



    //  
function foo() {
    const D = document
    const B = document.body

    const div = D.createElement('div')
    B.append(div)
    const p = D.createElement('p')
    p.textContent = 'Lorem ipsum dolor sit amet...'
    div.append(p)
    console.log(div)
    B.removeChild(div)
}
foo()

//  
function bar(D, B) {
    const div = D.createElement('div')
    B.append(div)
    const p = D.createElement('p')
    p.textContent = 'Lorem ipsum dolor sit amet...'
    div.append(p)
    console.log(div)
    B.removeChild(div)

bar(document, document.body)

;((D, B) => {
    const div = D.createElement('div')
    B.append(div)
    const p = D.createElement('p')
    p.textContent = 'Lorem ipsum dolor sit amet...'
    div.append(p)
    console.log(div)
    B.removeChild(div)

2. Generator



A generator is a special function that works like an iterator factory. An object is an iterator if it can access the elements of the collection one at a time, while tracking its current position inside this sequence. Generators allow you to define an algorithm for enumerating elements of a collection using a single function that supports its own state. 



//  1
function* takeItem(arr) {
    for (let i = 0; i < arr.length; i++) {
        yield arr[i]
    }
}

const arr = ['foo', 'bar', 'baz', 'qux']

const generator = takeItem(arr)

const timer = setInterval(() => {
        const item = generator.next()
        item.done
            ? clearInterval(timer)
            : console.log(item.value)
    }, 1000)

//  2
async function* range(start, end) {
    for (let i = start; i <= end; i++) {
        yield Promise.resolve(i)
    }
}

;(async () => {
    const generator = range(1, 4)
    for await (const item of generator) {
        console.log(item)
    }
})()


3. Async / await + fetch



Async / await is an alternative to promises, allowing for synchronous execution of asynchronous functions. In turn, fetch is an alternative to XMLHttpRequest, representing an interface for retrieving resources (including over the network).

4. For await



The expression for await ... of creates a loop passing through asynchronous iterable objects, as well as synchronous iterable entities. It calls a custom iterative hook with instructions that must be executed for the value of each individual property of the object. 



const delayedPromise = (id, ms) => new Promise(resolve => {
    const timer = setTimeout(() => {
        resolve(id)
        clearTimeout(timer)
    }, ms)
})

const promises = [
    delayedPromise(1, 1000),
    delayedPromise(2, 2000),
    delayedPromise(3, 3000)
]

//  
async function oldStyle() {
    for (const promise of await Promise.all(promises)) {
        console.log(promise)
    }
}
oldStyle() //    3

    for await (const promise of promises) {

5. Proxy



Proxies are used to declare the extended semantics of JS objects. The standard semantics are implemented in the JS engine, which is usually written in a low-level programming language, such as C ++. Proxies allow you to determine the behavior of an object using JS. In other words, they are a metaprogramming tool. 



const person = {
    firstname: 'Harry',
    lastname: 'Heman',
    city: 'Mountain View',
    company: 'Google'
}

const proxy = new Proxy(person, {
    get(target, property) {
        if (!(property in target)) {
            return property
                .split('_')
                .map(p => target[p])
                .sort()
                .join(' ')
        }
        console.log(` : ${property}`)
        return target[property]
    },
    set(target, property, value) {
        if (property in target) {
            target[property] = value
            console.log(` : ${property}`)
        } else {
            console.error('  ')
        }
    },
    has(target, property) {
        // return property in target
        return Object.entries(target)
            .flat()
            .includes(property)
    },
    deleteProperty(target, property) {
        if (property in target) {
            delete target[property]
            console.log(` : ${property}`)
        } else {
            console.error('  ')
        }
    }
})

console.log(proxy.company_city_firstname_lastname) // Google Harry Heman Mountain View
proxy.firstname = 'John' //  : firstname
proxy.surname = 'Smith' //   
console.log(proxy.city) //  : city Mountain View
console.log('company' in proxy) // true
delete proxy.age //

6. Reduce



The method reduce()



applies the function reducer



to each element of the array, returning the resulting value. This method takes four arguments: the initial value (or the value of the previous callback function), the value of the current element, the current index, and the iterable array (the last two are optional). In the simplest case, it looks like this: 



const arr = [1, 2, 3]

const total = arr.reduce((sum, cur) => sum + cur)
console.log(total) // 6

// forEach
let total2 = 0
arr.forEach(num => total2 += num)
console.log(total2) // 6

Talk about the grouping: 



const groupBy = (arr, criteria) =>
    arr.reduce((obj, item) => {
        const key = typeof criteria === 'function'
            ? criteria(item)
            : item[criteria]

        if (!obj.hasOwnProperty(key)) obj[key] = ''
        obj[key] = item

        return obj
    }, {})

const nums = [6.1, 4.2, 2.3]
console.log(groupBy(nums, Math.floor)) // {2: 2.3, 4: 4.2, 6: 6.1}

// forEach
const groupBy2 = (arr, criteria, obj = {}) => {
    arr.forEach(item => {
        const key = typeof criteria === 'function'
            ? criteria(item)
            : item[criteria]
        
        if (!obj.hasOwnProperty(key)) obj[key] = ''
        obj[key] = item
        return obj
    })
    return obj
}
const words = ['one', 'three', 'five']

10. Private variables in classes



class Person {
    //   
    static type = ''
    static #area = ''
    name = 'John'
    #year = 1990

    get age() {
        return new Date().getFullYear() - this.#year
    }

    set year(age) {
        if (age > 0) {
            this.#year = new Date().getFullYear() - age
        }
    }

    get year() {
        return this.#year
    }

    static area() {
        return Person.#area
    }
}

const person = new Person()
console.log(person) // Person {name: "John", #year: 1990}
console.log(person.age) // 30
