	let input_number = prompt("  ");
	// ,      
	let numbers = [];
	
	function toArray(input_number){
		for (let i = 0; i < input_number; i++) {

			numbers.push(x); //     
		}

	toArray(input_number); 
	
	function toAverage(numbers){
