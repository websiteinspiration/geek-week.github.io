Google Cloud Functionsを使用して、スケジュールされた通話レポートをCoMagicからBigQueryにスケジュールに従ってインポートします

何のために


広告キャンペーンの複雑な構造と多数の通話により、着信通​​話に関する情報を保存、処理、分析するための追加のツールが必要になります。多くの場合、長期間にわたってデータにすばやくアクセスする必要があります。複雑なデータ処理、特定のチャネルまたはキャンペーンへの呼び出しの関連付けが必要になる場合があります。

作業をスピードアップするためのオプションの1つは、追加の利点も提供しますが、呼び出しをCoMagicからGoogle BigQueryにインポートすることです。BigQueryの利点については多く書かれているので、作成に移りましょう。

自動インポートを作成するには、次のものが必要です。

  1. 作成されたプロジェクトのGoogleアカウント（まだの場合）
  2. Pythonの知識
  3. Google Cloudドキュメントの紹介

ここで は、プロジェクトの作成方法について説明します。プロジェクトが作成されたら、BigQueryでデータセットを作成する必要があります。データセットを作成するためのBQドキュメントと手順。

CoMagicからのデータの取得


CoMagicのドキュメント を参照してください。通話または通話のリストを取得するには、レポートセクションが必要です。

CoMagic APIを操作するための単純なクラスを作成します。必要なすべての要件は、GitHubへのリンクの最後に示されます。

import json
import requests
import random
import pandas as pd

class ComagicClient:
    """     CoMagic"""
    def __init__(self, login, password):
        """     CoMagic"""
        self.login = login
        self.password = password
        self.base_url = 'https://dataapi.comagic.ru/v2.0'
        self.payload_ = {"jsonrpc":"2.0",
                         "id":1,
                         "method":None,
                         "params": None}
        self.token = self.get_token(self.login, self.password)
    def base_request(self, method, params):
        """     CoMagic.      API
          .    JSON-like .
        : https://www.comagic.ru/support/api/data-api/"""
        id_ = random.randrange(10**7) # 
        payload = self.payload_.copy()
        payload["method"] = method
        payload["params"] = params
        payload["id"] = id_
        self.r = requests.post(self.base_url, data=json.dumps(payload))
        self.last_response = json.loads(self.r.text)
        return self.last_response
    def get_token(self, login, password):
        """   .    
          .  ."""
        method = "login.user"
        params = {"login":self.login,
                  "password":self.password}
        response = self.base_request(method, params)
        token = response['result']['data']['access_token']
        return token
    def get_report_per_page(self, method, params):
        """  .    
         10000 .    .
            110000 .
            JSON-like ."""
        response = self.base_request(method, params)
        print(f"""  c {params["date_from"]}  {params["date_till"]}.  = {params["offset"]}""")
        result = response['result']['data']
        if len(result) < 10000:
            return result
        else:
            params['offset'] += 10000
            add_result = self.get_report_per_page(method, params)
            return result + add_result
    def get_basic_report(self, method, fields, date_from, date_till, filter=None, offset=0):
        """   .
              method  fields.
              .
              ,
          ,       .

        method -- <string>  
        date_from -- <string>  .  "YYYY-MM-DD hh:mm:ss"
        date_till -- <string>  .  "YYYY-MM-DD hh:mm:ss"
        fields -- <list>,   
        filter [] - <dict> 
        offset [] -- <int> 

        return -- <list> 
        """
        params = {"access_token":self.token,
                  "limit":10000,
                  "date_from":date_from,
                  "date_till":date_till,
                  "fields": fields,
                  "offset": offset}
        if filter:
            params['filter'] = filter
        report = self.get_report_per_page(method, params)
        return report

BigQueryへのデータの送信


CoMagicからのデータを受信して​​変換したら、BigQueryに送信する必要があります。

from google.cloud import bigquery
from google.cloud.exceptions import NotFound
import pandas as pd


class BQTableHanler:
    """     BigQuery"""
    time_partition_field = 'PARTITION_DATE'
    def __init__(self, full_table_id, service_account_file_key_path = None):
        """      
        `myproject.mydataset.mytable`.

         ,   Application Default Credentials,
                
         ."""
        self.full_table_id = full_table_id
        project_id, dataset_id, table_id = full_table_id.split(".")
        self.project_id = project_id
        self.dataset_id = dataset_id
        self.table_id = table_id

        if service_account_file_key_path: #     
            from google.oauth2 import service_account
            self.credentials = service_account.Credentials.from_service_account_file(
                        service_account_file_key_path,
                        scopes=["https://www.googleapis.com/auth/cloud-platform"],)
            self.bq_client = bigquery.Client(credentials = self.credentials,
                                    project = self.project_id)
        else:
            self.bq_client = bigquery.Client()
        self.dataset = self.bq_client.get_dataset(self.dataset_id)
        self.location = self.dataset.location
        self.table_ref = self.dataset.table(self.table_id)

    def get_last_update(self):
        """       
        Pandas datetime.      False."""
        try:
            self.bq_client.get_table(self.full_table_id)
        except NotFound as error:
            return False
        query = f"""SELECT MAX({self.time_partition_field}) as last_call
                    FROM `{self.full_table_id}`"""
        result = self.bq_client.query(query,location=self.location).to_dataframe()
        date = pd.to_datetime(result.iloc[0,0]).date()
        return date

    def insert_dataframe(self, dataframe):
        """      BigQuery.
            Pandas DataFrame.
           ,       ."""
        job_config = bigquery.LoadJobConfig() #    
        job_config._properties['load']['timePartitioning'] = {'type': 'DAY',
                                            'field': self.time_partition_field}

        result = self.bq_client.load_table_from_dataframe(dataframe,
                 self.table_ref, job_config=job_config).result()
        return result

データ更新のロジックを決定する


CoMagicから受信するデータ行の数には制限があるため、要求されるデータの量を制限する必要があります。リクエスト期間を限定させていただきます。これを行うには、長期間を指定された長さのセグメントに分割する補助関数が必要です。

def interval_split(array, interval):
    """      .
      ,      2,
       -   ,
        -    .
    :
    get_intervals([1,2,3,4,5,6,7], 3) => [[1,3], [4,6], [7]]
    get_intervals([1,2,3], 4) => [[1,3]]"""
    intervals = []
    iw, i = 0, 0
    l = len(array)
    for v in array:
        if i==0 or (i)%interval==0:
            intervals.append([v])
        if (i+1)%interval == 0 or (i+1) == l:
            intervals[iw].append(v)
            iw+=1
        i+=1
    return intervals

これは、初めてデータをロードする場合や、長期間にわたってデータをダウンロードする必要がある場合に必要です。期間はいくつかの小さな期間に分割されます。ちなみに、これには時間制限があるため、Cloud Functionを使用せずに行うことをお勧めします。まあ、または、オプションとして、関数を何度も実行できます。

データとCoMagicからのデータを保存するBigQueryテーブルをリンクするコネクタクラスを作成します。

from helpfunctions import interval_split
import pandas as pd

class Connector:
    """      """
    time_partition_field = 'PARTITION_DATE' #  -.      
    def __init__ (self, source, dest):
        """     
            """
        self.source = source
        self.dest = dest
        self.source.time_partition_field = self.time_partition_field
        self.dest.time_partition_field = self.time_partition_field
    def insert_data_in_dest(self, start_date, end_date):
        """      .
                
        ,     ."""
        dates = pd.date_range(start_date, end_date)
        week_intervals = interval_split(dates, 7) #     7 
        for week_interval in week_intervals:
            date_from = week_interval[0].strftime("%Y-%m-%d") + " 00:00:00"
            date_till = week_interval[1].strftime("%Y-%m-%d") + " 23:59:59"
            calls_df = self.source.get_calls_report(date_from, date_till)
            self.dest.insert_dataframe(calls_df)
            print (f"  {date_from}  {date_till}   ")
        return True
    def update_dest_data(self):
        #     BigQuery
        last_date = self.dest.get_last_update()
        if not last_date: #   
            last_date = self.source.day_before_first_call
        yesterday = pd.Timestamp.today(tz='Europe/Moscow').date() - pd.Timedelta(days=1)
        if last_date == yesterday:
            print("  ")
        else:
            last_date = last_date + pd.Timedelta(days=1)
            self.insert_data_in_dest(last_date, yesterday)
        return True

Google Cloud Platformを構成する


すべてのファイルをZIPアーカイブに収集します。credfile.pyファイルに、トークンを受け取るためのCoMagicログインとパスワード、およびBigQueryのテーブルの完全な名前と、スクリプトがローカルマシンから実行される場合のサービスアカウントファイルへのパスを入力します。

Cloud Functionを作成する


  • コンソールに移動
  • 関数がまだ作成されていない場合は、「関数の作成」をクリックします
  • トリガーフィールドで、PUB / SUBを選択します。
  • PUB / SUBの新しいテーマを作成します。たとえば、「update_calls」

  • 環境：Python 3.7
  • zipファイルをダウンロード

  • 「呼び出された関数」フィールドに「main」と書き込みます
