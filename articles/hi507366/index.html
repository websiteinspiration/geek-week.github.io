class SampleSimulation extends Simulation {
  val stageWeight    = getDoubleParam("stageWeight") // Double  
  val startIntensity = getDoubleParam("startIntensity")
  val warmUpDuration = getDurationParam("warmUp") // FiniteDuration 
  setUp(
    CommonScenario().inject(
      rampUsersPerSec(0) to startIntensity during warmUpDuration,
      incrementUsersPerSec(intensity * stageWeight)
        .times(stagesNumber)
        .eachLevelLasting(stageDuration)
        .separatedByRampsLasting(rampDuration)
        .startingFrom(startIntensity)
    )
  ).protocols(httpProtocol).maxDuration(testDuration)
}
