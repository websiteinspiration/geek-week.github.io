Tidak seperti pembelian reguler, di mana Apple mengambil komisi 30%, pada langganan komisi ini berkurang menjadi 15% jika pengguna mendaftar selama 1 tahun atau lebih. extension Purchases: SKProductsRequestDelegate {
    func productsRequest(_ request: SKProductsRequest, didReceive response: SKProductsResponse) {
        guard !response.products.isEmpty else {
            print("Found 0 products")
            return
        }

        for product in response.products {
            print("Found product: \(product.productIdentifier)")
        }
    }

    func request(_ request: SKRequest, didFailWithError error: Error) {
        print("Failed to load products with error:\n \(error)")
    }
}
