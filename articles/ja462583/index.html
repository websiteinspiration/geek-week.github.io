確定的ガベージコレクターポインターの紹介

メモリ、C ++での作業は常に困難でした（Cの苦い遺産）...ここで、C ++ 11とそのstd :: shared_ptrが役立ちます。


ご想像のとおり、これらのプリミティブに問題がなければ、この記事はそうではなかったでしょう:)

std :: shared_ptr での古典的なメモリリークの次の例を見てみましょう。

#include <iostream>
#include <memory>

class Child;

class Parent {
 public:
  Parent() {
    std::cout << "Parent()" << std::endl;
  }

  ~Parent() {
    std::cout << "~Parent()" << std::endl;
  }

  void createChild() {
    child_ptr_ = std::make_shared<Child>();
  }

  std::shared_ptr<Child> getChild() {
    return child_ptr_;
  }

 private:
  std::shared_ptr<Child> child_ptr_;
};

class Child {
 public:
  Child() {
    std::cout << "Child()" << std::endl;
  }

  ~Child() {
    std::cout << "~Child()" << std::endl;
  }

  void setParent(std::shared_ptr<Parent> parentPtr) {
    parent_ptr_ = parentPtr;
  }

 private:
  std::shared_ptr<Parent> parent_ptr_;
};

int main() {
  auto parent = std::make_shared<Parent>();
  parent->createChild();
  parent->getChild()->setParent(parent);
  return 0;
}

明らかに、オブジェクトデストラクタの呼び出しは表示されません。これに対処するには？std :: weak_ptrが私たちの助けになります：

class Child {

  ...

  void setParent(std::shared_ptr<Parent> parentPtr) {
    parent_ptr_ = parentPtr;
  }

 private:
  std::weak_ptr<Parent> parent_ptr_;
};

はい、問題の解決に役立ちます。しかし、オブジェクトのより複雑な階層があり、誰がstd :: weak_ptrを実行し、誰をstd :: shared_ptrにする必要があるかを理解するのが非常に難しい場合は、それとも、弱いつながりをまったく台無しにしたくないですか？

ガベージコレクターが私たちのすべてです！

いいえ、もちろん違います。 C ++では、ガベージコレクターのネイティブサポートはありません。ガベージコレクターが追加された場合でも、ガベージコレクターのオーバーヘッドコストとRAIIブレークが発生します。

私たちは何をしますか？

確定的なガベージコレクタのポインタからのトラックのすべてのリンクそのポインタであるルートオブジェクトとのどれもすぐとしてルートオブジェクトは、私たちのオブジェクトを参照しない、それはすぐに削除されます。

その動作原理は似ていますstd :: shared_ptr（スコープを追跡します）だけでなく、それを参照するオブジェクトも。

前の例でその動作の原理を見てみましょう：

#include <iostream>
#include "gc_ptr.hpp"

class Child;

class Parent {
 public:
  Parent() {
    std::cout << "Parent()" << std::endl;
  }

  ~Parent() {
    std::cout << "~Parent()" << std::endl;
  }

  void createChild() {
    child_ptr_.create_object();
  }

  memory::gc_ptr<Child> getChild() {
    return child_ptr_;
  }

  void connectToRoot(void * rootPtr) {
    child_ptr_.connectToRoot(rootPtr);
  }

  void disconnectFromRoot(bool isRoot, void * rootPtr) {
    child_ptr_.disconnectFromRoot(isRoot, rootPtr);
  }

 private:
  memory::gc_ptr<Child> child_ptr_;
};

class Child {
 public:
  Child() {
    std::cout << "Child()" << std::endl;
  }

  ~Child() {
    std::cout << "~Child()" << std::endl;
  }

  void setParent(memory::gc_ptr<Parent> parentPtr) {
    parent_ptr_ = parentPtr;
  }

  void connectToRoot(void * rootPtr) {
    parent_ptr_.connectToRoot(rootPtr);
  }

  void disconnectFromRoot(bool isRoot, void * rootPtr) {
    parent_ptr_.disconnectFromRoot(isRoot, rootPtr);
  }

 private:
  memory::gc_ptr<Parent> parent_ptr_;
};

int main() {
  memory::gc_ptr<Parent> parent;
  parent.create_object();
  parent->createChild();
  parent->getChild()->setParent(parent);
  return 0;
}

ご覧のとおり、コードは少し増えていますが、これはオブジェクトを完全に自動的に削除するために必要な代償です。追加のconnectToRootメソッドとdisconnectFromRootメソッドが追加されたことがわかります。もちろん、いつも手で書くのはかなり難しいので、gc_ptrを使用するクラスでこれらのメソッドの小さなジェネレーターを作成するつもりです（これらを見ると、ゼロオーバーヘッドの原則に準拠しているため、使用しないものについては支払いません。その場合のコストは、私たちが手で書いた場合と同じくらいです。Gc_ptr.hpp

ライブラリスレッドセーフ、ガベージコレクション用の追加のスレッドは作成されません。すべてがコンストラクター、デストラクタ、および代入演算子で行われるため、オブジェクトを上書きし、ルートオブジェクトがそれを参照しない場合、割り当てられたメモリを返します。私たちの施設。
