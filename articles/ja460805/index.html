マイクロコントローラー間で100 Mbpsでデータを転送する方法

この種の問題が発生しました。2つのSTM32F407マイクロコントローラー間で、少なくとも100 Mbpsの速度でデータを転送する必要があります。イーサネット（MAC-to-MAC）を使用することは可能ですが、問題はビジーであるということです。これらのデータが取得されるのはこのデータからです...
アイドルペリフェラルからはおそらくSPIがありますが、42 Mbpsしかありません。

奇妙なことに、ネットワーク上で準備ができているものは何も見つかりませんでした。そして、パラレル8ビットクロックレジスタを実装することにしました。また、周波数は10 MHzに設定できます（つまり、クロック自体は2倍の速さですが、20 MHzはそれほど複雑ではありません）。したがって、このような低い周波数では、ボードの配線について心配する必要はありません。速度は100 Mbpsになります。

否や言うほどない。一般的に、システムは次のようになります。送信側でタイマーを使用し、比較信号の1つがピンに出力されます。これはクロック信号で、2つ目はDMAの1つのバーストを開始するために使用されます。

私は82 MHzのバス（より高い周波数での電流消費のため）、同じ周波数のタイマーを持っています。そのため、ARR = 8の周期で約10 MHzになります（つまり、約80 Mbpsです）。

DMAはメモリから1バイトを（もちろん自動インクリメントで）レジスタ出力ポートに直接送信します-私の場合、PORTEが起動しました-最初の8ビットはDMAレシーバーアドレスとしてちょうど適合します。

受信側では、両方のエッジでクロック信号を使用して、周期1でタイマーをクロックします。更新信号を使用してDMAの転送を開始します。DMAは、ポート（PORTEポートに再び接近した）からデータを読み取り、自動インクリメントでメモリに書き込みます。

これで、すべてを正しく構成し（以下のコード）、実行できます。両側の終了は、DMAからの割り込みによって決定されます。

ただし、完全を期すために、送信遅延のチェックとエラー処理をコードに含める必要がありますが、これは省略します。

以下のコードでは、TIM8タイマーはCC2チャネルを使用して信号を出力し、何が発生するかを確認します。

volatile int transmit_done;
volatile int receive_done;

void DMA2_Stream1_IRQHandler(void) {
    TIM8->CR1 &= ~TIM_CR1_CEN;
    DMA2->LIFCR |= 0b1111 << 8;
    receive_done = 1;
}

void DMA2_Stream4_IRQHandler(void) {
    TIM1->CR1 &= ~TIM_CR1_CEN;
    TIM1->EGR |= TIM_EGR_BG;
    DMA2->HIFCR |= 0b1111101;
    transmit_done = 1;
}

void ii_receive(uint8_t *data, int len) {
    GPIOE->MODER = (GPIOE->MODER & 0xFFFF0000) | 0x0000;
    DMA2_Stream1->PAR = (uint32_t) &(GPIOE->IDR);
    DMA2_Stream1->M0AR = (uint32_t) data;
    DMA2_Stream1->NDTR = len;
    TIM8->CNT = 0;
    TIM8->BDTR |= TIM_BDTR_MOE;
    receive_done = 0;
    DMA2_Stream1->CR |= DMA_SxCR_EN;
    TIM8->CR1 |= TIM_CR1_CEN;
}

void ii_transmit(uint8_t *data, int len) {
    GPIOE->MODER = (GPIOE->MODER & 0xFFFF0000) | 0x5555;
    DMA2_Stream4->PAR = (uint32_t) &(GPIOE->ODR);
    DMA2_Stream4->M0AR = (uint32_t) data;
    DMA2_Stream4->NDTR = len;
    TIM1->CNT = 6;
    transmit_done = 0;
    DMA2_Stream4->CR |= DMA_SxCR_EN;
    TIM1->SR |= TIM_SR_BIF;
    TIM1->BDTR |= TIM_BDTR_MOE;
    TIM1->CR1 |= TIM_CR1_CEN;
}

// tx: TIM1 CH4 on DMA2/stream4/channel6, CH1 on output clock in PE9
// rx: TIM8 CH2 on DMA2/stream3/channel7, CH1 on input clock in PC6
void ii_init() {
    __HAL_RCC_GPIOC_CLK_ENABLE();
    __HAL_RCC_GPIOE_CLK_ENABLE();
    __HAL_RCC_TIM1_CLK_ENABLE();
    __HAL_RCC_TIM8_CLK_ENABLE();
    __HAL_RCC_TIM2_CLK_ENABLE();
    __HAL_RCC_DMA2_CLK_ENABLE();
    GPIOC->MODER |= (0b10 << GPIO_MODER_MODE6_Pos)
            | (0b10 << GPIO_MODER_MODE7_Pos);
    GPIOC->PUPDR |= (0b10 << GPIO_PUPDR_PUPD7_Pos);
    GPIOC->AFR[0] |= (GPIO_AF3_TIM8 << 24) | (GPIO_AF3_TIM8 << 28);
    GPIOE->MODER |= (0b10 << GPIO_MODER_MODE9_Pos);
    GPIOE->OSPEEDR |= GPIO_OSPEEDER_OSPEEDR9 | 0xFFFF;
    GPIOE->AFR[1] |= GPIO_AF1_TIM1 << 4;
    GPIOE->PUPDR |= (0b10 << GPIO_PUPDR_PUPD9_Pos);
    TIM1->ARR = 8;
    TIM1->CCR1 = 5;
    TIM1->CCR4 = 1;
    TIM1->EGR |= TIM_EGR_CC4G;
    TIM1->DIER |= TIM_DIER_CC4DE;
    TIM1->CCMR1 |= (0b110 << TIM_CCMR1_OC1M_Pos);
    TIM1->CCER |= TIM_CCER_CC1E;
    TIM1->EGR |= TIM_EGR_BG;
    TIM8->ARR = 1;
    TIM8->CCR2 = 1;
    TIM8->EGR |= TIM_EGR_UG;
    TIM8->DIER |= TIM_DIER_UDE;
    TIM8->SMCR |= (0b100 << TIM_SMCR_TS_Pos) | (0b111 << TIM_SMCR_SMS_Pos);
    TIM8->CCMR1 = (0b01 << TIM_CCMR1_CC1S_Pos) | (0b110 << TIM_CCMR1_OC2M_Pos);
    TIM8->CCER |= (0b11 << TIM_CCER_CC1P_Pos) | TIM_CCER_CC2E;
    DMA2_Stream1->CR = DMA_CHANNEL_7 | DMA_PRIORITY_VERY_HIGH | DMA_MINC_ENABLE
            | (0b00 << DMA_SxCR_DIR_Pos) | DMA_SxCR_TCIE | DMA_SxCR_TEIE
            | DMA_SxCR_DMEIE;
    DMA2_Stream1->FCR |= DMA_FIFOMODE_ENABLE;
    DMA2_Stream4->CR = DMA_CHANNEL_6 | DMA_PRIORITY_VERY_HIGH | DMA_MINC_ENABLE
            | (0b01 << DMA_SxCR_DIR_Pos) | DMA_SxCR_TCIE | DMA_SxCR_TEIE
            | DMA_SxCR_DMEIE;
    DMA2_Stream4->FCR |= DMA_FIFOMODE_ENABLE;
    HAL_NVIC_SetPriority(DMA2_Stream1_IRQn, 0, 0);
    HAL_NVIC_EnableIRQ(DMA2_Stream1_IRQn);
    HAL_NVIC_SetPriority(DMA2_Stream4_IRQn, 0, 0);
    HAL_NVIC_EnableIRQ(DMA2_Stream4_IRQn);
}

テストでは、同じボードが使用され、PE9クロック出力がPC6入力に接続されました。メインループは次のようになります。

 ii_receive(rdata, 256);
 ii_transmit(tdata, 256);
 while (!transmit_done);
 while (!receive_done);

結果によると、データは損失なしで30〜31マイクロ秒間完全に送信されました。信号は次のようになります。


ここで、白はTIM8タイマーの出力、赤はクロック信号（TIM1）、オレンジはデータの最下位ビット（0-1-0-1 -...）です。

これについて私が気に入らないのは、GPIO入力からのDMAの割り込みを開始できないため、タイマーを使用する必要があることです。多分誰かがあなたに別の方法を教えてくれますか？

PS新鮮な実験の結果、周波数を168 MHzに上げると速度が自然に2倍になり、データが14マイクロ秒（つまり、150 Mbps）で送信されましたが、マスタータイマーの境界が7未満に減少すると、受信側でグリッチが発生し始めました-タイマーには時間がありませんTIM8。7でそれはまだ機能しますが、6で既になくなっており、結局それは200 Mbpsになります...
