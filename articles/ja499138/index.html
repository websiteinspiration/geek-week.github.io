トポロジカルソーティング

「開発者のためのアルゴリズム」コースの開始前に、記事の翻訳が準備されました。



有向非巡回グラフ（有向非巡回グラフ、以下DAGと呼ぶ）のトポロジカルソートは、次の条件が当てはまる頂点の線形順序です。各有向エッジuvの場合、頂点uが頂点vの前に順序付けされます。グラフがDAGでない場合、トポロジーによるソートはできません。

たとえば、下のグラフのトポロジカルソートは「5 4 2 3 1 0」です。グラフには、いくつかのトポロジカルソートが存在する可能性があります。たとえば、同じグラフの別のトポロジカルソートは「4 5 2 3 1 0」です。トポロジカルソートの最初の頂点は、常に入力エッジのない頂点です。



トポロジカルソーティングとディープウォーク：深く行く

場合（深さ優先トラバーサル、以下DFSと呼びます）頂点を出力し、隣接する頂点のDFSを再帰的に呼び出します。トポロジカルソートでは、隣接する頂点の前に頂点を表示する必要があります。たとえば、このグラフでは、頂点「5」が頂点「0」の前に表示され、DFSとは異なり、頂点「4」も頂点「0」の前に表示されます。これにより、トポロジーソートとDFSが区別されます。たとえば、上記の列のDFSは「5 2 3 1 0 4」ですが、これはトポロジカルソートではありません。

推奨事項：アルゴリズムの実装に進む前に、まず実際のタスクを理解してください。

トポロジカルソート検索アルゴリズム。

開始するには、この DFS実装に慣れることをお勧めします。グラフのトポロジカルソートが得られるようにDFSを変更できます。ではDFS最初に出力する適切な頂点から始め、次に隣接する頂点に対してDFSを再帰的に呼び出します。トポロジカルソートでは、一時スタックを使用します。頂点はすぐには表示されません。最初に、隣接するすべての頂点に対してトポロジカルソートが呼び出され、次に頂点がスタックにプッシュされます。すべての頂点をトラバースした後にのみ、スタックの内容が表示されます。隣接するすべての頂点（およびそれらに隣接する頂点など）が既にスタック上にある場合にのみ、頂点がスタックにプッシュされることに注意してください。

以下は、トポロジーソートの実装です。切断されたグラフのDFT実装を確認し、そこに示されている2番目のコードと以下のコードの違いに注意してください。

//   C++     DAG
 
#include<iostream>
#include <list>
#include <stack>
using namespace std;
  
//    
class Graph
{
    int V;	//  
  
    //    ,   
    list<int> *adj;
  
    // ,  topologicalSort
    void topologicalSortUtil(int v, bool visited[], stack<int> &Stack);
public:
    Graph(int V);   // 
  
     //      
    void addEdge(int v, int w);
  
    //    
    void topologicalSort();
};
  
Graph::Graph(int V)
{
    this->V = V;
    adj = new list<int>[V];
}
  
void Graph::addEdge(int v, int w)
{
    adj[v].push_back(w); // Add w to v’s list.
}
  
//  ,  topologicalSort
void Graph::topologicalSortUtil(int v, bool visited[], 
                                stack<int> &Stack)
{
    //     
    visited[v] = true;
  
    //       
    list<int>::iterator i;
    for (i = adj[v].begin(); i != adj[v].end(); ++i)
        if (!visited[*i])
            topologicalSortUtil(*i, visited, Stack);
  
    //       
    Stack.push(v);
}
  
//     . 
//   topologicalSortUtil()
void Graph::topologicalSort()
{
    stack<int> Stack;
  
    //     
    bool *visited = new bool[V];
    for (int i = 0; i < V; i++)
        visited[i] = false;
  
    //     
    //       
    for (int i = 0; i < V; i++)
      if (visited[i] == false)
        topologicalSortUtil(i, visited, Stack);
  
    //   
    while (Stack.empty() == false)
    {
        cout << Stack.top() << " ";
        Stack.pop();
    }
}
  
//    
int main()
{
    //  ,    
    Graph g(6);
    g.addEdge(5, 2);
    g.addEdge(5, 0);
    g.addEdge(4, 0);
    g.addEdge(4, 1);
    g.addEdge(2, 3);
    g.addEdge(3, 1);
  
    cout << "Following is a Topological Sort of the given graph \n";
    g.topologicalSort();
  
    return 0;
}

//   Java    
import java.io.*;
import java.util.*;
  
//          
class Graph
{
    private int V;   //  
    private LinkedList<Integer> adj[]; //  
  
    // 
    Graph(int v)
    {
        V = v;
        adj = new LinkedList[v];
        for (int i=0; i<v; ++i)
            adj[i] = new LinkedList();
    }
  
    //      
    void addEdge(int v,int w) { adj[v].add(w); }
  
    //  ,  topologicalSort
    void topologicalSortUtil(int v, boolean visited[],
                             Stack stack)
    {
        //      
        visited[v] = true;
        Integer i;
  
        //       
        Iterator<Integer> it = adj[v].iterator();
        while (it.hasNext())
        {
            i = it.next();
            if (!visited[i])
                topologicalSortUtil(i, visited, stack);
        }
  
        //       
        stack.push(new Integer(v));
    }
  
    //     .
    //   topologicalSortUtil()
    void topologicalSort()
    {
        Stack stack = new Stack();
  
        //     
        boolean visited[] = new boolean[V];
        for (int i = 0; i < V; i++)
            visited[i] = false;
  
        //    
        //       
        for (int i = 0; i < V; i++)
            if (visited[i] == false)
                topologicalSortUtil(i, visited, stack);
  
        //   
        while (stack.empty()==false)
            System.out.print(stack.pop() + " ");
    }
  
    //   
    public static void main(String args[])
    {
        //  ,    
        Graph g = new Graph(6);
        g.addEdge(5, 2);
        g.addEdge(5, 0);
        g.addEdge(4, 0);
        g.addEdge(4, 1);
        g.addEdge(2, 3);
        g.addEdge(3, 1);
  
        System.out.println("Following is a Topological " +
                           "sort of the given graph");
        g.topologicalSort();
    }
}

時間の難しさ：上記のアルゴリズムは、追加のスタックを備えたDFSです。したがって、時間の複雑さはDFSの場合と同じで、O（V + E）です。

注：スタックの代わりにベクトルを使用することもできます。ベクトルを使用してトポロジカルソートを行う場合は、要素を逆の順序で出力する必要があります。

応用：

トポロジカルソートは主に、それらの間の特定の依存関係から作業をスケジュールするために使用されます。コンピューターサイエンスでは、コマンドの計画、スプレッドシートの数式値を再計算するときの数式計算用のセルの編成、論理合成、メイクファイルで実行されるコンパイルタスクの順序の決定、データのシリアル化、リンカーでのシンボリック依存関係の解決に使用されます[ 2 ]。
