NAND / NORを使用したコンパイル時の難読化

前書き


現代のC ++に精通しているすべての人は、C ++ 11以降、constexpr指定子が標準に導入され、限られたコンパイル時の計算を実行できると聞いています。constexprおよびconstexprラムダが後続の標準に追加された場合、ある程度の制限が取り除かれ、コンパイル時間の計算を含むコードの記述に役立ちます。今日は、コンパイル時のランダムコード生成について説明します。

「ランダムな」コンパイル時の値を受け取るためのベースを選択します


コードのコンパイル時に特定のパスを選択するには、constexpr PRNG cの出力データが必要です。これにはジェネレータxorshiftを選択しましたが、xorshiftは経験的にBigCrush（testu01）のテストに合格しています。

xorshift256 ++
#include <stdint.h>

/* This is xoshiro256++ 1.0, one of our all-purpose, rock-solid generators.
   It has excellent (sub-ns) speed, a state (256 bits) that is large
   enough for any parallel application, and it passes all tests we are
   aware of.

   For generating just floating-point numbers, xoshiro256+ is even faster.

   The state must be seeded so that it is not everywhere zero. If you have
   a 64-bit seed, we suggest to seed a splitmix64 generator and use its
   output to fill s. */

static inline uint64_t rotl(const uint64_t x, int k) {
	return (x << k) | (x >> (64 - k));
}


static uint64_t s[4];

uint64_t next(void) {
	const uint64_t result = rotl(s[0] + s[3], 23) + s[0];

	const uint64_t t = s[1] << 17;

	s[2] ^= s[0];
	s[3] ^= s[1];
	s[1] ^= s[2];
	s[0] ^= s[3];

	s[2] ^= t;

	s[3] = rotl(s[3], 45);

	return result;
}


/* This is the jump function for the generator. It is equivalent
   to 2^128 calls to next(); it can be used to generate 2^128
   non-overlapping subsequences for parallel computations. */

void jump(void) {
	static const uint64_t JUMP[] = { 0x180ec6d33cfd0aba, 0xd5a61266f0c9392c, 0xa9582618e03fc9aa, 0x39abdc4529b1661c };

	uint64_t s0 = 0;
	uint64_t s1 = 0;
	uint64_t s2 = 0;
	uint64_t s3 = 0;
	for(int i = 0; i < sizeof JUMP / sizeof *JUMP; i++)
		for(int b = 0; b < 64; b++) {
			if (JUMP[i] & UINT64_C(1) << b) {
				s0 ^= s[0];
				s1 ^= s[1];
				s2 ^= s[2];
				s3 ^= s[3];
			}
			next();	
		}
		
	s[0] = s0;
	s[1] = s1;
	s[2] = s2;
	s[3] = s3;
}



/* This is the long-jump function for the generator. It is equivalent to
   2^192 calls to next(); it can be used to generate 2^64 starting points,
   from each of which jump() will generate 2^64 non-overlapping
   subsequences for parallel distributed computations. */

void long_jump(void) {
	static const uint64_t LONG_JUMP[] = { 0x76e15d3efefdcbbf, 0xc5004e441c522fb3, 0x77710069854ee241, 0x39109bb02acbe635 };

	uint64_t s0 = 0;
	uint64_t s1 = 0;
	uint64_t s2 = 0;
	uint64_t s3 = 0;
	for(int i = 0; i < sizeof LONG_JUMP / sizeof *LONG_JUMP; i++)
		for(int b = 0; b < 64; b++) {
			if (LONG_JUMP[i] & UINT64_C(1) << b) {
				s0 ^= s[0];
				s1 ^= s[1];
				s2 ^= s[2];
				s3 ^= s[3];
			}
			next();	
		}
		
	s[0] = s0;
	s[1] = s1;
	s[2] = s2;
	s[3] = s3;
}


これからは、計算が長くなりすぎないように、関数rotlとnextのみが必要です。MSVCコンパイラの開発者に心からの挨拶を送ります。書き込みプロセス中に発見されたコンパイラのバグを考慮して、これについて以下で説明します。

上記のコードからわかるように、状態を格納する4つの64ビットの静的変数が必要です。このコードをコンパイル時に機能させるには、ランダムな状態をパラメーターとして渡すことでそれらを取り除く必要があります。constexprハッシュfnv1とマクロ__COUNTER__を使用して整理し、これに従ってコードを書き直すと、次のようになります。

コンパイル時の「ランダム」
#include <cstdint>
#define STRING(s) #s

// an enumeration can also be used here
template <typename T, T value>
constexpr T ensure_constexpr() {
  return value;
}
#define CONSTEXPR(x) ensure_constexpr<decltype(x), x>()

constexpr uint64_t fnv1impl(uint64_t h, const char* s) {
  return (*s == 0)
             ? h
             : fnv1impl((h * 1099511628211ull) ^ static_cast<uint64_t>(*s),
                        s + 1);
}

constexpr uint64_t fnv1(const char* s) {
  return fnv1impl(14695981039346656037ull, s);
}

template <uint64_t n>
constexpr uint64_t get_seed(const uint64_t x,
                            const uint64_t y = CONSTEXPR(fnv1(STRING(n)))) {
  return x ^ y * n;
}

#define SEED CONSTEXPR(get_seed<__COUNTER__ + 1>(fnv1(__TIME__)))

//! Rotate left by
constexpr uint64_t rotl(uint64_t x, int k) {
  return (x << k) | (x >> (64 - k));
}

//! XorShift 256 compile time random implementation
template <uint64_t a, uint64_t b, uint64_t c, uint64_t d>
constexpr uint64_t xorshift256_next() {
  uint64_t s[4] = {a, b, c, d};
  const uint64_t t = s[1] << 17;

  s[2] ^= s[0];
  s[3] ^= s[1];
  s[1] ^= s[2];
  s[0] ^= s[3];

  s[2] ^= t;

  s[3] = rotl(s[3], 45);

  return rotl(s[0] + s[3], 23) + s[0];
}

#define RND                                                         \
  xorshift256_next<                                                 \
      CONSTEXPR(SEED + __COUNTER__), CONSTEXPR(SEED + __COUNTER__), \
      CONSTEXPR(SEED + __COUNTER__), CONSTEXPR(SEED + __COUNTER__)>()

やったー！期待どおりの結果が得られ、コンパイラーはRNDマクロの2つの呼び出しを2つのmov命令にデプロイしました。リリースモードでのclangコンパイラとMSVCコンパイラの動作は似ています。コンパイル時のコード生成に進むことができます。

さらに理解するには、すべてのロジックを構築できるユニバーサルロジックエレメントである
NOR（ピアアロー）NAND（シェーファーバー）に慣れる必要があります。＆|を実装するためのすべてのコードを添付しません。^ ++-、NANDテンプレート関数のランダムサンプリングに基づいて1つの例のみを分析します。他のエミュレートされた命令では、サンプリングの原理は同じです。

ランダム化したい5つのテンプレートNAND実装があるとします。

//! (Not a) Or (Not b)
template <typename T>
FORCEINLINE T Nand_1(volatile T a, volatile T b) {
  return Or<T>(Not<T>(a), Not<T>(b));
}
//! Not(a And b)
template <typename T>
FORCEINLINE T Nand_2(volatile T a, volatile T b) {
  return Not<T>(And<T>(a, b));
}
//! Not ((Not a) Nor (Not b))
template <typename T>
FORCEINLINE T Nand_3(volatile T a, volatile T b) {
  return Not<T>(Nor<T>(Not<T>(a), Not<T>(b)));
}
//! (Not a) Or (Not b)
template <typename T>
FORCEINLINE T Nand_4(volatile T a, volatile T b) {
  return Or<T>(Not<T>(a), Not<T>(b));
}
//! ((Not a) And b) Xor (Not b)
template <typename T>
FORCEINLINE T Nand_5(volatile T a, volatile T b) {
  return Xor<T>(And<T>(Not<T>(a), b), Not<T>(b));
}

これを行うには、次の手順を実行するだけです。

template <typename T, const uint64_t n>
FORCEINLINE volatile T NandR(volatile T a, volatile T b) {
  switch (n % 5) {
    case 0:
      return Nand_1<T>(a, b);
    case 1:
      return Nand_2<T>(a, b);
    case 2:
      return Nand_3<T>(a, b);
    case 3:
      return Nand_4<T>(a, b);
    case 4:
      return Nand_5<T>(a, b);
    default:
      return Nand_1<T>(a, b);
  }
}

次のようなコードで呼び出すことができます。

NandR<uint64_t, RND>(a,b)

番号はコンパイル時にわかっているため、コンパイラーは1つの関数呼び出しのみを残し、RNDマクロからの結果に従って選択されます。RNDは、上で説明したxorshift + fnv1 + __COUNTER__を使用するマクロです。また、便宜上、コードの可読性を向上させるために必要な演算子のオーバーロードを含むプリミティブテンプレートクラスCNNInt <>が作成されました。

次の関数について、生成されたアセンブラコードと2つの異なるコンパイルのDIFFを考えます。

uint64_t diff_me(uint64_t x) {
  using u64 = uint64_t;
  CNNInt<u64> r(std::chrono::system_clock::now().time_since_epoch().count());
  CNNInt<u64> test(x);
  test++;
  test--;
  test|=r;
  test^=r;
  test&=r;
  test^=r;
  test = ~test;
  return test.value();
}
