for (let i = 0; i <= l; i++) {

任意の色を取得する関数は次のようになります。

function getRandomColor() {
    let letters = '0123456789abcdef',
        color = '#'
    for (let i = 0; i < 6; i++) {
        color += letters[Math.trunc(Math.random() * 16)]
    }
    return color
}

しかし、簡単な方法を探しているわけではないので、次のようになります。

(Math.random()*0xffffff<<0).toString(16)

ビット単位の左シフトを使用し、16進数の文字列にキャストします。

次に、作成したパーティクルがアニメを使用してアニメーション化されます。

anime({
    //  
    targets: document.querySelectorAll('.dot'),

    loop: false,
    //   
    //     ()

    // 
    //

        },
        {
            value: 0,
            duration: 1200
        }
    ],

    //   ()
    // 
    width: {
        value: 2,
        duration: 500,
        delay: anime.stagger(2)
    },
    // 
    height: {
        value: 2,
        duration: 500,
        delay: anime.stagger(2)
    },

    //  
    //    
    translateX: {
        value: () => anime.random(-30, 30),
        duration: 1500,
        delay: anime.stagger(2)
    },
    //    y
    translateY: {
        value: () => anime.random(-30, 30),
        duration: 1500,
        delay: anime.stagger(2)
    }
})
