private byte[] transformClass(Class classToTransform, byte[] b, Set<Settings> methodsToInstrument) {
        ClassPool pool = ClassPool.getDefault();

        CtClass cl = null;

            for (int i = 0; i < methods.length; i++) {

            b = cl.toBytecode();
        } catch (Exception e) {
            e.printStackTrace();
