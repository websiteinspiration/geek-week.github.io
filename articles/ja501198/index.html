IIFEモジュール：JSモジュールテンプレート


JSで変数を定義することにより、それをグローバル変数として定義します。つまり、このような変数は、現在のページにロードされたすべてのJSファイルで使用できます。

    //   
    let count = 0
    const increase = () => ++count
    const reset = () => {
        count = 0
        console.log(' .')
    }

    //   
    increase()
    reset()

グローバル名前空間の汚染を回避するために、無名関数を使用できます。

    (() => {
        let count = 0
        // ...
    })

ほら、これ以上グローバル変数はありません。ただし、関数内のコードは実行されません。

IIFE：即時関数式


関数内でコードを実行するfには、()how を使用して呼び出す必要がありますf()。無名関数内でコードを実行するために(() => {})も使用する必要があります()。次のようになります(() => {})()。

    (() => {
        let count = 0
        // ...
    })()

これはIIFE（すぐに関数式と呼ばれます）と呼ばれます。モジュールは次のように定義できます。

    //  IIFE 
    const iifeCounterModule = (() => {
        let count = 0
        return {
            increase: () => ++count,
            reset: () => {
                count = 0
                console.log(' .')
            }
        }
    })()

    //  IIFE 
    iifeCounterModule.increase()
    iifeCounterModule.reset()

モジュールコードをIIFEでラップします。無名関数はオブジェクトを返します。これは、エクスポートインターフェイスを置き換えます。グローバル変数は1つしかありません-モジュールの名前（またはその名前空間）。その後、モジュールの名前を使用してそれを呼び出す（エクスポートする）ことができます。これはJSモジュールテンプレートと呼ばれます。

輸入不純物


モジュールを定義するとき、いくつかの依存関係が必要になる場合があります。モジュラーテンプレートを使用する場合、各依存モジュールはグローバル変数です。依存モジュールは、無名関数内で定義するか、引数としてそれに渡すことができます。

    //  IIFE   
    const iifeCounterModule = ((dependencyModule1, dependencyModule2) => {
        let count = 0
        return {
            increase: () => ++count,
            reset: () => {
                count = 0
                console.log(' .')
            }
        }
    })(dependencyModule1, dependencyModule2)

jQueryなどの以前のバージョンの人気のあるライブラリはこのテンプレートを使用していました（jQueryの最新バージョンはUMDモジュールを使用しています）。

モジュールを開く：JSモジュールテンプレートを開く


オープンモジュールテンプレートは、Christian Heilmannによって作成されました。このテンプレートもIIFEですが、すべてのインターフェイスを無名関数内のローカル変数として定義することに重点が置かれています。

    //   
    const revealingCounterModule = (() => {
        let count = 0
        const increase = () => ++count
        const reset = () => {
            count = 0
            console.log(' .')
        }

        return {
            increase,
            reset
        }
    })()

    //   
    revealingCounterModule.increase()
    revealingCounterModule.reset()

この構文により、各インターフェイスが何を担当するか（または何を行うか）を理解しやすくなります。

CJSモジュール：CommonJSモジュールまたはNode.jsモジュール


CommonJSは、当初ServerJSと呼ばれていましたが、モジュールを定義および使用するためのテンプレートです。Node.jsに組み込まれています。デフォルトでは、各JSファイルはCJSです。変数は、モジュール（ファイル）のエクスポートmoduleもexports提供します。この関数requireは、モジュールのロードと使用を提供します。次のコードは、CommonJS構文でのカウンターモジュールの定義を示しています。

    //  CommonJS : commonJSCounterModule.js
    const dependencyModule1 = require('./dependencyModule1')
    const dependencyModule2 = require('./dependencyModule2')

    let count = 0
    const increase = () => ++count
    const reset = () => {
        count = 0
        console.log(' .')
    }

    exports.increase = increase
    exports.reset = reset
    //  ()
    module.exports = {
        increase,
        reset
    }

このモジュールの使用方法は次のとおりです。

    //  CommonJS 
    const {
        increase,
        reset
    } = require('./commonJSCounterModule')
    increase()
    reset()
    // 
    const commonJSCounterModule = require('./commonJSCounterModule')
    commonJSCounterModule.increase()
    commonJSCounterModule.reset()

Node.jsランタイム（エンジン）では、ファイル内のコードを関数にラップしてこの変数を使用します。exports, module関数と変数は、パラメーターとして渡されますrequire。

    //  CommonJS 
    (function(exports, require, module, __filename, __dirname) {
        const dependencyModule1 = require('./dependencyModule1')
        const dependencyModule2 = require('./dependencyModule2')

        let count = 0
        const increase = () => ++count
        const reset = () => {
            count = 0
            console.log(' .')
        }

        module.exports = {
            increase,
            reset
        }

        return module.exports
    }).call(thisValue, exports, require, module, filename, dirname)

    //  CommonJS 
    (function(exports, require, module, __filename, __dirname) {
        const commonJSCounterModule = require('./commonJSCounterModule')
        commonJSCounterModule.increase()
        commonJSCounterModule.reset()
    }).call(thisValue, exports, require, module, filename, dirname)

AMDモジュールまたはRequireJSモジュール


AMD（非同期モジュール定義）は、モジュールを定義および使用するためのテンプレートです。RequireJSライブラリで使用されます。AMDにdefineは、モジュール名、依存関係名、およびファクトリー関数を受け入れるモジュール定義関数が含まれています。

    //  AMD 
    define('amdCounterModule', ['dependencyModule1', 'dependencyModule2'], (dependencyModule1, dependencyModule2) => {
        let count = 0
        const increase = () => ++count
        const reset = () => {
            count = 0
            console.log(' .')
        }

        return {
            increase,
            reset
        }
    })

またrequire、モジュールを使用するための関数も含まれています。

    //  AMD 
    require(['amdCounterModule'], amdCounterModule => {
        amdCounterModule.increase()
        amdCounterModule.reset()
    })

requireAMDはrequire、モジュールの名前とモジュール自体を関数の引数として取るという点でCommonJS とは異なります。

動的ローディング


関数のdefine目的も異なります。コールバック関数を受け取り、CommonJSのようなrequire関数をこの関数に渡します。コールバック関数の中で、モジュールを動的にロードするためにrequireが呼び出されます：

    //   AMD 
    define(require => {
        const dynamicDependencyModule1 = require('dependencyModule1')
        const dynamicDependencyModule2 = require('dependencyModule2')

        let count = 0
        const increase = () => ++count
        const reset = () => {
            count = 0
            console.log(' .')
        }

        return {
            increase,
            reset
        }
    })

CommonJSモジュールのAMDモジュール


上記の機能はdefine、ほかにrequire、変数を取ることができますexportsし、引数としてmodule。したがって、defineCommonJSのコードは次の場所で実行できます。

    //  AMD   CommonJS 
    define((require, exports, module) => {
        // CommonJS 
        const dependencyModule1 = require('dependencyModule1')
        const dependencyModule2 = require('dependencyModule2')

        let count = 0
        const increase = () => ++count
        const reset = () => {
            count = 0
            console.log(' .')
        }

        exports.increase = increase
        exports.reset = reset
    })

    //  AMD   CommonJS 
    define(require => {
        // CommonJS 
        const counterModule = require('amdCounterModule')
        counterModule.increase()
        counterModule.reset()
    })

UMDモジュール：ユニバーサルモジュール定義またはUmdJSモジュール


UMD（ユニバーサルモジュール定義）-さまざまなランタイム環境でのモジュールの動作を保証するためのテンプレートのセット。

AMD（RequireJS）とブラウザ用のUMD


次のコードは、AMD（RequireJS）とブラウザーの両方でモジュールを提供します。

    //  UMD   AMD (RequireJS)  
    ((root, factory) => {
        //   define AMD/RequireJS
        if (typeof define === 'function' && define.amd) {
            //  ,      
            define('umdCounterModule', ['dependencyModule1', 'dependencyModule2'], factory)
        } else {
            //  ,    
            //      (  Window)
            //    
            root.umdCounterModule = factory(root.dependencyModule1, root.dependencyModule2)
        }
    })(typeof self !== undefined ? self : this, (dependencyModule1, dependencyModule2) => {
        //  
        let count = 0
        const increase = () => ++count
        const reset = () => {
            count = 0
            console.log(' ')
        }

        return {
            increase,
            reset
        }
    })

複雑に見えますが、それは単なるIIFEです。無名関数は、defineAMD / RequireJSからの関数があるかどうかを決定します。

  • 場合はdefine検出され、出荷時の機能は、それを介して呼び出されます。
  • define見つからない場合、ファクトリー関数が直接呼び出されます。この時点で、引数rootはブラウザーのWindowオブジェクトです。グローバル変数（Windowオブジェクトのプロパティ）から依存モジュールを受け取ります。場合factoryモジュール戻り、それはまた、グローバル変数（Windowオブジェクトのプロパティ）となります。

AMD（RequireJS）およびCommonJS（Node.js）用のUMD


次のコードは、AMD（RequireJS）とCommonJS（Node.js）の両方でモジュールを提供します。

    (define => define((require, exports, module) => {
        //  
        const dependencyModule1 = require("dependencyModule1")
        const dependencyModule2 = require("dependencyModule2")

        let count = 0
        const increase = () => ++count
        const reset = () => {
            count = 0
            console.log("Count is reset.")
        }

        module.export = {
            increase,
            reset
        }
    }))( //   module  exports  CommonJS/Node.js
        //    define  AMD/RequireJS
        typeof module === 'object' && module.exports && typeof define !== 'function'
            ? // CommonJS/Node.js.    define
                factory => module.exports = factory(require, exports, module)
            : // AMD/RequireJS.   define
                define)

恐れることはありません。それはまたIIFEです。無名関数が呼び出されると、その引数は「評価」されます。引数を評価して実行環境を決定します（変数moduleとexportsCommonJS / Node.jsの存在、およびdefineAMD / RequireJSの関数によって定義されます）。

  • ランタイムがCommonJS / Node.jsの場合、無名関数の引数によって関数が手動で作成されますdefine。
  • ランタイムがAMD / RequireJSの場合、無名関数への引数はdefineその環境からの関数です。無名関数を実行すると、関数が確実に機能しますdefine。無名関数の内部では、モジュールを作成するために関数が呼び出されますdefine。

ESモジュール：ECMAScript2015またはES6モジュール


2015年、JS仕様のバージョン6は新しいモジュラー構文を導入しました。これはECMAScript 2015（ES2015）またはECMAScript 6（ES6）と呼ばれます。新しい構文の基礎はキーワードimportand exportです。次のコードは、名前付きおよびデフォルト（デフォルト）のインポート/エクスポートのためのESモジュールの使用を示しています。

    //  ES : esCounterModule.js  esCounterModule.mjs
    import dependencyModule1 from './dependencyModule1.mjs'
    import dependencyModule2 from './dependencyModule2.mjs'

    let count = 0
    //  
    export const increase = () => ++count
    export const reset = () => {
        count = 0
        console.log(' .')
    }
    //    
    export default {
        increase,
        reset
    }

ブラウザでモジュールファイルを使用するには、タグ<script>を追加してモジュールとして識別します<script type="module" src="esCounterModule.js"></script>。Node.jsでこのモジュールを使用するには、拡張子を.mjs次のように変更します。

    //  ES 
    //    
    import {
        increase,
        reset
    } from './esCounterModule.mjs'
    increase()
    reset()
    //      
    import esCounterModule from './esCounterModule.mjs'
    esCounterModule.increase()
    esCounterModule.reset()


ブラウザでの下位互換性のために<script>、属性を持つタグを追加できますnomodule：

    <script nomodule>
        警告（「サポートされていません。」）
    </ script>

ES動的モジュール：ECMAScript2020またはES11動的モジュール


JS 2020仕様の最新の11バージョンではimport、ESモジュールを動的に使用するための組み込み関数が導入されています。この関数はpromiseを返すので、次のようにモジュールを使用できますthen：

    //   ES    ,    
    import('./esCounterModule.js').then(({
        increase,
        reset
    }) => {
        increase()
        reset()
    })
    //      
    import('./esCounterModule.js').then(dynamicESCounterModule => {
        dynamicESCounterModule.increase()
        dynamicESCounterModule.reset()
    })

この関数importはpromiseを返すため、キーワードを使用できますawait。

    //   ES   async/await
    (async () => {
        //    
        const {
            increase,
            reset
        } = await import('./esCounterModule.js')
        increase()
        reset()

        //      
        const dynamicESCounterModule = await import('./esCounterModule.js')
        dynamicESCounterModule.increase()
        dynamicESCounterModule.reset()
    })

システムモジュール：SystemJSモジュール


SystemJSは、古いブラウザでESモジュールをサポートするためのライブラリです。たとえば、次のモジュールはES6構文を使用して記述されています。

    //  ES 
    import dependencyModule1 from "./dependencyModule1.js"
    import dependencyModule2 from "./dependencyModule2.js"
    dependencyModule1.api1()
    dependencyModule2.api2()

    let count = 0
    //  
    export const increase = function() {
        return ++count
    }
    export const reset = function() {
        count = 0
        console.log("Count is reset.")
    }
    //    
    export default {
        increase,
        reset
    }

このコードは、ES6構文をサポートしていないブラウザーでは機能しません。この問題の1つの解決策は、System.registerSystemJSライブラリインターフェイスを使用してコードを変換することです。

    //  SystemJS 
    System.register(['./dependencyModule1', './dependencyModule2'], function(exports_1, context_1) {
        'use strict'
        var dependencyModule1_js_1, dependencyModule2_js_1, count, increase, reset
        var __moduleName = context_1 && context_1.id
        return {
            setters: [
                function(dependencyModule1_js_1_1) {
                    dependencyModule1_js_1 = dependencyModule1_js_1_1
                },
                function(dependencyModule2_js_1_1) {
                    dependencyModule2_js_1 = dependencyModule2_js_1_1
                }
            ],
            execute: function() {
                dependencyModule1_js_1.default.api1()
                dependencyModule2_js_1.default.api2()
                count = 0
                //  
                exports_1('increase', increase = function() {
                    return ++count
                })
                exports_1('reset', reset = function() {
                    count = 0
                    console.log(' .')
                })
                //    
                exports_1('default', {
                    increase,
                    reset
                })
            }
        }
    })

新しいモジュラーES6構文はなくなりました。ただし、コードは古いブラウザで正常に動作します。この変換は、Webpack、TypeScriptなどを使用して自動的に実行できます。

動的モジュールの読み込み


SystemJSには、import動的インポートのための関数も含まれています。

    //  SystemJS   
    System.import('./esCounterModule.js').then(dynamicESCounterModule => {
        dynamicESCounterModule.increase()
        dynamicESCounterModule.reset()
    })

Webpackモジュール：CJS、AMD、ESモジュールのコンパイルとアセンブリ


Webpackはモジュールビルダーです。そのトランスパイラーは、CommonJS、AMD、ESモジュールを単一のバランスのとれたモジュラーテンプレートに組み合わせ、すべてのコードを1つのファイルに収集します。たとえば、次の3つのファイルでは、3つのモジュールが異なる構文を使用して定義されています。

    //  AMD : amdDependencyModule1.js
    define('amdDependencyModule1', () => {
        const api1 = () => {}
        return {
            api1
        }
    })

    //  CommonJS : commonJSDependencyModule2.js
    const dependencyModule2 = require('./commonJSDependencyModule2')
    const api2 = () => dependencyModule1.api1()
    exports.api2 = api2

    //  ES : esCounterModule.js
    import dependencyModule1 from './amdDependencyModule1'
    import dependencyModule2 from './commonJSDependencyModule2'

    let count = 0
    const increase = () => ++count
    const reset = () => {
        count = 0
        console.log(' .')
    }

    export default {
        increase,
        reset
    }

次のコードは、このモジュールの使用方法を示しています。

    //  ES : index.js
    import counterModule from './esCounterModule'
    counterModule.increase()
    counterModule.reset()

Webpackは、これらが異なるモジュラーシステムであるにもかかわらず、これらのファイルを1つのファイルに結合できますmain.js。

    ルート

            esCounterModule.js
            index.js
        webpack.config.js

WebpackはNode.jsに基づいているため、CommonJSのモジュール構文を使用します。でwebpack.config.js：

    const path = require('path')

    module.exports = {
        entry: './src/index.js',
        mode: 'none', //          
        output: {
            filename: 'main.js',
            path: path.resolve(__dirname, 'dist'),
        },
    }

コンパイルしてビルドするには、次のコマンドを実行する必要があります。

    npm install webpack webpack-cli --save-dev
    npx webpack --config webpack.config.js

Babelモジュール：ESモジュールの翻訳

内部モジュールと名前空間


TypeScriptにはキーワードmoduleともありnamespaceます。それらは内部モジュールと呼ばれます：

    module Counter {
        let count = 0
        export const increase = () => ++count
        export const reset = () => {
            count = 0
            console.log(' .')
        }
    }

    namespace Counter {
        let count = 0
        export const increase = () => ++count
        export const reset = () => {
            count = 0
            console.log(' .')
        }
    }

どちらもJSオブジェクトに置き換えられます。

    var Counter;
    (function(Counter) {
        var count = 0
        Counter.increase = function() {
            return ++count
        }
        Counter.reset = function() => {
            count = 0
            console.log(' .')
        }
    })(Counter || (Counter = {}))

TypeScriptモジュールと名前空間は、セパレーターを介していくつかのレベルの入れ子を持つことができます.。

    module Counter.Sub {
        let count = 0
        export const increase = () => ++count
    }

    namespace Counter.Sub {
        let count = 0
        export const increase = () => ++count
    }

サブモジュールとサブ名前空間はオブジェクトのプロパティに置き換えられます：

    var Counter;
    (function(Counter) {
        var Sub;
        (function(Sub) {
            var count = 0
            Sub.increase = function() {
                return ++count
            }
        })(Sub = Counter.Sub || (Counter.Sub = {}))
    })(Counter || (Counter = {}))

TypeScriptモジュールと名前空間は、ステートメントでも使用できますexport。

    module Counter {
        let count = 0
        export module Sub {
            export const increase = () => ++count
        }
    }

    module Counter {
        let count = 0
        export namespace Sub {
            export const increase = () => ++count
        }
    }

上記のコードは、サブモジュールとサブ名前空間にも変換されます。

    var Counter;
    (function(Counter) {
        var count = 0
        var Sub;
        (function(Sub) {
            Sub.increase = function() {
                return ++count
            }
        })(Sub = Counter.Sub || (Counter.Sub = {}))
    })(Counter || (Counter = {}))
