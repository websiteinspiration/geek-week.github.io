UARTパート2によるARM Cortex-Mマイクロコントローラーのデバッグ

前回の記事では、DebugMon割り込みとそれに関連するレジスタについて説明しました。

この記事では、UARTのデバッガーの実装について説明します。

低レベル部品


こことここに、GDBサーバーのリクエストとレスポンスの構造の説明があります。単純に見えますが、次の理由により、マイクロコントローラーには実装しません。

  • ビッグデータの冗長性。アドレス、レジスタの値、変数は16進文字列としてエンコードされ、メッセージ量が2倍に増加します
  • メッセージの解析と収集には追加のリソースが必要になります
  • パケットの終わりの追跡は、タイムアウト（タイマーがビジーになります）またはUART割り込みに費やされる時間を増やす複雑な自動マシンのいずれかで必要です

最も簡単で最速のデバッグモジュールを取得するために、制御シーケンスを含むバイナリプロトコルを使用します。

  • 0xAA 0xFF-フレームの開始
  • 0xAA 0x00-フレームの終わり
  • 0xAA 0xA5-割り込み
  • 0xAA 0xAA-0xAAに置き換えられました

受信中にこれらのシーケンスを処理するには、4つの状態を持つオートマトンが必要です。

  • ESC文字を待っています
  • フレームシーケンスの開始の2番目の文字を待機しています
  • データ受信
  • Escキャラクターが最後に受け入れられたとき

しかし、状態を送信するには、すでに7が必要です。

  • 最初のバイトの送信フレームの開始
  • 2番目のバイトの送信フレームの開始
  • データを送信する
  • フレームの終わりを送信しています
  • Esc文字置換の送信
  • 最初の割り込みバイトを送信する
  • 2番目の割り込みバイトの送信

すべてのモジュール変数が配置される構造の定義を書きましょう：

typedef struct 
{    
  // disable receive data
  unsigned tx:1;
  // program stopped
  unsigned StopProgramm:1;
  union {
    enum rx_state_e 
    {
      rxWaitS = 0, // wait Esc symbol
      rxWaitC = 1, // wait Start of frame
      rxReceive = 2, // receiving
      rxEsc = 3, // Esc received
    } rx_state;
    enum tx_state_e 
    {
      txSendS = 0, // send first byte of Start of frame
      txSendC = 1, // send second byte
      txSendN = 2, // send byte of data
      txEsc = 3,   // send escaped byte of data
      txEnd = 4,   // send End of frame
      txSendS2 = 5,// send first byte of Interrupt
      txBrk = 6,   // send second byte
    } tx_state;
  };
  uint8_t pos; // receive/send position
  uint8_t buf[128]; // offset = 3
  uint8_t txCnt; // size of send data
} dbg_t;
#define dbgG ((dbg_t*)DBG_ADDR) //   ,

作業は半二重モードで実行されるため、受信機と送信機の状態は1つの変数に結合されます。これで、割り込みハンドラーを使用してオートマトン自体を作成できます。

UARTハンドラー
void USART6_IRQHandler(void)
{
  if (((USART6->ISR & USART_ISR_RXNE) != 0U)
      && ((USART6->CR1 & USART_CR1_RXNEIE) != 0U))
  {
    rxCb(USART6->RDR);
    return;
  }

  if (((USART6->ISR & USART_ISR_TXE) != 0U)
      && ((USART6->CR1 & USART_CR1_TXEIE) != 0U))
  {
    txCb();
    return;
  }
}

void rxCb(uint8_t byte)
{
  dbg_t* dbg = dbgG; // debug vars pointer
  
  if (dbg->tx) // use half duplex mode
    return;
  
  switch(dbg->rx_state)
  {
  default:
  case rxWaitS:
    if (byte==0xAA)
      dbg->rx_state = rxWaitC;
    break;
  case rxWaitC:
    if (byte == 0xFF)
      dbg->rx_state = rxReceive;
    else
      dbg->rx_state = rxWaitS;
    dbg->pos = 0;
    break;
  case rxReceive:
    if (byte == 0xAA)
      dbg->rx_state = rxEsc;
    else
      dbg->buf[dbg->pos++] = byte;
    break;
  case rxEsc:
    if (byte == 0xAA)
    {
      dbg->buf[dbg->pos++] = byte;
      dbg->rx_state  = rxReceive;
    }
    else if (byte == 0x00)
    {
      parseAnswer();
    }
    else
      dbg->rx_state = rxWaitS;
  }
}

void txCb()
{
  dbg_t* dbg = dbgG;
  switch (dbg->tx_state)
  {
  case txSendS:
    USART6->TDR = 0xAA;
    dbg->tx_state = txSendC;
    break;
  case txSendC:
    USART6->TDR = 0xFF;
    dbg->tx_state = txSendN;
    break;
  case txSendN:
    if (dbg->txCnt>=dbg->pos)
    {
      USART6->TDR = 0xAA;
      dbg->tx_state = txEnd;
      break;
    }
    if (dbg->buf[dbg->txCnt]==0xAA)
    {
      USART6->TDR = 0xAA;
      dbg->tx_state = txEsc;
      break;
    }
    USART6->TDR = dbg->buf[dbg->txCnt++];
    break;
  case txEsc:
    USART6->TDR = 0xAA;
    dbg->txCnt++;
    dbg->tx_state = txSendN;
    break;
  case txEnd:
    USART6->TDR = 0x00;
    dbg->rx_state = rxWaitS;
    dbg->tx = 0;
    CLEAR_BIT(USART6->CR1, USART_CR1_TXEIE);
    break;
  case txSendS2:
    USART6->TDR = 0xAA;
    dbg->tx_state = txBrk;
    break;
  case txBrk:
    USART6->TDR = 0xA5;
    dbg->rx_state = rxWaitS;
    dbg->tx = 0;
    CLEAR_BIT(USART6->CR1, USART_CR1_TXEIE);
    break;
  }
}


ここではすべてが非常に単純です。発生したイベントに応じて、割り込みハンドラーは受信マシンまたは送信マシンを呼び出します。すべてが機能することを確認するために、1バイトで応答するパケットハンドラを記述します。

void parseAnswer()
{
  dbg_t* dbg = dbgG;
  dbg->pos = 1;
  dbg->buf[0] = 0x33;
  dbg->txCnt = 0;
  dbg->tx = 1;
  dbg->tx_state = txSendS;
  SET_BIT(USART6->CR1, USART_CR1_TXEIE);
}

モジュールが読み取り/書き込み操作中にBusFault例外を発生させないようにするには、M3以降で使用する場合はそれをマスクするか、M0のHardFaultハンドラーを作成する必要があります。

安全なmemcpy
int memcpySafe(uint8_t* to,uint8_t* from, int len)
{
    /* Cortex-M3, Cortex-M4, Cortex-M4F, Cortex-M7 are supported */
    static const uint32_t BFARVALID_MASK = (0x80 << SCB_CFSR_BUSFAULTSR_Pos);
    int cnt = 0;

    /* Clear BFARVALID flag by writing 1 to it */
    SCB->CFSR |= BFARVALID_MASK;

    /* Ignore BusFault by enabling BFHFNMIGN and disabling interrupts */
    uint32_t mask = __get_FAULTMASK();
    __disable_fault_irq();
    SCB->CCR |= SCB_CCR_BFHFNMIGN_Msk;

    while ((cnt<len))
    {
      *(to++) = *(from++);
      cnt++;
    }

    /* Reenable BusFault by clearing  BFHFNMIGN */
    SCB->CCR &= ~SCB_CCR_BFHFNMIGN_Msk;
    __set_FAULTMASK(mask);

    return cnt;
}

クリーニングは、設定されたブレークポイントを検索することによって行われます。実行を停止すると、現在のPCにブレークポイントが設定されます。UART割り込みを終了すると、カーネルはすぐにDebugMon_Handlerに入ります。

DebugMonハンドラー自体は非常に単純です。

  • 1.実行を停止するフラグが設定されます。
  • 2.設定されたすべてのブレークポイントがクリアされます。
  • 3.コマンドへの応答の送信がuartで完了するのを待つ（実行する時間がなかった場合）
  • 4.割り込みシーケンスの送信が始まります
  • 5.ループでは、送信フラグと受信マシンのハンドラーが停止フラグが下がるまで呼び出されます。

void DebugMon_Handler(void)
{
  dbgG->StopProgramm = 1; //   
  
  for (int i=0;i<NUMOFBKPTS;i++) //  breakpoint
    FP->FP_COMP[i] = 0;
  
  while (USART6->CR1 & USART_CR1_TXEIE) //    
    if ((USART6->ISR & USART_ISR_TXE) != 0U)
      txCb();

  
  dbgG->tx_state = txSendS2; //   Interrupt 
  dbgG->tx = 1;

  while (dbgG->StopProgramm) //       
  {
  	//   UART  
    if (((USART6->ISR & USART_ISR_RXNE) != 0U)
        && ((USART6->CR1 & USART_CR1_RXNEIE) != 0U))
