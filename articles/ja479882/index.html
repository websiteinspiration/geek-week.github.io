// TimeValue is an implementation of flag.Value interface.
type TimeValue struct {
    P      *time.Time
    Layout string
}

func (t *TimeValue) Set(s string) error {
    v, err := time.Parse(t.Layout, s)
    if err == nil {
        (*t.P) = v
    }
    return err
}

func (t *TimeValue) String() string {
    return t.P.Format(t.Layout)
}
