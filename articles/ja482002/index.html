    start:=time.Now()
    md,_:=metadata.FromIncomingContext(ctx)

    // Calls the handler
    reply, err := handler(ctx, req)

    var traceId,userId,userRole string
    if len(md["trace-id"])>0{
        traceId=md["trace-id"][0]
    }
    if len(md["user-id"])>0{
        userId=md["user-id"][0]
    }
    if len(md["user-role"])>0{
        userRole=md["user-role"][0]
    }

    app.AccesLog(msg)

    return reply, err
}
