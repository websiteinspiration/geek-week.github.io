F＃、接続された画像コンポーネントのマーキングアルゴリズム

バイナリイメージ内の接続コンポーネントの検索とマーキングは、イメージの分析と処理の基本的なアルゴリズムの1つです。特に、このアルゴリズムをマシンビジョンで使用して、画像内の一般的な構造を検索およびカウントし、その後の分析を行うことができます。

この記事のフレームワークでは、接続されたコンポーネントをマークするための2つのアルゴリズムが考慮されます。プログラミング言語F＃での実装が示されています。また、アルゴリズムの比較分析、弱点の検索。アルゴリズムの実装のソース資料は、「コンピュータビジョン」という本（著者L.シャピロ、J。ストックマン。

再帰的なラベル付けアルゴリズム


まず、再帰ベースのアルゴリズムを最も明白なものとして見ていきます。このようなアルゴリズムの動作原理は次のようになります：

関連するコンポーネントをマークするラベルを初期値に割り当てます。

  1. 座標0x0（左上隅）から始めて、テーブル内のすべてのポイントを並べ替えます。次のポイントを処理した後、ラベルを1つ増やします
  2. 現在の座標のピクセルが1の場合、このポイントの4つの近傍すべて（上、下、右、左）をスキャンします
  3. 近傍点も1の場合、関数は、配列または点0の境界に到達するまで、近傍を処理するためにそれ自体を再帰的に呼び出します
  4. ネイバーの再帰的な処理中に、ポイントがまだ処理されておらず、現在のマーカーでマークされていることを確認する必要があります。

したがって、配列内のすべてのポイントを並べ替えて、それぞれの再帰アルゴリズムを呼び出し、検索の最後に完全にマークされたテーブルを取得します。特定のポイントがすでにマークされているかどうかのチェックがあるため、各ポイントは一度だけ処理されます。再帰の深さは、現在のポイントに接続されている近傍の数にのみ依存します。このアルゴリズムは、大きな画像の塗りつぶしがない画像に適しています。つまり、細くて短い線があり、その他はすべて処理されていない背景画像です。さらに、そのようなアルゴリズムの速度は、行ごとのアルゴリズムよりも速い場合があります（この記事の後半で説明します）。

2次元のバイナリイメージは2次元の行列なので、アルゴリズムを作成するときは、ここで読むことができる別の記事を作成した行列の概念を操作します。

let recMarkingOfConnectedComponents matrix =

        let copy = Matrix.clone matrix

        let (|Value|Zero|Out|) (x, y) = 
            if x < 0 || y < 0
                || x > (copy.values.[0, *].Length - 1)
                || y > (copy.values.[*, 0].Length - 1) then
                Out
            else
                let row = copy.values.[y, *]
                match row.[x] with
                    | 0 -> Zero
                    | v -> Value(v)

        let rec markBits x y value =
            match (x, y) with
            | Value(v) ->
                if value > v then
                    copy.values.[y, x] <- value
                    markBits (x + 1) y value
                    markBits (x - 1) y value
                    markBits x (y + 1) value
                    markBits x (y - 1) value
            | Zero | Out -> ()

        let mutable value = 2
        copy.values
        |> Array2D.iteri (fun y x v -> if v = 1 then
                                            markBits x y value
                                            value <- value + 1)

        copy

再帰的アルゴリズムの利点には、記述と理解が簡単なことが含まれます。これは、大きな背景を含み、ピクセルの小さなグループ（ライン、小さなスポットなど）が点在するバイナリイメージに最適です。この場合、アルゴリズムは画像全体を1つのパスで処理するため、画像処理速度が高速になる場合があります。

欠点も明らかです-これは再帰です。画像が大きな点で構成されている場合や完全に塗りつぶされている場合は、処理速度が大幅に低下し（桁違い）、スタックの空き領域が不足しているためにプログラムがまったくクラッシュするリスクがあります。

残念ながら、テール再帰には2つの要件があるため、このアルゴリズムではテール再帰の原則を適用できません。

  1. 各再帰フレームは計算を実行し、その結果をスタックに送信されるアキュムレータに追加する必要があります。
  2. 1つの再帰フレームに複数の再帰関数呼び出しがあってはなりません。

私たちの場合、再帰は再帰の次のフレームに渡すことができる最終的な計算を実行せず、単に2次元配列を変更します。原則として、これは、単一の数値を返すだけで回避できますが、それ自体は何も意味しません。また、ポイントの4つの近傍すべてを処理する必要があるため、4つの呼び出しを再帰的に行います。

procedure union(X, Y, PARENT);
{
j := X;
k := Y;
while PARENT[j] <> 0
    j := PARENT[j];
while PARENT[k]] <> 0
    k := PARENT[k];

if j <> k then PARENT[k] := j;
}

F＃実装は次のようになります
let union x y =

F＃コードの実装は次のとおりです
let markingOfConnectedComponents matrix =

    // up to 10 markers
    let parents = Array2D.init 2 10 (fun x y -> if x = 0 then y+1 else 0)

    // create a zero initialized copy
    let step1 = (Matrix.cloneO matrix).values

    let rec find x =
        let index = Array.tryFindIndex ((=)x) parents.[0, *]
        match index with
        | Some(i) -> 
            match parents.[1, i] with
            | p when p <> 0 -> find p
            | _ -> x
        | None -> x

    let union x y =
        let j = find x
        let k = find y
        if j <> k then parents.[1, k-1] <- j

    // returns up and left neighbors of pixel
    let neighbors_labels x y =
        match (x, y) with
        | (0, 0) -> []
        | (0, _) -> [step1.[0, y-1]]
        | (_, 0) -> [step1.[x-1, 0]]
        | _ -> [step1.[x, y-1]; step1.[x-1, y]]
        |> List.filter ((<>)0)

    let mutable label = 0
    matrix.values
    |> Array2D.iteri (fun x y v ->
                        if v = 1 then
                            let n = neighbors_labels x y
                            let m = if n.IsEmpty then
                                        label <- label + 1
                                        label
                                    else
                                        n |> List.min
                            n |> List.iter (fun v -> if v <> m then union m v)
                            step1.[x, y] <- m)

    let step2 = matrix.values
                |> Array2D.mapi (fun x y v ->
                        if v = 1 then step1.[x, y] <- find step1.[x, y]
                        step1.[x, y])

    { values = step2 }

上記ですべてが説明されているため、コードの行ごとの説明はもはや意味がありません。ここでは、F＃コードを疑似コードにマップして、それがどのように機能するかを明確にすることができます。
