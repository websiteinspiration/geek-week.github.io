Perangkat keras


Jadi, untuk membuat prototipe yang berfungsi minimal, kita perlu:

  1. Sepotong papan atau apa pun yang dapat Anda pasang semua komponen
  2. Hampir semua mikrokontroler. Saya menggunakan arduino uno sebagai opsi termudah
  3. Modul GPS (kami mengambil tanggal, waktu dan koordinat dari itu). Secara teoritis, Anda dapat mengambil modul jam waktu nyata sebagai gantinya, tetapi kemudian Anda harus memasukkan koordinat Anda secara manual dan perangkat tidak akan berfungsi di luar kotak, tetapi waktu mulai dingin akan sangat berkurang
  4. Dua servos, atau lebih baik lagi dua motor stepper
  5. Laser penunjuk
  6. Agak berbeda: besi solder, perekat panas meleleh, papan tempat memotong roti, kabel, tombol, kapasitor, dan tangan langsung.

Tidak ada trik untuk dirakit: hanya memperbaiki semua komponen di pangkalan dengan perekat meleleh panas.
Coba saja untuk memastikan bahwa sumbu servos dan laser saling tegak lurus mungkin, ini akan mengurangi kesalahan menunjuk.

Koneksi


Tidak ada yang rumit di sini:

  • GPS terhubung melalui uart (hanya Rx yang diperlukan, karena kita tidak perlu mengirim apa pun ke modul
  • servos - pada pin 10 (sumbu azimuth) dan 11 (sumbu tinggi)
  • tombol - 2 pin
  • kekuatan untuk semua modul
  • opsional - kapasitor daya

TinyGPSPlus gps;
ServoSmooth yaw;
ServoSmooth pitch;
GButton but(2);
int yr = 0, mo = 0, d = 0, h = 0, m = 0, s = 0;
float phi = 0, lambda = 0;
float az = 0, height = 0;
int counter = 0;
float alpha = 0.0, delta = 0.0;
float sunalpha = 0, sundelta = 0;
float Coordinates[10][2] =
{
  {0, 0},
  {sunalpha * 360 / 2 / PI, sundelta * 360 / 2 / PI}, //sun
  {297.9458, 8.9233}, //altair
  {279.4083, 38.8038}, //vega

int Days(int d, int m, int y)// ,        (21.03)
{
  int days = 0;
  int yearNotLeap[12] = { 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
  int yearIsLeap[12] = { 31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };

  if (((y % 4 == 0) && (y % 100 != 0)) || (y % 400 == 0))
  {
    for (int i = 0; i < m - 1; i++)
    {
      days += yearIsLeap[i];
    }
  }
  else
  {
    for (int i = 0; i < m - 1; i++)
    {
      days += yearNotLeap[i];
    }
  }
  if (m == 1) {

    return d - 81;
  }
  else
    return days + d - 81;
}
void GpsGetData()
{
  while (Serial.available() > 0)
    gps.encode(Serial.read());
  if (gps.location.isUpdated())
  {
    phi = gps.location.lat(); //   (double)
    lambda = gps.location.lng(); //    (double)
    yr = gps.date.year(); //  (2000+) (u16)
    mo = gps.date.month(); //  (1-12) (u8)
    d = gps.date.day(); //  (1-31) (u8)
    h = gps.time.hour(); //  (0-23) (u8)
    m = gps.time.minute(); //  (0-59) (u8)
    s = gps.time.second(); //  (0-59) (u8)
  }
}
void CalculateParams()
{
  float _time = (h + (float)m / 60 + (float)s / 3600) * 360.0 / 24.0;//   
  float sunlambda = (float)Days(d, mo, yr) * 360 / 365; //      
  sundelta = asin(0.398749 * sin(sunlambda * 2 * PI / 360));//    
  sunalpha = asin(tan(sundelta) / 0.434812); //      
  float tS = _time + lambda - 180 + sunalpha * 360 / (2 * PI) - alpha; //     
  height = 360 / 2 / PI * asin(sin(phi * 2 * PI / 360) * sin(delta * 2 * PI / 360) + cos(phi * 2 * PI / 360) * cos(delta * 2 * PI / 360) * cos(tS * 2 * PI / 360));// 
  az = 360 / 2 / PI * asin(sin(tS * 2 * PI / 360) * cos(delta * 2 * PI / 360) / cos(height * 2 * PI / 360)); //   (   )
  if (tS > 90 || tS < -90)
  {
    az = 180 - az;
  }
 // if (az > 180 && az < 270)
  //  az = az - 360;
}

void setup() {
  Serial.begin(9600);
  yaw.attach(10, 90);
  pitch.attach(11, 180);
  yaw.setSpeed(20);
  yaw.setAccel(0.1);
  pitch.setSpeed(20);
  pitch.setAccel(0.1);
  yaw.setAutoDetach(false);
  pitch.setAutoDetach(false);
  pinMode(LED_BUILTIN, OUTPUT);
}

void loop() {
  but.tick();
  yaw.tick();
  pitch.tick();
  if (but.isSingle())
  {
    counter += 1;
    if (counter == 9)
      counter = 0;
  }
  alpha = Coordinates[counter][0];
  delta = Coordinates[counter][1];
  GpsGetData();
  CalculateParams();
  digitalWrite(LED_BUILTIN, gps.location.isValid());
  if (az > 90 && az < 180)
  {
    yaw.setTargetDeg(270 - az);
    pitch.setTargetDeg(height);
  }
  if (az < -90 && az > -180)
  {
    yaw.setTargetDeg(-90 - az);
    pitch.setTargetDeg(height);
  }
  if (az < 90 && az > -90)
  {
    yaw.setTargetDeg(90.0 - az);
    pitch.setTargetDeg(180 - height);
  }
}

... dan hanya ada satu saat - fungsi CalculateParams ()
Apa fungsi ini harus dilakukan: menerima koordinat bintang dalam sistem ekuatorial (kenaikan dan deklinasi kanan), waktu dan koordinat pengamat, dan memberikan ketinggian dan azimuth objek, mis. pada dasarnya menerjemahkan koordinat bintang dari sistem khatulistiwa (di mana bintang-bintang tidak bergerak) ke horisontal (di mana bintang-bintang bergerak pada siang hari).

Ini diwujudkan dengan menggunakan rumus trigonometri bola , serta astronomi bola dalam ruang hampa .

Algoritma adalah sebagai berikut:

  1. menghitung garis bujur ekliptika matahari
  2. menghitung deklinasi matahari
  3. menghitung kenaikan matahari langsung
  4. menghitung sudut jam bintang
  5. menghitung ketinggian dan azimuth

Estimasi kesalahan

  1. Lengkungan tangan - ini adalah faktor terpenting bagi saya

  3. posisi nol azimuth yang tidak akurat (saya ingat bahwa para astronom menghitung azimuth dari selatan searah jarum jam)
