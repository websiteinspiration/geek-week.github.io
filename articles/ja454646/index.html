比較のルールを定義するPythonオブジェクトには、6つの「魔法の」メソッドがあります。

  • __lt__ ために <
  • __gt__ ために >
  • __le__ ために <=
  • __ge__ ために >=
  • __eq__ ために ==
  • __ne__ ために !=

これらのメソッドのいずれかが定義または返されないNotImplemented場合、次のルールが適用されます。

  • a.__lt__(b) と同じ b.__gt__(a)
  • a.__le__(b) と同じ b.__ge__(a)
  • a.__eq__(b)同じことnot a.__ne__(b)（この場合には、そのノートaおよびbスワップません）

ただし、条件はa >= b、それをa != b自動的に意味するものではありませんa > b。デコレータは、functools.total_orderingに基づいて、すべての6つのメソッドを作成__eq__し、これらの1： 、、__lt__ または。__gt____le____ge__

from functools import total_ordering           
                                               
                                               
@total_ordering                                
class User:                                    
    def __init__(self, pk, name):              
        self.pk = pk                           
        self.name = name                       
                                               
    def __le__(self, other):                   
        return self.pk <= other.pk             
                                               
    def __eq__(self, other):                   
        return self.pk == other.pk             
                                               
                                               
assert User(2, 'Vadim') < User(13, 'Catherine')


関数の装飾バージョンと非装飾バージョンの両方を使用する必要がある場合があります。これを実現する最も簡単な方法は、特別な装飾構文（記号を使用）を使用せず@、装飾関数を手動で作成する場合です。

import json

def ensure_list(f):
    def decorated(*args, **kwargs):
        result = f(*args, **kwargs)

        if isinstance(result, list):
            return result
        else:
            return [result]

    return decorated

def load_data_orig(string):
    return json.loads(string)
  
load_data = ensure_list(load_data_orig)

print(load_data('3'))     # [3]
print(load_data_orig('4')) 4

すべてのデコレータがを介して作成されているfunctools.wraps場合、属性__wrapped__を使用して装飾されていない関数にアクセスできます。

import json
from functools import wraps

def ensure_list(f):
    @wraps(f)
    def decorated(*args, **kwargs):
        result = f(*args, **kwargs)

        if isinstance(result, list):
            return result
        else:
            return [result]

    return decorated

@ensure_list
def load_data(string):
    return json.loads(string)

print(load_data('3'))      # [3]
print(load_data.__wrapped__('4')) # 4

ただし、このアプローチは、複数のデコレーターで装飾された関数では機能しないことに注意してください。__wrapped__適用された各デコレーターを参照する必要があります。

def ensure_list(f):
    ...

def ensure_ints(f):
    @wraps(f)
    def decorated(*args, **kwargs):
        result = f(*args, **kwargs)
        return [int(x) for x in result]

    return decorated

@ensure_ints
@ensure_list
def load_data(string):
    return json.loads(string)

for f in (
    load_data,
    load_data.__wrapped__,
    load_data.__wrapped__.__wrapped__,
):
    print(repr(f('"4"')))

結果：

[4]
['4']
'4'

上記@saving_origのデコレータは、引数として別のデコレータを取ります。そして、それがパラメーター化されるのであれば？パラメータ化されたデコレータは実際のデコレータを返す関数であるため、この状況は自動的に処理されます。

import json
from functools import wraps

def saving_orig(another_decorator):
    def decorator(f):
        decorated = another_decorator(f)
        decorated.orig = f
        return decorated

    return decorator

def ensure_ints(*, default=None):
    def decorator(f):
        @wraps(f)
        def decorated(*args, **kwargs):
            result = f(*args, **kwargs)
            ints = []
            for x in result:
                try:
                    x_int = int(x)
                except ValueError:
                    if default is None:
                        raise
                    else:
                        x_int = default
                ints.append(x_int)
            return ints
        return decorated
    return decorator

@saving_orig(ensure_ints(default=0))
def load_data(string):
    return json.loads(string)

print(repr(load_data('["2", "3", "A"]')))
print(repr(load_data.orig('["2", "3", "A"]')))

別の解決策があります。ある関数から別の関数にsaving_orig渡すようにorigします。

def saving_orig(another_decorator):
    def decorator(f):
        decorated = another_decorator(f)
        if hasattr(f, 'orig'):
            decorated.orig = f.orig
        else:
            decorated.orig = f
        return decorated

    return decorator

@saving_orig(ensure_ints(default=42))
@saving_orig(ensure_list)
def load_data(string):
    return json.loads(string)

デコレータが複雑になりすぎた場合は、関数からメソッドを使用してクラスに変換するのが最善__call__です。

class SavingOrig:
    def __init__(self, another_decorator):
        self._another = another_decorator
  
    def __call__(self, f):
        decorated = self._another(f)
        if hasattr(f, 'orig'):
            decorated.orig = f.orig
        else:
            decorated.orig = f
        return decorated

saving_orig = SavingOrig

最後の行では、キャメルケースでクラスに名前を付け、スネークケースでデコレータ名を保存できます。

装飾された関数を変換する代わりに、別の呼び出されたクラスを作成し、関数の代わりに関数のインスタンスを返すことができます。

class CallableWithOrig:
    def __init__(self, to_call, orig):
        self._to_call = to_call
        self._orig = orig
    
    def __call__(self, *args, **kwargs):
        return self._to_call(*args, **kwargs)

    @property
    def orig(self):
        if isinstance(self._orig, type(self)):
            return self._orig.orig
        else:
            return self._orig

class SavingOrig:
    def __init__(self, another_decorator):
        self._another = another_decorator
  
    def __call__(self, f):
        return CallableWithOrig(self._another(f), f)

saving_orig = SavingOrig
