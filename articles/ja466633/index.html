SSHチャット

こんにちは、ハブル。コンソールチャットは素晴らしいことですが、フロントエンドでは、バックエンドで同じようにしたい場合はどうでしょうか。はいの場合、この記事はあなたのためです。しかし、どのツールがバックエンドでよく使用されますか？そうです、sshなので、私はsshchatを表しています。


それはどのように見えますか


サーバーのどこかで、ノードのプログラムが回転しています。
誰かがチャットに接続することを望むとすぐに、彼は入ります：


ssh server -p 8022

その後、システムはパスワードを要求し、それを特殊ファイル内のパスワードで検証します。パスワードが一致した場合、チャットに接続します（ユーザーは以前に100件のメッセージを受信し、他のすべてのユーザーは彼が接続していることを確認します）。


それから彼は他の人のメッセージを受け取り、自分のメッセージを書くことができます。


ここにもっと興味深いメッセージがあります：


@box{@color(red){Red text in box}}

ボックスに赤いテキストを送信します。


始めましょう


sshを使用するには、https：//www.npmjs.com/package/ssh2を使用します。
フォーマットには、チョークとボクセンを使用します。
それらをインストールしてください：


npm i ssh2 chalk boxen

コード自体は、このメッセージパーサー（GitHub）の最も重要な部分の1つです。


//  chalk  boxen
const chalk = require('chalk');
const boxen = require('boxen');

//         @
//   2          
let methods = {
  color: function(args, text) {
    return chalk.keyword(args)(text);
  },

  bold: function(args, text) {
    return chalk.bold(text);
  },

  underline: function(args, text) {
    return chalk.underline(text);
  },

  hex: function(args, text) {
    return chalk.hex(args)(text);
  },

  box: function(args, text) {
    return boxen(text, {
      borderStyle: 'round',
      padding: 1,
      borderColor: 'blueBright'
    });
  }
};

//   
function parseAndExecute(str) {
  let pos = 0;
  let stage = 0;
  let nS = '';
  let bufs = ['', '', '', ''];
  let level = 0;

  while (pos < str.length) {
    let symbol = str[pos];
    pos++;

    if (symbol == '\\' && '(){}@'.indexOf(str[pos]) !== -1) {
      bufs[stage] += str[pos];
      pos++;
      continue;
    }

    if (stage == 0 && symbol == '@') {
      stage++;
      nS += bufs[0];
      bufs[0] = '';
      continue;
    } else if (stage >= 1) {
      if (symbol == '(')
        if (stage < 2) {
          stage = 2;
        } else {
          level++;
        }

      if (symbol == ')' && stage >= 2 && level > 0) level--;

      if (symbol == '{')
        if (stage != 3) {
          stage = 3;
        } else {
          level++;
        }

      if (symbol == '}') {
        if (level == 0) {
          bufs[3] += '}';

          nS += methods[bufs[1]](bufs[2].slice(1, -1), parseAndExecute(bufs[3].slice(1, -1)));

          bufs = ['', '', '', ''];
          stage = 0;
          continue;
        } else {
          level--;
        }
      }
    }
    bufs[stage] += symbol;
  }
  return nS + bufs[0];
}

module.exports.parseAndExecute = parseAndExecute;

function getNick(nick) {
  let hash = 0;
  for (var i = 0; i < nick.length; i++) hash += nick.charCodeAt(i) - 32;

  return chalk.hsv((hash + 160) % 360, 90, 90)(chalk.bold(nick));
}

module.exports.format = function(nick, message) {
  const nickSpace = '\r  ' + ' '.repeat(nick.length);
  nick = getNick(nick) + ': ';

  message = message.replace(/\\n/gm, '\n'); //  \n  
  message = parseAndExecute(message) // 

  //      
  message = message
    .split('\n')
    .map((e, i) => '' + (i !== 0 ? nickSpace : '') + e)
    .join('\n');

  return nick + message;
};

const { Server } = require('ssh2');
const { readFileSync } = require('fs');

const hostKey = readFileSync('./ssh'); //  
const users = JSON.parse(readFileSync('./users.json')); // 

let connectionCallback = () => {};

module.exports.createServer = function createServer({ lobby }) {
  //  
  const server = new Server(
    {
      banner: lobby, //     
      hostKeys: [hostKey]
    },
    function(client) {
      nick = '';
      client
        .on('authentication', ctx => {  // 
          if (ctx.method !== 'password') return ctx.reject();
          if (ctx.password !== users[ctx.username]) ctx.reject();
          nick = ctx.username;
          ctx.accept();
        })
        .on('ready', function() {
          connectionCallback(client, nick);
        });
    }
  );

  return server
};

module.exports.setConnectCallback = callback => { //    
  connectionCallback = callback;
};
