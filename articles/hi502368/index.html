    u8 value = gpio_get_value(receive_pin);

    reset_recv_timer();

    if ((timings_pos & 1) == value)
    {
        time_passed = now - last_time;
        if (timings_pos < TIMINGS_BUFFER_SIZE)
        {
            timings[timings_pos] = time_passed;
            timings_pos++;
        }
        last_time = now;

    return (irq_handler_t) IRQ_HANDLED;
}
