    {
        if (animalType == AnimalType.Cat)
            return new Cat();

            return new Dog();

        switch (animalType)

                return new Cat();
            case AnimalType.Dog:
                return new Dog();

                return new Parrot();
            default:
                throw new InvalidOperationException();
        }
    }
