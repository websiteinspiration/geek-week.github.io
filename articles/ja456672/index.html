Nginxレシピ：PostgreSQLからWebSocketへの非同期通知

PostgreSQLからwebsocketへの非同期通知を準備するには、nginx自体とそのプラグインpostgres、push-stream、set-miscが必要です。（元のリポジトリにまだ詰め込まれていないいくつかの変更を加えたため、私のフォークへのリンクを提供しました。既製のイメージを使用することもできます。）

クライアントをwebsocket経由でnginxに接続するには、次を作成します。 

location =/websocket {
    push_stream_subscriber websocket; #    websocket
    push_stream_channels_path $arg_id; #   websocket    id
    push_stream_websocket_allow_publish on; #   -   
    push_stream_client_subscribed_request /subscribe; #       
    push_stream_client_unsubscribed_request /unsubscribe; #       
    push_stream_client_publish_request /publish; #         websocket
}

クライアントをWebSocketに接続すると、PostgreSQLで非同期通知のリスニングを開始します 

location =/subscribe {
    internal;
    postgres_pass ngx; #    PostgreSQL
    set_quote_json_str $channel $arg_id; #   
    postgres_query "listen $channel"; #    
}

クライアントをWebSocketから切断すると、PostgreSQLでの非同期通知のリッスンが停止します 

location =/unsubscribe {
    internal;
    postgres_pass ngx; #    PostgreSQL
    set_quote_json_str $channel $arg_id; #   
    postgres_query "unlisten $channel"; #    
}

クライアントがWebSocketに投稿するとき、私たちは何かをします 

location =/publish {
    internal;
    postgres_pass ngx; #    PostgreSQL
    postgres_query "select now()"; #    PostgreSQL      websocket
}

また、単にwebsocketでクライアントに何かを送信できます 

location =/publisher {
    allow 127.0.0.1/16; #    
    deny all; #    
    push_stream_channel_info_on_publish off; #        
    push_stream_publisher; #  
    push_stream_channels_path $arg_id; #   websocket    id
}
