SwiftUIでのStoreオブジェクトを使用したアプリケーション状態のモデリング

今週は、SwiftUIでのデータレイヤーのモデリングについてお話します。SwiftUIだけを使用して作成した最初のアプリケーションの作業はすでに完了しています。これで、NapBotアプリケーションの開発時に使用したStoreオブジェクトを使用してモデルレイヤーを作成する方法を共有できます。

オブジェクトを保存


ストアオブジェクトは、状態を維持し、状態を変更するアクションを提供する責任があります。必要なだけ多くのStoreオブジェクトを持つことができ、できればシンプルで、アプリケーションの状態のごく一部を担当します。たとえば、ユーザー設定の状態を保存するSettingsStoreとカスタムタスクを保存するTodoStoreがあるとします。

Storeオブジェクトを作成するには、ObservableObjectプロトコルに準拠するクラスを作成する必要があります。ObservableObjectプロトコルを使用すると、SwiftUIでデータの変更を監視して応答できます。ObservableObjectの詳細については、「SwiftUIでのデータフローの管理」の記事を参照してください。SettingsStoreオブジェクトの簡単な例を見てみましょう。

import Foundation
import Combine

final class SettingsStore: ObservableObject {
    let objectWillChange = PassthroughSubject<Void, Never>()

    @UserDefault(Constants.UserDefaults.sleepGoal, defaultValue: 8.0)
    var sleepGoal: Double

    @UserDefault(Constants.UserDefaults.notifications, defaultValue: true)
    var isNotificationsEnabled: Bool

    private var didChangeCancellable: AnyCancellable?

    override init() {
        super.init()
        didChangeCancellable = NotificationCenter.default
            .publisher(for: UserDefaults.didChangeNotification)
            .map { _ in () }
            .receive(on: DispatchQueue.main)
            .subscribe(objectWillChange)
    }
}

上記のコード例には、ユーザー設定へのアクセスを提供するSettingsStoreクラスがあります。また、didChangeNotificationを使用して、ユーザーがデフォルト設定を変更するたびにSwiftUIに通知します。

長期使用


単純なTodoアプリを作成して、ストアオブジェクトの別の使用方法を見てみましょう。タスクのリストを格納し、それらを変更するためのアクションを提供するストアオブジェクトを作成する必要があります。たとえば、タスクの削除やフィルタリングなどです。

import Foundation
import Combine

struct Todo: Identifiable, Hashable {
    let id = UUID()
    var title: String
    var date: Date
    var isDone: Bool
    var priority: Int
}

final class TodosStore: ObservableObject {
    @Published var todos: [Todo] = []

    func orderByDate() {
        todos.sort { $0.date < $1.date }
    }

    func orderByPriority() {
        todos.sort { $0.priority > $1.priority }
    }

    func removeCompleted() {
        todos.removeAll { $0.isDone }
    }
}

ObservableObjectプロトコルに準拠 するTodosStoreクラスがあります。 TodosStoreは、状態を変更するためのいくつかのアクションを提供します。これらのメソッドをビューから使用できます。デフォルトでは、SwiftUIは@Publishedフィールドが変更されるたびにビューを更新します。これが、Todo要素の配列が@Publishedとして指定されている理由です。この配列に要素を追加または削除するとすぐに、SwiftUIはTodosStoreにサブスクライブしたビューを更新します。

これで、タスクのリストと、タスクに完了のマークを付ける、タスクの表示順序を削除および変更するなどのアクションを表示するビューを作成できます。まず、タスクのタイトルを表示するビューと、タスクに完了のマークを付けるラジオボタンを作成します。

import SwiftUI

struct TodoItemView: View {
    let todo: Binding<Todo>

    var body: some View {
        HStack {
            Toggle(isOn: todo.isDone) {
                Text(todo.title.wrappedValue)
                    .strikethrough(todo.isDone.wrappedValue)
            }
        }
    }
}

上記の例では、Bindingを使用して、値型へのアクセスなどの参照を提供しました。つまり、todo要素への書き込みアクセスを許可します。TodoItemViewはTodo構造のインスタンスを所有していませんが、Bindingを介してTodoStoreへの書き込みアクセス権を持っています。

import SwiftUI

struct TodosView: View {
    @EnvironmentObject var store: TodosStore
    @State private var draft: String = ""

    var body: some View {
        NavigationView {
            List {
                TextField("Type something...", text: $draft, onCommit: addTodo)
                ForEach(store.todos.indexed(), id: \.1.id) { index, _ in
                    TodoItemView(todo: self.$store.todos[index])
                }
                .onDelete(perform: delete)
                .onMove(perform: move)
            }
            .navigationBarItems(trailing: EditButton())
            .navigationBarTitle("Todos")
        }
    }

    private func delete(_ indexes: IndexSet) {
        store.todos.remove(atOffsets: indexes)
    }

    private func move(_ indexes: IndexSet, to offset: Int) {
        store.todos.move(fromOffsets: indexes, toOffset: offset)
    }

    private func addTodo() {
        let newTodo = Todo(title: draft, date: Date(), isDone: false, priority: 0)
        store.todos.insert(newTodo, at: 0)
        draft = ""
    }
}

これで、タスクを表示するためにListコンポーネントを使用する要素であるTodosViewができました。Listコンポーネントは、並べ替えと削除も提供します。もう1つ興味深いのは、indexed（）関数です。この関数は、インデックスを持つ要素のコレクションを返します。これを使用して、Bindingを通じてストア内のアイテムにアクセスします。この拡張の完全なソースは次のとおりです。

import Foundation

struct IndexedCollection<Base: RandomAccessCollection>: RandomAccessCollection {
    typealias Index = Base.Index
    typealias Element = (index: Index, element: Base.Element)

    let base: Base

    var startIndex: Index { base.startIndex }

    var endIndex: Index { base.endIndex }

    func index(after i: Index) -> Index {
        base.index(after: i)
    }

    func index(before i: Index) -> Index {
        base.index(before: i)
    }

    func index(_ i: Index, offsetBy distance: Int) -> Index {
        base.index(i, offsetBy: distance)
    }

    subscript(position: Index) -> Element {
        (index: position, element: base[position])
    }
}

extension RandomAccessCollection {
    func indexed() -> IndexedCollection<Self> {
        IndexedCollection(base: self)
    }
}

環境は、ストアオブジェクトを格納するための理想的な候補です。環境は、initメソッドを使用して明示的に実装しなくても、複数のビュー間でそれらを分割できます。SwiftUIの環境の利点の詳細については、「SwiftUIの環境機能」の記事をご覧ください。
