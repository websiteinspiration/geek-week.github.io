float select(x, y, moduleA, moduleB, moduleC, selectPoint, falloff)
{
    float s = moduleC(x,y);
    if(s >= selectPoint)
        return moduleB(x,y);
    else if(s <= selectPoint - falloff)
        return moduleA(x,y);
    else
    {
        float a = moduleA(x,y);
        float b = moduleB(x,y);
        return lerp(a, b, (1.0 / ((selectPoint - (selectPoint-falloff)) / (selectPoint - s)));
    }
}
