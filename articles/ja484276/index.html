class GfG
{
    public static int maxNodeLevel(Node node)
    {
        Queue<Node> q = new LinkedList<>();
        if (node == null)
            return 0;
        q.add(node);
        int max = 0;
        int level = 0;
        int x = 0;
        while (!q.isEmpty())
        {
            int size = q.size();
            if (max < size)
            {
                max = size;
                level = x;
            }
            for (int i = 0; i < size; i++)
            {
                Node curr = q.remove();
                if (curr.left != null)
                    q.add(curr.left);
                if (curr.right != null)
                    q.add(curr.right);
            }
            x++;
        }
        return level;
    }
}
