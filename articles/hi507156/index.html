    {

    }

    //    unmanaged,      T

    {
        fixed (T* p = buffer) return new V256((byte*)p);
    }

        return array;

        Avx2.BroadcastScalarToVector256(&b);
    //  16  short- s
    public static V256 FromShort(short s) =>
        Avx2.BroadcastScalarToVector256(&s).AsByte();
    //

        Avx2.BroadcastScalarToVector256(&ul).AsByte();
