GOアプリケーションでのPostgresql並列統合テスト


統合テストは、テストピラミッドのレベルの 1つです。通常、彼らはより多くの時間を必要とします、なぜなら それらでは、実際のコンポーネントのシミュレーションで何も置き換えません。このようなテストの時間を短縮するために、それらを並行して実行できます。ここでは、Postgresqlのそのようなテストについて具体的に説明します。

理想的には、各テストは独立している必要があり、お互いに影響を与えません。つまり、各テスト関数には独自の状態があります。これは、並列テストを使用する良い兆候です。各テスト関数の個人データセットを取得するために、テストを開始するときに一時的な回路を作成し、そこにデータを読み込み、テストの完了後に回路を破棄する関数を作成しました。作成された各回路 名前の競合を防ぐために、名前にハッシュが含まれています。


ヘルパー機能


テストのエラーを表示 するヘルパー関数から始めましょう。私は、Ben Johnsonのヘルパー関数（Ben Johnson）を使用しました。これにより、数行のコードを保存し、間違いをより明確かつ詳細に説明することができました。

テストデータ


データベース統合テストを実行するには、テストデータを提供する必要があります。Goテストツールは、ファイルからテストデータをロードするための優れたサポートを備えています。最初に、go buildは「testdata」というフォルダーをスキップします。次に、「go test」を実行すると、現在のフォルダーがパッケージフォルダーに変更されます。これにより、testdataフォルダーへの相対パスを使用して、テストデータセットをロードできます。

テスト用のデータベース接続の作成


package database

import (
    "math/rand" 
    "strconv" 
    "testing" 
    "time" 

    _ "github.com/lib/pq" 
    "database/sql" 
)

const (
    dbPort     = 5439
    dbUser     = "postgres" 
    dbPassword = "postgres" 
    dbName     = "test" 
)

func CreateTestDatabase(t *testing.T) (*sql.DB, string, func()) {
    connectionString := fmt.Sprintf("port=%d user=%s password=%s dbname=%s sslmode=disable", dbPort, dbUser, dbPassword, dbName)
    db, dbErr := sql.Open("postgres", connectionString)
    if dbErr != nil {
        t.Fatalf("Fail to create database. %s", dbErr.Error())
    }

    rand.Seed(time.Now().UnixNano())
    schemaName := "test" + strconv.FormatInt(rand.Int63(), 10)

    _, err := db.Exec("CREATE SCHEMA " + schemaName)
    if err != nil {
        t.Fatalf("Fail to create schema. %s", err.Error())
    }

    return db, schemaName, func() {
        _, err := db.Exec("DROP SCHEMA " + schemaName + " CASCADE")
        if err != nil {
            t.Fatalf("Fail to drop database. %s", err.Error())
        }
    }
}


「CreateTestDatabase」を呼び出して、テストデータベースへの接続を作成し、テスト用の新しいデータスキーマを作成します。この関数は、データベース接続、作成されたスキーマの名前、およびこのスキーマを削除するパージ関数を返します。テストの場合、呼び出し元にエラーを返すよりもテストに失敗した方がよいです。（注：クリーンアップ機能の復帰は、Mitchell HashimotoのAdvanced Testing with Goトークに基づいています）。

テストデータセットのダウンロード


「.sql」ファイルを使用しました。1つのsqlには、1つのテーブルのデータが含まれています。これには、テーブルの作成とデータの入力が含まれます。すべてのSQLファイルは「testdata」フォルダーに保存されます。以下はSQLファイルの例です。

CREATE TABLE book (
    title character varying(50),
    author character varying(50)
);

INSERT INTO book VALUES
('First Book','First Author'),
('Second Book','Second Author')
;

そして、ここが複雑な部分です。各関数は独自のデータスキームで実行されるため、これらのSQLファイルでクエリを実行（書き込み）することはできません。テーブルを作成したり、目的の一時スキーマにデータを挿入したりするには、テーブル名の前にスキーマを指定する必要があります。たとえば、CREATE TABLE book ...はCREATE TABLE uniqueschema.book ...として記述し、INSERT INTO book ...はINSERT INTO uniqueschema.book ...に変更する必要があります。実行前に正規表現を使用してクエリを変更しました。テストデータのダウンロードコードは次のとおりです。

package datalayer

import (
    "bufio" 
    "fmt" 
    "io" 
    "os" 
    "regexp" 
    "testing" 

    "database/sql" 
    "github.com/Hendra-Huang/databaseintegrationtest/testingutil" //     (   ,  79)
)

//        
var schemaPrefixRegexps = [...]*regexp.Regexp{
    regexp.MustCompile(`(?i)(^CREATE SEQUENCE\s)(["\w]+)(.*)`),
    regexp.MustCompile(`(?i)(^CREATE TABLE\s)(["\w]+)(\s.+)`),
    regexp.MustCompile(`(?i)(^ALTER TABLE\s)(["\w]+)(\s.+)`),
    regexp.MustCompile(`(?i)(^UPDATE\s)(["\w]+)(\s.+)`),
    regexp.MustCompile(`(?i)(^INSERT INTO\s)(["\w]+)(\s.+)`),
    regexp.MustCompile(`(?i)(^DELETE FROM\s)(["\w]+)(.*)`),
    regexp.MustCompile(`(?i)(.+\sFROM\s)(["\w]+)(.*)`),
    regexp.MustCompile(`(?i)(\sJOIN\s)(["\w]+)(.*)`),
}

//     
func addSchemaPrefix(schemaName, query string) string {
    prefixedQuery := query
    for _, re := range schemaPrefixRegexps {
        prefixedQuery = re.ReplaceAllString(prefixedQuery, fmt.Sprintf("${1}%s.${2}${3}", schemaName))
    }
    return prefixedQuery
}

func loadTestData(t *testing.T, db *sql.DB, schemaName string, testDataNames ...string) {
    for _, testDataName := range testDataNames {
        file, err := os.Open(fmt.Sprintf("./testdata/%s.sql", testDataName))
        testingutil.Ok(t, err)
        reader := bufio.NewReader(file)
        var query string
        for {
            line, err := reader.ReadString('\n')
            if err == io.EOF {
                break
            }
            testingutil.Ok(t, err)
            line = line[:len(line)-1]
            if line == "" {
                query = addSchemaPrefix(schemaName, query)
                _, err := db.Exec(query)
                testingutil.Ok(t, err)
                query = "" 
            }
            query += line
        }
        file.Close()
    }
}


テスト作成


各テストを開始する前に、スキームの一意の名前を持つテストデータベースが作成され、このスキームを削除するクリーンアップ関数の実行が延期されます。テストでは、スキーマ名がリクエストに挿入されます。この実装で最も重要なことは、接続を実際のデータベースからテストデータベースとの接続に変更するには、データベース接続をカスタマイズできる必要があるということです。各テスト関数の先頭に「t.Parallel（）」を追加して、このテストを並行して実行する必要があることをテスト環境に示します。
以下は完全なコードです：

//            "integration" (. build flags) 
// +build integration

package datalayer

import (
    "context" 
    "testing" 

    "github.com/Hendra-Huang/databaseintegrationtest/database" 
    "github.com/Hendra-Huang/databaseintegrationtest/testingutil" 
)

func TestInsertBook(t *testing.T) {
    t.Parallel()
    db, schemaName, cleanup := database.CreateTestDatabase(t)
    defer cleanup()

    loadTestData(t, db, schemaName, "book") // will load data which the filename is book

    title := "New title" 
    author := "New author" 
    // those 2 lines code below are not a good practice
    // but it is intentional to keep the focus only on integration test part
    // the important part is database connection has to be configurable
    insertBookQuery = addSchemaPrefix(schemaName, insertBookQuery) // override the query and add schema to the query
    err := InsertBook(context.Background(), db, title, author) // will execute insertBookQuery with the provided connection

    testingutil.Ok(t, err)
}

func TestGetBooks(t *testing.T) {
    t.Parallel()
    db, schemaName, cleanup := database.CreateTestDatabase(t)
    defer cleanup()

    loadTestData(t, db, schemaName, "book")

    getBooksQuery = addSchemaPrefix(schemaName, getBooksQuery)
    books, err := GetBooks(context.Background(), db)

    testingutil.Ok(t, err)
    testingutil.Equals(t, 2, len(books))
}


注：「TestGetBooks」では、クエリは次のように2冊の本を返すと想定しています。上記の挿入テストがありますが、「testdata / book.sql」に非常に多くのテストデータセットを取り込みました。2つのテスト間で回路を共有しない場合、「TestGetBooks」は失敗します。表の3行、テストから2行、上記のテスト挿入から1行。これは、テスト用の個別のスキームの利点です。それらのデータは独立しているため、テストは互いに独立しています。

結論


私のプロジェクトでは、このアプローチにより、順次テストと比較してテスト時間が40〜50％短縮されます。テストを並行して実行するもう1つの利点は、アプリケーションが複数の競合アクションを処理するときに発生する可能性があるいくつかのエラーを回避できることです。
