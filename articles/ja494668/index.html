強力なTypescriptデコレータ-どのように機能し、何にコンパイルされ、どのアプリケーションタスクに適用できるか

各Angular開発者は、タイムコードコードでデコレーターを確認しました。これらは、モジュールの説明、依存性注入の構成、またはコンポーネントの構成に使用されます。言い換えると、デコレータは、フレームワークまたはコンパイラ（Angularの場合）の追加情報またはメタデータを記述するために使用されます。さらに、Angularはほんの一例です。宣言的アプローチとして、コードの簡素化と視覚化のためにデコレータを使用する他の多くのライブラリがあります。過去の.NET開発者として、TSデコレーターと.NET属性の間には多くの類似点があると思います。最後に、成長するNestJSバックエンドアプリケーションフレームワーク（Nodeを抽象化したもの）も、デコレータの多用と宣言型アプローチに基づいて構築されています。すべてがどのように機能し、コードでデコレータを使用するか、それをより便利で読みやすくするには？ TSコードをコンパイルすると、Javascriptコードが得られることは誰もが理解しています。他の多くのTypescript機能のように、デコレータの概念はありません。したがって、私にとって最も興味深い質問は、デコレータがコンパイル後に何になるかです。この問題を追求するために、ミンスクでの会議でスピーチを行い、記事を共有したいと思います。

import 'reflect-metadata';

const instancesMap: Map<Object, Object> = new Map<Object, Object>();

function getInstance<T>(tType: new () => T): T {
    let metadata = Reflect.getMetadata('metadataKey', tType) as Metadata;
    if (metadata.singleton) {
        if (!instancesMap.has(tType)) {
            instancesMap.set(tType, new tType());
        }
        return instancesMap.get(tType) as T;
    } else {
        return new tType() as T;
    }
}

function validate<T>(object: T) {
    const properties = Object.getOwnPropertyNames(object);
    properties.forEach(propertyName => {
        let metadata = Reflect.getMetadata(metaKey + propertyName, object.constructor);
        if (metadata && metadata.range) {
            const value = object[metadata.propertyName];
            if (value < metadata.range.from || value > metadata.range.to) {
                throw new Error('Validation failed');
            }
        }
    });
}
