PASO 1


En el archivo Startup.cs en el método ConfigureServices , agregue el siguiente código:

  using Microsoft.AspNetCore.Authentication.JwtBearer;

  public void ConfigureServices(IServiceCollection services)
  {
    // .....
    services.AddAuthentication(sharedOptions =>
    {
        sharedOptions.DefaultScheme = JwtBearerDefaults.AuthenticationScheme;
    })
    .AddJwtBearer(options =>
    {
        // ClientID   Azure AD B2C
        options.Audience = "777aaa77a-7a77-7777-bb77-8888888aabc";
        //  issuer    "".    
        // " ".     ,
        //    (B2C_1A_signup_signin).
        options.Authority = "https://antekesd.b2clogin.com/antekesd.onmicrosoft.com/B2C_1A_signup_signin";
        options.TokenValidationParameters = new Microsoft.IdentityModel.Tokens.TokenValidationParameters
        {
            // ,   .       
            // Azure AD B2C     -    teanat-id
            //  " "
            ValidateIssuer = true,
            ValidIssuer = "https://antekesd.b2clogin.com/teanat-id/v2.0/"
        };
    });
    //  ,      
    services.AddAuthorization(options =>
    {
        options.AddPolicy("AuthOnly", policy => policy.RequireAuthenticatedUser());
    });
    // .....
  }

PASO 2


En el mismo archivo, en el método Configure , debe agregar UseAuthentication y UseAuthorization . Deben estar entre UseRouting y UseEndpoints , como se muestra en el siguiente ejemplo:

  public void Configure(IApplicationBuilder app, IWebHostEnvironment env)
  {
    // .....
    app.UseRouting();
    app.UseAuthentication();
    app.UseAuthorization();
    app.UseEndpoints(endpoints =>
    {
        endpoints.MapControllerRoute(
            name: "default",
            pattern: "{controller}/{action=Index}/{id?}");
    });
    // .....
  }

PASO 3


Ahora todo está listo, solo queda agregar un cheque al controlador. Para hacer esto, agregue cualquier controlador,
[Authorize(Policy = "AuthOnly")]

como se muestra en el siguiente ejemplo:

    [Authorize(Policy = "AuthOnly")]
    [Route("api/[controller]")]
    [ApiController]
    public class ClientController : ControllerBase
    {
        private readonly ClientService _clientService;
        public ClientController(ClientService clientService)
        {
            _clientService = clientService;
        }

        [HttpGet()]
        public async Task<IActionResult> GetClientList()
        {
            try
            {
                var result = await _clientService.GetClientsList();
                return Ok(result);
            }
            catch (HttpRequestException httpException)
            {
                throw httpException;
            }
        }
    }

Enviar una solicitud de un servidor a otro


A veces surge una situación cuando una aplicación usa más de un servidor. Entonces es necesario enviar una solicitud de forma consecutiva. Para hacer esto, necesitamos crear un cliente HTTP y agregarle un token. Aquí te explicamos cómo hacerlo.

PASO 1


Creamos un servicio que enviará solicitudes.

  using System.Net.Http;
  using System.Net.Http.Headers;
  using Microsoft.AspNetCore.Http;

  public class ClientService
    {
        //  HttpClient 
        HttpClient Client { get;  }
        public ClientService(HttpClient client, IHttpContextAccessor httpContextAccessor)
        {
            //    
            client.BaseAddress = new Uri("https://another-server.azurewebsites.net/api/");
            string _ContentType = "application/json";
            //     JSON
            client.DefaultRequestHeaders.Accept.Add(new MediaTypeWithQualityHeaderValue(_ContentType));
            //  JWT 
            client.DefaultRequestHeaders.Add("Authorization", (string)httpContextAccessor.HttpContext.Request.Headers["Authorization"]);
            Client = client;
        }

        public async Task<string> GetClientsList()
        {
            //  
            var response = await Client.GetAsync("url-to-another-endpoint");
            if (response.StatusCode != HttpStatusCode.OK)
            {
                return response.StatusCode.ToString();
            }
            string result = await response.Content.ReadAsStringAsync();
            return result;
        }
    }

PASO 2


Agregue el cliente Http al servicio creado. Para hacer esto, agregue el siguiente código en el archivo Startup.cs :

  public void ConfigureServices(IServiceCollection services)
  {
    // ...
    services.AddHttpClient<ClientService>();
    // ...
  }
