class Question(models.Model):
    title = models.CharField(max_length=4096)
    visible = models.BooleanField(default=False)
    max_points = models.FloatField()

    def __str__(self):
           return self.title

class Choice(models.Model):
    question = models.ForeignKey(Question, on_delete=models.DO_NOTHING)
    title = models.CharField(max_length=4096)
    points = models.FloatField()
    lock_other = models.BooleanField(default=False)

    def __str__(self):
        return self.title

class Answer(models.Model):
    user = models.ForeignKey(settings.AUTH_USER_MODEL, on_delete=models.DO_NOTHING)
    question = models.ForeignKey(Question, on_delete=models.DO_NOTHING)
    choice = models.ForeignKey(Choice, on_delete=models.DO_NOTHING)
    created = models.DateTimeField(auto_now_add=True)

    def __str__(self):
        return self.choice.title

from rest_framework import serializers
from .models import Answer, Question, Choice


class ChoiceSerializer(serializers.ModelSerializer):
    percent = serializers.SerializerMethodField()

    class Meta:
        model = Choice
        fields = ['pk', 'title', 'points', 'percent', 'lock_other', ]

    def get_percent(self, obj):
        total = Answer.objects.filter(question=obj.question).count()
        current = Answer.objects.filter(question=obj.question, choice=obj).count()
        if total != 0:
            return float(current * 100 / total)
        else:
            return float(0)


class QuestionSerializer(serializers.ModelSerializer):
    choices = ChoiceSerializer(many=True, source='choice_set', )

    class Meta:
        model = Question
        fields = ['pk', 'title', 'choices', 'max_points', ]


class AnswerSerializer(serializers.Serializer):
    answers = serializers.JSONField()

    def validate_answers(self, answers):
        if not answers:
            raise serializers.Validationerror("Answers must be not null.")
        return answers

    def save(self):
        answers = self.data['answers']
        user = self.context.user
        for question_id, in answers:  #     ,    
            question = Question.objects.get(pk=question_id)
            choices = answers[question_id]
            for choice_id in choices:
                choice = Choice.objects.get(pk=choice_id)
                Answer(user=user, question=question, choice=choice).save()
                user.is_answer = True
                user.save()


次に、views.pyに2つのDRFビューを記述します。これにより、すべての質問にオプションが付けられ、ユーザーからのすべての回答が受け入れられます。


ここにテキストをコード化 
from .serializers import QuestionSerializer, AnswerSerializer
from rest_framework.permissions import IsAuthenticated
from rest_framework.generics import GenericAPIView
from rest_framework.response import Response
from .models import Question


class GetQuestion(GenericAPIView):
    permission_classes = (IsAuthenticated,)
    serializer_class = QuestionSerializer

    def get(self, request, format=None):
        questions = Question.objects.filter(visible=True, )
        last_point = QuestionSerializer(questions, many=True)
        return Response(last_point.data)


class QuestionAnswer(GenericAPIView):
    permission_classes = (IsAuthenticated,)
    serializer_class = AnswerSerializer

    def post(self, request, format=None):
        answer = AnswerSerializer(data=request.data, context=request)
        if answer.is_valid(raise_exception=True):
            answer.save()
            return Response({'result': 'OK'})

from django.contrib import admin
from .models import Question, Answer, Choice


class QuestionAdmin(admin.ModelAdmin):
    list_display = (
        'title',

        'max_points',
    )


class ChoiceAdmin(admin.ModelAdmin):
    list_display = (
        'title',
        'question',
        'points',
        'lock_other',
    )
    list_filter = ('question',)


class AnswerAdmin(admin.ModelAdmin):
    list_display = (
        'user',
        'question',
        'choice',
    )
    list_filter = ('user',)


admin.site.register(Question, QuestionAdmin)
admin.site.register(Choice, ChoiceAdmin)
admin.site.register(Answer, AnswerAdmin)

その中で、質問と回答のリストを取り除き、各要素を最後まで調べます。質問のタイプに応じて、独自のロジックとスタイルでコンポーネントを変更します。したがって、リストに質問が残っていない場合は、結果を背面に送信し、ポイント数を回答します。ポイント数を受け取った後、結果ページを開きます。ポイントがある場合、質問は表示されなくなります。

次に、ソーシャルネットワークで調査結果の共有を実装するとよいでしょう。 OK。 VKontakteおよびFacebookポイントと画像を共有できるソーイング機能。

VKとFacebookのポイントを個別に反映した100種類の画像のバリエーションを生成します（異なる解像度）。次に、リンクの転送をフロントエンド部分のソーシャルコンポーネントの画像に接続します。 VKontakteを使用すると、すべてが単純であることがわかりました。目的の画像のURLを直接指定して画像パラメーターを渡します。しかし、私はFacebookをいじくり回さなければなりませんでした。 APIでメディアを受け入れないことがわかりました。画像のURLから画像または画像を転送すると、大きな空のフィールドが投稿に表示されました。後で判明したように、彼はサイト自体のmetainf（og：image）から写真を撮り、それを共有しました（リンクでパラメーターuを渡します）。そして、彼女は、とりわけ、動的に変更されなければなりませんでした。不要なリダイレクトやメカニックを背面で実行したくなかったため、SPA（シングルページアプリ）を前面でSSR（サーバー側レンダリング）に変換することにしました。そのため、リクエストに応じて、ブラウザでJavaScriptが起動される前に、head-metaにスコアがある画像のURLが変更されます。さいわい、Nuxt.jsフレームワークをベースとして、モードを切り替えるだけでこれを実現できます。クライアントのみのタグをスケッチし、クエリスコアの存在からヘッドを変更するロジックを追加します。



さらに、サーバーでは、デーモンサービスを開始して生成されたページを提供し、静的情報をnginxに残す必要がありました。すべての利益！
