関数を受け取ってそれを1回だけ実行する演算子を作成する必要があるとします。TapOnceという名前で実装します


function tapOnce<T>(fn: Function): OperatorFunction<T, T> {
  return function(source: Observable<any>) {
    let run = false;
    return source.pipe(
      tap(() => {
        if (!run) {
          fn();
          run = true;
        }
      })
    );
  };
}

コードは明確です-これはtap関数を実行するために使用され、フラグはrunこれを一度だけ実行するために必要です。次に、演算子を使用します。

import { defer } from 'rxjs';

function tapOnce<T>(fn: Function): OperatorFunction<T, T> {
  return function(source: Observable<any>) {
    return defer(() => {
      let run = false;
      return source.pipe(
        tap(() => {
          if (!run) {
            fn();
            run = true;
          }
        })
      );
    });
  };
}

// This already executing regardless the numbers of handlers

  return new Promise(resolve => {
    resolve();
  });
});

promiseDefered.subscribe(console.log);
