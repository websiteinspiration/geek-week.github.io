この時点で、パフォーマンスの問題が始まりました。事実は、特定のレベルまで下がって、そこでポジションを評価することはできないということです。たとえば、クイーンをポーンで連れて行くことで検索を終了する場合があります。次の動きでクイーンを連れて行くことができるということに注意せずに、私たちにとって最良の結果は得られません。何かを評価する前に、すべての強制的な移動をすべて「失う」必要があります。これには、すべての捕獲、および王がチェック下にある状況でのすべての移動が含まれます。

それが問題でした。実際のところ、割り当てられた時間のすべてのチェックは、「アルファ-ベータクリッピング」に関するコードのその部分で実行され、再生ではありません。そのようなアフターゲームが巨大に分岐した形を取ることができるゲームがあります。そこで、プログラムの実行は死にかけていました。同時に、「プレイアウト」に時間チェックを単純に追加することは不可能でした。その結果、すべてが、おそらく本当に良い動きに到達する前にそこで終了してしまうためです。私はどういうわけかこの問題を制限しなければなりませんでした（もちろん、ゲームの品質が低下しました）。

var pawnShift = function(ctx, params) {
    if (ctx.go(params, 0) && ctx.isEmpty()) {
        if (ctx.inZone(0)) {
            ctx.promote(4);
        }    
        ctx.end();
    }
}

var pawnLeap = function(ctx, params) {
    if (ctx.go(params, 0) && ctx.isEnemy()) {
        if (ctx.inZone(0)) {
            ctx.promote(4);
        }    
        ctx.end();
    }
}

var pawnJump = function(ctx, params) {
    if (ctx.go(params, 0) && 
        ctx.isEmpty() && 
        ctx.inZone(1) && 
        ctx.go(params, 0) && 
        ctx.isEmpty()) {
        ctx.end();
    }
}

var enPassant = function(ctx, params) {
    if (ctx.go(params, 0) &&
        ctx.isEnemy() &&
        ctx.isPiece(0)) {
        ctx.capture();
        if (ctx.go(params, 1)) {
            ctx.put();
            if (ctx.go(params, 1) &&
                ctx.isLastFrom()) {
                ctx.end();
            }
        }
    }
}

var jump = function(ctx, params) {
    if (ctx.go(params, 0) && 
        ctx.go(params, 1) && 
       !ctx.isFriend()) {
        ctx.end();
    }
}

var slide = function(ctx, params) {
    while (ctx.go(params, 0)) {
        if (ctx.isFriend()) break;
        ctx.end();
        if (!ctx.isEmpty()) break;
    }
}
