class mainConversation extends conversation
{

   public $response = [];

   public function run () {
       $this->setName();
   }

   private function setName() {
       $question = BotManQuestion::create("!   ?");

       $this->ask( $question, function ( BotManAnswer $answer ) {
           if( $answer->getText () != '' ){
               array_push ($this->response, $answer->getText());

               $this->askWeather ();
          }
      });
   }
}
