    database.setPort(3306);

        "SELECT `horse_color` FROM `horses` WHERE `horse_hash`=:horse_hash");

    query.bindValue(":horse_hash", horseHash);

        horseColor = query.value("horse_color").toString();

        horseColor = "unknown";
