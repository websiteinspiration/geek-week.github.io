        const rewrite = id in this.items;

            mutation.indices = [...this.indices, id];

        return mutation;
    }

        let item = await api.post(this.url, data);
        this.add(item);
    }

        let item = await api.get(`${this.url}/${id}`);
        this.add(item);
    }
