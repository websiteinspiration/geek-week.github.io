Googleアシスタントでスマートホームセンサーを管理する

こんにちは、このガイドの同僚は、例としてESP8266ボードとLEDを使用して、Googleアシスタントとmqttプロトコルを使用してスマートホームセンサーを制御する方法を説明します。また、ブラックジャックとphpスクリプトを使用して独自のアシスタントアプリを作成します。みんなに猫をお願いします。

これを行うには、ESP8266コントローラー、またはインターネット接続を備えた他のコントローラー、および有効なssl証明書を備えたサーバー（有効なsslの代わりに、既に持っているリバースプロキシを使用できます）とMQTTサーバー（ブローカー）が必要です。

Googleアシスタントとの統合には、GoogleアクションサービスとDialogflowを使用します。それでは始めましょう。

PHPスクリプトの設定、トピックへのサブスクリプション


実際にはスクリプト自体です。その目的は、Dialogflowからの投稿リクエストを受け入れ、アクションをアクションから分離し、スクリプトの最後に名前が示されている件名のブローカーにメッセージとして送信することです。スクリプトを何と呼んでもかまいません。ちなみに、これは「フルフィルメント」タブで示したスクリプトです。スクリプトを/ var / www / htmlに配置

します。トピックはサブスクライバーによって作成されます。トピックを作成するには、次のコマンドを使用します。

mosquitto_sub -h localhost - t /test/light

localhostの代わりに、パブリッシャー（pub）を作成する他のアドレスまたはドメインを指定できます。

/ test / lightの代わりに、任意のトピックを指定できます。私たちの場合、主なことは、それがスクリプトに示されていることです。

メッセージは発行者によって作成されます。メッセージを作成するには、コマンドを使用できます。

mosquitto_pub -h localhost - t /test/light -m “light”

しかし、パブリッシャー（pub）がアプリケーションになるので、それは必要ありません。スキームはこれで、アプリケーションがコマンドを受信すると、スクリプトにリクエストを送信します。スクリプトはブローカー用で、ブローカーはサブスクライバー用（esp8266）です。

ファームウェアESP8266


ファームウェアには、Arduino IDEを使用します。誰かが初めてIDEをインストールする場合は、ch340ドライバーを忘れないでください。デフォルトでは、arduinoではそのような支払いはありません。ファイル>>設定で、追加のボードのアドレスを指定する必要があります：arduino.esp8266.com/stable/package_esp8266com_index.json。ツール>>ボード>>ボードマネージャーで、esp8266パッケージをインストールする必要があります。

const char * ssidの後のスケッチでは、Wi-Fiネットワークの名前を示す必要があります。const char * passwordの後、彼女のパスワード。const char * mqtt_serverの後に、サーバーのIPアドレスを指定します。

Arduino IDEのスケッチ
#include <ESP8266WiFi.h>
#include <PubSubClient.h>

// Update these with values suitable for your network.

const char* ssid = "***";
const char* password = "********";
const char* mqtt_server = "**.**.*.*";

WiFiClient espClient;
PubSubClient client(espClient);
long lastMsg = 0;
char msg[50];
int value = 0;

int led = D5;

void setup_wifi() {

  delay(10);
  // We start by connecting to a WiFi network
  Serial.println();
  Serial.print("Connecting to ");
  Serial.println(ssid);

  WiFi.begin(ssid, password);

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  randomSeed(micros());

  Serial.println("");
  Serial.println("WiFi connected");
  Serial.println("IP address: ");
  Serial.println(WiFi.localIP());
}

void callback(char* topic, byte* payload, unsigned int length) {
  String msg="";
  Serial.print("Message arrived [");
  Serial.print(topic);
  Serial.print("] ");
  for (int i = 0; i < length; i++) {
    Serial.print((char)payload[i]);
    msg+=(char)payload[i];
  }
  Serial.println();

  // Switch on the LED if an 1 was received as first character
//  if ((char)payload[0] == '1') {
  if (msg == "light") {
    digitalWrite(led, HIGH);   // Turn the LED on 
  } else {
    digitalWrite(led, LOW);  // Turn the LED off 
  }

}

void reconnect() {
  // Loop until we're reconnected
  while (!client.connected()) {
    Serial.print("Attempting MQTT connection...");
    // Create a random client ID
    String clientId = "ESP8266Client-";
    clientId += String(random(0xffff), HEX);
    // Attempt to connect
    if (client.connect(clientId.c_str())) {
      Serial.println("connected");
      // ... and resubscribe
      client.subscribe("/test/light");
    } else {
      Serial.print("failed, rc=");
      Serial.print(client.state());
      Serial.println(" try again in 5 seconds");
      // Wait 5 seconds before retrying
      delay(5000);
    }
  }
}

void setup() {
  pinMode(led, OUTPUT);     // Initialize the led pin as an output
  Serial.begin(115200);
  setup_wifi();
  client.setServer(mqtt_server, 1883);
  client.setCallback(callback);
}

void loop() {

  if (!client.connected()) {
    reconnect();
  }
  client.loop();

}
