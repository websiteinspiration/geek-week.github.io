Nano-Neuron-マシンが「学習」する方法を示す7つの単純なJavaScript関数

ナノニューロンは、ニューラルネットワークの概念からのニューロンの簡略化バージョンです。ナノニューロンは最も単純なタスクを実行し、温度を摂氏から華氏に変換するようにトレーニングされています。


NanoNeuron.jsコードは、トレーニング、トレーニング、予測、モデルからの信号の順方向および逆方向の伝播に影響を与える7つの単純なJavaScript関数で構成されています。これらの関数を作成する目的は、結局のところ、マシンが「学習」する方法について最小限の基本的な説明（直感）を読者に提供することでした。コードはサードパーティのライブラリを使用しません。ことわざにあるように、単純な「バニラ」JavaScript関数だけです。


これらの関数は、機械学習の完全なガイドではありません。多くの機械学習の概念が省略または簡略化されています！この単純化は唯一の目的で許可されます。つまり、機械が原理的に「学習」する方法について最も基本的な理解と直感を読者に提供することです。その結果、「機械学習のマジック」は「機械学習の数学」として読者にますます聞こえるようになります。

function NanoNeuron(w, b) {
  this.w = w;
  this.b = b;
  this.predict = (x) => {
    return x * this.w + this.b;
  }

function backwardPropagation(predictions, xTrain, yTrain) {
  const m = xTrain.length;
  //           'w'  'b'.
  //      0.
  let dW = 0;
  let dB = 0;
  for (let i = 0; i < m; i += 1) {
    dW += (yTrain[i] - predictions[i]) * xTrain[i];
    dB += yTrain[i] - predictions[i];
  }
  //    .
  dW /= m;
  dB /= m;
  return [dW, dB];
}

function trainModel({model, epochs, alpha, xTrain, yTrain}) {
  //     -.  .
  const costHistory = [];

  //    () 
  for (let epoch = 0; epoch < epochs; epoch += 1) {
    //   .
    const [predictions, cost] = forwardPropagation(model, xTrain, yTrain);
    costHistory.push(cost);

    //   .
    const [dW, dB] = backwardPropagation(predictions, xTrain, yTrain);

    //    -,    .
    nanoNeuron.w += alpha * dW;
    nanoNeuron.b += alpha * dB;
  }

  return costHistory;
}
