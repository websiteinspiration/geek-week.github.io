func subtractionGCD(_ firstNumber: Int, _ secondNumber: Int) -> Int {
    if let simpleGCD = simpleCasesGCD(firstNumber, secondNumber) {
        return simpleGCD
    }

    var firstNumber = firstNumber
    var secondNumber = secondNumber
    while firstNumber != 0, secondNumber != 0 {
        if firstNumber > secondNumber {
            firstNumber = firstNumber - secondNumber
        } else {
            secondNumber = secondNumber - firstNumber
        }
    }

    return firstNumber + secondNumber // One of them is 0.
}

ここでは、私のために、再利用のために、GCDを検索する別の関数のケースを作成しました。これは、アルゴリズムがわからなくても、すぐにわかったときに実行できます。

func simpleCasesGCD(_ firstNumber: Int, _ secondNumber: Int) -> Int? {
    if firstNumber == secondNumber {
        return firstNumber // Any.
    }

    if firstNumber == 0 {
        return secondNumber
    }
    if secondNumber == 0 {
        return firstNumber
    }

    return nil
}

ユークリッドアルゴリズムの「改良」バージョン


文献では、秒ごとの除算の残りの代わりに、各ステップの数値の1つがこのオフセットと2番目の数の差で置き換えられるアルゴリズムのバージョンを見つけることができますが、除算の残りが2番目の数の半分を超えている場合のみです。このバージョンの実装は次のようになります。

func improvedDivisionGCD(_ firstNumber: Int, _ secondNumber: Int) -> Int {
    if let simpleGCD = simpleCasesGCD(firstNumber, secondNumber) {
        return simpleGCD
    }

    var firstNumber = firstNumber
    var secondNumber = secondNumber
    while firstNumber != 0, secondNumber != 0 {
        if firstNumber > secondNumber {
            let firstNumberClaim = firstNumber % secondNumber
            if firstNumberClaim > secondNumber / 2 {
                firstNumber = abs(firstNumberClaim - secondNumber)
            } else {
                firstNumber = firstNumberClaim
            }
        } else {
            let secondNumberClaim = secondNumber % firstNumber
            if secondNumberClaim > firstNumber / 2 {
                secondNumber = abs(secondNumberClaim - firstNumber)
            } else {
                secondNumber = secondNumberClaim
            }
        }
    }

    return firstNumber + secondNumber // One of them is 0.
}

このような変更により、アルゴリズムのステップ数が減少しますが、コンピューターでの測定結果から判断すると、各ステップで追加の計算とチェックを行うと、この利点がさらに損なわれます

NODを見つけるためのバイナリー方法


GCDの検索について言えば、前世紀の60年代に、私がWebでまったく情報を見つけられなかった特定のジョセフスタインによってすでに提案されたアルゴリズムに言及する価値があります。それ（アルゴリズム）はバイナリ演算を指向しており、除算演算は含まれていません。アルゴリズムは、パリティチェックと半分にのみ機能します。これは、バイナリ演算のみの機能で実現可能です。

アルゴリズムは4つの事実に基づいています。

  1. uとvが両方とも偶数の場合、gcd（u、v）= 2 * gcd（u / 2、v / 2）;
  2. uが偶数でvが偶数でない場合、gcd（u、v）= gcd（u / 2、v）;
  3. gcd（u、v）= gcd（u-v、v）（これはユークリッドアルゴリズムによる）。
  4. uとvが両方とも奇数の場合、u-vは偶数で、| u-v | <max（u、v）

ウィキペディアでは、アルゴリズムの再帰バージョンを確認できます（最新のプログラミング言語では数行で記述されています）。Swiftでは書き直していません。そしてここで、反復的な実装オプションを示します。

func binaryGCD(_ firstNumber: Int, _ secondNumber: Int) -> Int {
    if let simpleGCD = simpleCasesGCD(firstNumber, secondNumber) {
        return simpleGCD
    }

    var firstNumber = firstNumber
    var secondNumber = secondNumber

    var shift = 0
    while (firstNumber | secondNumber) & 1 == 0 {
        shift += 1
        firstNumber >>= 1
        secondNumber >>= 1
    }
    while firstNumber & 1 == 0 {
        firstNumber >>= 1
    }
    repeat {
        while secondNumber & 1 == 0 {
            secondNumber >>= 1
        }
        if firstNumber > secondNumber {
            swap(&firstNumber, &secondNumber)
        }
        secondNumber -= firstNumber
    } while secondNumber != 0

    return firstNumber << shift
}
