「比較：Svelte and React」という記事で、私はゲームTic Tac Toeの開発を繰り返してみました。そこでは、移動の履歴をサポートせずに、Reactの元のチュートリアルの最初の部分だけを完了しました。この記事では、動きの履歴をサポートするSvelteフレームワークを使用して、このゲームの開発を開始します。移動の履歴は、実際には元に戻す/やり直しシステムです。元のReactチュートリアルでは、任意の状態にランダムにアクセスできる状態ストレージを備えた元に戻す/やり直しシステムが実装されています。元に戻す/やり直しシステムを実装する場合、通常はコマンドパターンが使用され、元に戻す/やり直しコマンドはコマンドのリストに格納されます。後でこのアプローチを実装しようとしますが、今度は、状態ストレージを使用して元に戻す/やり直しシステムを実行します。


開発では、ストレージを使用したFluxアーキテクチャソリューションを使用します。これは、記事の別のセクションです。

Board.svelte
<script>
    import { onMount } from 'svelte';

    export let width  = 3;
    export let height = 3;
    export let cellWidth  = 34;
    export let cellHeight = 34;
    export let colorStroke = "#999";
    let boardWidth  = 1 + (width * cellWidth);
    let boardHeight = 1 + (height * cellHeight);
    let canvas;

    onMount(() => {
        const ctx = canvas.getContext('2d');
        ctx.clearRect(0, 0, boardWidth, boardHeight);
        ctx.beginPath();

        // vertical lines
        for (let x = 0; x <= boardWidth; x += cellWidth) {
            ctx.moveTo(0.5 + x, 0);
            ctx.lineTo(0.5 + x, boardHeight);
        }

        // horizontal lines
        for (let y = 0; y <= boardHeight; y += cellHeight) {
            ctx.moveTo(0, 0.5 + y);
            ctx.lineTo(boardWidth, 0.5 +  y);
        }

        // draw the board
        ctx.strokeStyle = colorStroke;
        ctx.stroke();

        ctx.closePath();
    });

</script>

<canvas
    bind:this={canvas}
    width={boardWidth}
    height={boardHeight}
></canvas>

import { writable } from 'svelte/store';

class History {
    constructor() {
        this.history = new Array;
        this.current = -1;
    }

    currentState() {
        return this.history[this.current];
    }

    push(state) {
        // TODO: remove all redo states

        this.current++;
        this.history.push(state);
    }
}

function createHistory() {
    const { subscribe, set, update } = writable(new History);

    return {
        subscribe,
        push: (state) => update(h => { h.push(state); return h; }),
    };
}

export const history = createHistory();
