複素数で遊ぶ

こんにちは！

別のエッセイ。今回は、数式とその視覚化を使用して、複素数で遊んでみましょう。



考え


複素数は実数の特定の拡張であり、実際には、公理のセット全体が定義されているベクトルです。複素数（したがって実数）は、次のように書くことができます。z=a+biここで、aは実部、bは虚部、iは方程式の根です。 x2=−1。彼にとって、実数に対して定義されている多くの操作が定義されています。たとえば、z1+z2=a1+a2+(b1+b2)i。興味深いことに、それらを使ってさまざまな演算を行う場合は、累乗、乗算などを行ってから、Re(z) （実数部）軸Ox、および Im(z)（架空の部分）Oy軸では、面白い写真を得ることができます。

ちなみに、私自身も以下の数式を考え出しました。

可視化機能


ルーチン。この反復関数に従ってフィールド上のすべてを描画する関数：

import random
import numpy as np
def vis(A, f, step=1.0, c=None):
    x = []
    y = []
    for B in np.arange(0, A, step):
        v = f(A, B)
        x.append(v.real)
        y.append(v.imag)
    plt.figure(figsize=[8, 8])
    mxabs = max([i[0] ** 2 + i[1] ** 2 for i in zip(x, y)]) ** 0.5
    x = np.array(x) / mxabs
    y = np.array(y) / mxabs
    if c is None:
        plt.scatter(x, y)
    else:
        plt.scatter(x, y, color=[c(x[i], y[i]) for i in range(len(x))])
    plt.show()

すべての関数は2つのパラメーターAとBに依存しています。さらに、vis（）内でBを反復処理します。Aは関数のグローバルパラメーターです。

全コード

import numpy as np
import random
import matplotlib.pyplot as plt
import math

def vis(A, f, step=1.0, c=None):
    x = []
    y = []
    for B in np.arange(0, A, step):
        v = f(A, B)
        x.append(v.real)
        y.append(v.imag)
    plt.figure(figsize=[7, 7])
    mxabs = max([i[0] ** 2 + i[1] ** 2 for i in zip(x, y)]) ** 0.5
    x = np.array(x) / mxabs
    y = np.array(y) / mxabs
    if c is None:
        plt.scatter(x, y)
    else:
        plt.scatter(x, y, color=[c(x[i], y[i]) for i in range(len(x))])
    plt.show()

def func_1(A, B):
    return math.sin(B) * B * math.e ** (1j * (B * math.cos(A)))

def func_2(A, B):
    return math.cos(B) * math.sin(B) * B * math.e ** (1j * (B * math.cos(A)))

def func_3(A, B):
    return math.cos((A + 1) * B) * math.e ** (1j * (B * math.cos(A)))

def func_4(A, B):
    return math.sin(A + B) * B * math.e ** (1j * B * math.sin(A))

def func_5(A, B):
    return math.cos((A + 1) * B) ** 1.5 * math.e ** (1j * (B * math.cos(A)))

def sigm(x):
    return (1 / (1 + 1.2 ** (-x*50)) - 0.5) * 2

color_1 = lambda x, y: (0.2, sigm(x ** 2 + y ** 2) / 1.4, 1 - sigm(x ** 2 + y ** 2))
color_2 = lambda x, y: (sigm(x ** 2 + y ** 2), 0.5, 0.5)
color_3 = lambda x, y: (0.5, 0.5, sigm(x ** 2 + y ** 2))
color_4 = lambda x, y: (sigm(x ** 2 + y ** 2) / 2, 0.5, 1 - sigm(x ** 2 + y ** 2))
colors = [color_1, color_2, color_3, color_4]
funcs = [func_1, func_2, func_3, func_4, func_5]
while True:
    col = random.choice(colors)
    func = random.choice(funcs)
    vis(random.random() * 200 + 100, func, step=0.1, c=col)
    if input() == "exit":
        break
