単一のエントリポイントを持つLD_PRELOADを使用したシステムコールの処理

LLVMでQInstを再実装しているときに、次の問題に遭遇しました。1つのプロセスのエミュレーションモードのQEMUは、すべての「ゲスト」システムコールを自然にインターセプトします。その結果、インストルメンテーションプラグインには、前処理のための単一のエントリポイントがSYS_*あり、引数の数と値によって決定を行うことができます。これはかなり便利です。問題は、すべてのシステムコールが主に行われlibc、コードを静的に書き換えることです。ほとんどの場合、この部分には到達できません。もちろん使用できますptrace、これも同様に設計されています。しかし、別のプロセスなしで実行できるかどうかはわかりません。QInstのセマンティクスは、些細な「同期」インターセプトを暗示していました。どういうわけか、ハンドラーへの呼び出しを挿入する必要があり、これは通常よりもはるかに複雑ですLD_PRELOAD。各システムコールをラップできますが、これは少なくとも不便です（この場合、システムコール自体ではなく、特定のラッパーをインターセプトしないため、何かをスキップできます）。


Under the cut-ソリューションはLLVMに関連付けられていませんが、Linux on x86_64で強化されています（ただし、他のアーキテクチャのLinuxに適応しています）。


注：この記事では、汎用の既製のソリューションを紹介していません- かなり幅広いケースのリストにのみ適しています。しかし、この記事はレビュー-金曜日と見なすことができます。興味深い（願わくば）新しい（ほとんどの場合）情報、ブルートフォースプログラミングのピンチと実例。そして、今日が木曜日であるという事実によって私たちの金曜日が曖昧にならないようにしてください！


まず、決定してみましょう。この記事では、目標はデバッグ防止メカニズムをバイパスすることではなく、一般にプログラムがシステムコールのインターセプトに抵抗しないことを前提としています。一方で、プログラムが私たちを助けるように手動でソースコードを書き直すことも、純粋に労働集約度の点でまあまあの見込みです。したがってLD_PRELOAD。ただし、そのようなメカニズムを独自のプログラムに静的にコンパイルする場合、LD_PRELOADそれは絶対に必須ではありません。

#define MARKER 0x12345678

static int in_handler;

static void handle_sigsys(int num, siginfo_t *si, void *arg)
{
  ucontext_t *ctx = arg;
  greg_t *gregs = ctx->uc_mcontext.gregs;
  uint32_t drop = 0;
  uint64_t res;
  if (!in_handler) {
    in_handler = 1;
    res = handle_syscall(gregs[REG_RAX], &drop,
                         gregs[REG_RDI], gregs[REG_RSI], gregs[REG_RDX],
                         gregs[REG_R10], gregs[REG_R8], gregs[REG_R9]);
    in_handler = 0;
  }
  if (!drop) {
    res = syscall(gregs[REG_RAX],
                  gregs[REG_RDI], gregs[REG_RSI], gregs[REG_RDX],
                  gregs[REG_R10], gregs[REG_R8], MARKER);
  }
  gregs[REG_RAX] = res;
}

void initialize_handler(void)
{
  //  BPF- 
  struct sock_filter filt[] = {
    BPF_STMT(BPF_LD  | BPF_W   | BPF_ABS, (offsetof(struct seccomp_data, args[5]))),
    BPF_JUMP(BPF_JMP | BPF_JEQ | BPF_K  , MARKER, 0, 1),
    BPF_STMT(BPF_RET           | BPF_K  , SECCOMP_RET_ALLOW),
    BPF_STMT(BPF_RET           | BPF_K  , SECCOMP_RET_TRAP),
  };

  struct sock_fprog prog = {
    sizeof(filt) / sizeof(filt[0]), filt
  };

  //  userspace-
  struct sigaction sig;
  memset(&sig, 0, sizeof(sig));
  sig.sa_sigaction = handle_sigsys;
  sig.sa_flags = SA_SIGINFO;
  sigaction(SIGSYS, &sig, NULL);

  // ,  
  prctl(PR_SET_NO_NEW_PRIVS, 1, 0, 0, 0);
  syscall(SYS_seccomp, SECCOMP_SET_MODE_FILTER, 0, &prog);
}

#define ALLOW(sys) \

    BPF_JUMP(BPF_JMP | BPF_JEQ | BPF_K  , sys, 0, 1), \

  struct sock_filter filt[] = {
    BPF_STMT(BPF_LD  | BPF_W   | BPF_ABS, (offsetof(struct seccomp_data, args[5]))),
    BPF_JUMP(BPF_JMP | BPF_JEQ | BPF_K  , MARKER, 0, 1),
    BPF_STMT(BPF_RET           | BPF_K  , SECCOMP_RET_ALLOW),
    ALLOW(SYS_mmap)

  };

#define _GNU_SOURCE

#include <errno.h>
#include <stddef.h>
#include <stdio.h>
#include <stdint.h>

#include <linux/audit.h>
#include <linux/filter.h>

#include <sys/prctl.h>
#include <unistd.h>
#include <syscall.h>
#include <signal.h>
#include <string.h>

uint64_t handle_syscall(uint32_t num, uint32_t *drop_syscall,
                        uint64_t arg1, uint64_t arg2, uint64_t arg3,
                        uint64_t arg4, uint64_t arg5, uint64_t arg6)
{
  if (num == SYS_openat) {
    fprintf(stderr, "openat: %s\n", (const char *)arg2);
  }
  return 0;
}

#define MARKER 0x12345678

static int in_handler;

static void handle_sigsys(int num, siginfo_t *si, void *arg)
{
  ucontext_t *ctx = arg;
  greg_t *gregs = ctx->uc_mcontext.gregs;
  uint32_t drop = 0;
  uint64_t res;
  if (!in_handler) {
    in_handler = 1;
    res = handle_syscall(gregs[REG_RAX], &drop,
                         gregs[REG_RDI], gregs[REG_RSI], gregs[REG_RDX],
                         gregs[REG_R10], gregs[REG_R8], gregs[REG_R9]);
    in_handler = 0;
  }
  if (!drop) {
    res = syscall(gregs[REG_RAX],
                  gregs[REG_RDI], gregs[REG_RSI], gregs[REG_RDX],
                  gregs[REG_R10], gregs[REG_R8], MARKER);
  }
  gregs[REG_RAX] = res;
}

void initialize_handler(void)
{
#define ALLOW(sys) \
    BPF_STMT(BPF_LD  | BPF_W   | BPF_ABS, (offsetof(struct seccomp_data, nr))), \
    BPF_JUMP(BPF_JMP | BPF_JEQ | BPF_K  , sys, 0, 1), \
    BPF_STMT(BPF_RET           | BPF_K  , SECCOMP_RET_ALLOW),
  struct sock_filter filt[] = {
    BPF_STMT(BPF_LD  | BPF_W   | BPF_ABS, (offsetof(struct seccomp_data, args[5]))),
    BPF_JUMP(BPF_JMP | BPF_JEQ | BPF_K  , MARKER, 0, 1),
    BPF_STMT(BPF_RET           | BPF_K  , SECCOMP_RET_ALLOW),
    ALLOW(SYS_mmap)
    BPF_STMT(BPF_RET           | BPF_K  , SECCOMP_RET_TRAP),
  };

  struct sock_fprog prog = {
    sizeof(filt) / sizeof(filt[0]), filt
  };

  struct sigaction sig;
  memset(&sig, 0, sizeof(sig));
  sig.sa_sigaction = handle_sigsys;
  sig.sa_flags = SA_SIGINFO | SA_NODEFER;
  sigaction(SIGSYS, &sig, NULL);

  prctl(PR_SET_NO_NEW_PRIVS, 1, 0, 0, 0);
  syscall(SYS_seccomp, SECCOMP_SET_MODE_FILTER, 0, &prog);
}

static void __attribute__((constructor))constr(void)
{
  initialize_handler();
}
