Hardware


Um einen minimal funktionierenden Prototyp zu erstellen, benötigen wir:

  1. Ein Stück Brett oder irgendetwas, an dem Sie alle Komponenten befestigen können
  2. Fast jeder Mikrocontroller. Ich nahm Arduino Uno als die einfachste Option
  3. GPS-Modul (wir nehmen Datum, Uhrzeit und Koordinaten daraus). Theoretisch können Sie stattdessen das Echtzeituhrmodul verwenden, aber dann müssen Sie Ihre Koordinaten manuell eingeben, und das Gerät funktioniert nicht sofort, aber die Zeit für den Kaltstart wird erheblich verkürzt
  4. Zwei Servos oder besser noch zwei Schrittmotoren
  5. Laserpointer
  6. Eine andere Kleinigkeit: ein Lötkolben, ein Schmelzkleber, ein Steckbrett, Drähte, ein Knopf, ein Kondensator und direkte Hände.

Es gibt keine Tricks zu montieren: Befestigen Sie einfach alle Komponenten mit Schmelzkleber auf der Basis.
Stellen Sie einfach sicher, dass die Achsen der Servos und des Lasers so senkrecht wie möglich zueinander stehen. Dadurch wird der Zeigefehler verringert.

Verbindung


Auch hier nichts Kompliziertes:

  • GPS ist über uart verbunden (nur Rx wird benötigt, da wir nichts an das Modul senden müssen
  • Servos - in Stiften 10 (Azimutachse) und 11 (Höhenachse)
  • Taste - 2 Pin
  • Stromversorgung für alle Module
  • optional - Leistungskondensator

void GpsGetData()
{
  while (Serial.available() > 0)
    gps.encode(Serial.read());
  if (gps.location.isUpdated())
  {
    phi = gps.location.lat(); //   (double)
    lambda = gps.location.lng(); //    (double)
    yr = gps.date.year(); //  (2000+) (u16)
    mo = gps.date.month(); //  (1-12) (u8)
    d = gps.date.day(); //  (1-31) (u8)
    h = gps.time.hour(); //  (0-23) (u8)
    m = gps.time.minute(); //  (0-59) (u8)
    s = gps.time.second(); //  (0-59) (u8)
  }
}
void CalculateParams()
{
  float _time = (h + (float)m / 60 + (float)s / 3600) * 360.0 / 24.0;//   
  float sunlambda = (float)Days(d, mo, yr) * 360 / 365; //      
  sundelta = asin(0.398749 * sin(sunlambda * 2 * PI / 360));//    
  sunalpha = asin(tan(sundelta) / 0.434812); //      
  float tS = _time + lambda - 180 + sunalpha * 360 / (2 * PI) - alpha; //     
  height = 360 / 2 / PI * asin(sin(phi * 2 * PI / 360) * sin(delta * 2 * PI / 360) + cos(phi * 2 * PI / 360) * cos(delta * 2 * PI / 360) * cos(tS * 2 * PI / 360));// 
  az = 360 / 2 / PI * asin(sin(tS * 2 * PI / 360) * cos(delta * 2 * PI / 360) / cos(height * 2 * PI / 360)); //   (   )
  if (tS > 90 || tS < -90)
  {
    az = 180 - az;
  }
 // if (az > 180 && az < 270)

}

void setup() {
  Serial.begin(9600);
  yaw.attach(10, 90);
  pitch.attach(11, 180);
  yaw.setSpeed(20);
  yaw.setAccel(0.1);
  pitch.setSpeed(20);
  pitch.setAccel(0.1);
  yaw.setAutoDetach(false);
  pitch.setAutoDetach(false);
  pinMode(LED_BUILTIN, OUTPUT);
}

void loop() {
  but.tick();
  yaw.tick();
  pitch.tick();
  if (but.isSingle())
  {
    counter += 1;
    if (counter == 9)
      counter = 0;
  }
  alpha = Coordinates[counter][0];
  delta = Coordinates[counter][1];
  GpsGetData();
  CalculateParams();
  digitalWrite(LED_BUILTIN, gps.location.isValid());
  if (az > 90 && az < 180)
  {
    yaw.setTargetDeg(270 - az);
    pitch.setTargetDeg(height);
  }
  if (az < -90 && az > -180)
  {
    yaw.setTargetDeg(-90 - az);
    pitch.setTargetDeg(height);
  }
  if (az < 90 && az > -90)
  {
    yaw.setTargetDeg(90.0 - az);
    pitch.setTargetDeg(180 - height);
  }
}

Astronomieunterricht


Über Astrotermin und etwas anderes
- , – .

... und es gibt nur einen Moment - die Funktion CalculateParams ()
Was diese Funktion tun sollte: Akzeptieren Sie die Koordinaten des Sterns im äquatorialen System (Aufstieg und Abstieg nach rechts), die Zeit und die Koordinaten des Beobachters und geben Sie die Höhe und den Azimut des Objekts an, d. h. Übersetzen Sie im Wesentlichen die Koordinaten des Sterns vom Äquatorialsystem (in dem die Sterne bewegungslos sind) in die Horizontale (in der sich die Sterne tagsüber bewegen).

Dies wird unter Verwendung der Formeln der sphärischen Trigonometrie sowie der sphärischen Astronomie im Vakuum realisiert .

Der Algorithmus ist wie folgt:

  1. Berechnen Sie die Ekliptiklänge der Sonne
  2. Sonnendeklination berechnen
  3. Berechnen Sie den direkten Sonnenaufgang
  4. Berechnen Sie den Stundenwinkel eines Sterns
  5. Höhe und Azimut berechnen

So wird es implementiert:
