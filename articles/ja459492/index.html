//UART1 initialize
// desired baud rate:19200
// actual baud rate:19231 (0,2%)
// char size: 9 bit
// parity: Disabled
void uart1_init(void)
{
 UCSR1B = 0x00; //disable while setting baud rate
 UCSR1A = 0x00;
 UCSR1C = 0x06;
 UBRR1L = 0x33; //set baud rate lo
 UBRR1H = 0x00; //set baud rate hi
 UCSR1B = 0x94;
}

#pragma interrupt_handler uart1_rx_isr:iv_USART1_RXC
void uart1_rx_isr(void) {
  unsigned char tmp;
  unsigned int rcv = 0;

  if (UCSR1B & 0x02) {
    rcv = 0x100;
  }

  rcv |= UDR1;

  tmp = (rcv >> 4) & 0x0F;

  if (rcv & 0x100) {
    tmp |= 0xC0;
  }
  else {
    tmp |= 0x80;
  }

  txBuf12 = (rcv & 0x0F);
  txBuf11 = tmp;
  txState1 = 0;

  TX_ON();

  msCounter0 = 5000;
}
