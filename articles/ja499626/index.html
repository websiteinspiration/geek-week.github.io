LaravelおよびJSON-RPC 2.0上のSOA

SOA（サービス指向アーキテクチャー）は、疎結合されたサービスを組み合わせて相互作用させることによって構築されます。

実例として、クライアントとサーバーの2つのアプリケーションを作成し、リモートプロシージャコールプロトコルを使用してそれらの相互作用を整理しますJSON-RPC 2.0。

お客様


クライアントアプリケーションは、特定のコンテンツを作成および表示するためのサイトです。クライアントには独自のデータベースは含まれていませんが、サーバーアプリケーションとの対話によりデータを受信して​​追加します。

クライアントでは、インタラクションがクラスを提供しますJsonRpcClient

namespace ClientApp\Services;

use GuzzleHttp\Client;
use GuzzleHttp\RequestOptions;

class JsonRpcClient
{
    const JSON_RPC_VERSION = '2.0';

    const METHOD_URI = 'data';

    protected $client;

    public function __construct()
    {
        $this->client = new Client([
            'headers' => ['Content-Type' => 'application/json'],
            'base_uri' => config('services.data.base_uri')
        ]);
    }

    public function send(string $method, array $params): array
    {
        $response = $this->client
            ->post(self::METHOD_URI, [
                RequestOptions::JSON => [
                    'jsonrpc' => self::JSON_RPC_VERSION,
                    'id' => time(),
                    'method' => $method,
                    'params' => $params
                ]
            ])->getBody()->getContents();

        return json_decode($response, true);
    }
}

ライブラリが必要GuzzleHttpです。事前にインストールしてください。
をPOST使用して完全に標準的なリクエストを作成しGuzzleHttp\Clientます。ここでの主な注意点は、リクエストの形式です。
仕様によると、JSON-RPC 2.0リクエストは次のようになります。

{
    "jsonrpc": "2.0", 
    "method": "getPageById",
    "params": {
        "page_uid": "f09f7c040131"
    }, 
    "id": "54645"
}

  • jsonrpc プロトコルバージョン、「2.0」を示す必要があります
  • method メソッド名
  • params パラメータを持つ配列
  • id リクエストID

回答 

{
    "jsonrpc": "2.0",
    "result": {
        "id": 2,
        "title": "Index Page",
        "content": "Content",
        "description": "Description",
        "page_uid": "f09f7c040131"
    },
    "id": "54645"
}

リクエストがエラーで完了した場合、

{
    "jsonrpc": "2.0",
    "error": {
        "code": -32700,
        "message": "Parse error"
    },
    "id": "null"
}

  • jsonrpc プロトコルバージョン、「2.0」を示す必要があります
  • result成功したクエリ結果の必須フィールド。エラーが発生したときに存在してはならない
  • errorエラーが発生した場合の必須フィールド。成功した結果には存在すべきではない
  • id クライアントによって設定されたリクエスト識別子

サーバーが答えを形成するので、戻ってきます。

コントローラでは、必要なパラメータを使用してリクエストを作成し、レスポンスを処理する必要があります。

namespace ClientApp\Http\Controllers;

use App\Services\JsonRpcClient;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Redirect;

class SiteController extends Controller
{
    protected $client;

    public function __construct(JsonRpcClient $client)
    {
        $this->client = $client;
    }

    public function show(Request $request)
    {
        $data = $this->client->send('getPageById', ['page_uid' => $request->get('page_uid')]);

        if (empty($data['result'])) {
            abort(404);
        }

        return view('page', ['data' => $data['result']]);
    }

    public function create()
    {
        return view('create-form');
    }

    public function store(Request $request)
    {
        $data = $this->client->send('create', $request->all());

        if (isset($data['error'])) {
            return Redirect::back()->withErrors($data['error']);
        }

        return view('page', ['data' => $data['result']]);
    }
}

JSON-RPC固定応答フォーマットを使用すると、リクエストが成功したかどうかを簡単に確認でき、応答にエラーが含まれている場合にアクションを実行できます。

アドレスのサーバーによって受信されたすべての要求は<server_base_uri>/data、クラスによって処理されますJsonRpcServer

namespace ServerApp\Services;

class JsonRpcServer
{
    public function handle(Request $request, Controller $controller)
    {        
        try {
            $content = json_decode($request->getContent(), true);

            if (empty($content)) {
                throw new JsonRpcException('Parse error', JsonRpcException::PARSE_ERROR);
            }
            $result = $controller->{$content['method']}(...[$content['params']]);

            return JsonRpcResponse::success($result, $content['id']);
        } catch (\Exception $e) {
            return JsonRpcResponse::error($e->getMessage());
        }
    }
}

クラスJsonRpcServerは、渡されたパラメーターを使用して目的のコントローラーメソッドをバインドします。そして、クラスによって生成された応答JsonRpcResponseを、JSON-RPC 2.0上記の仕様に従った形式で返します。

use ServerApp\Http\Response;

class JsonRpcResponse
{
    const JSON_RPC_VERSION = '2.0';

    public static function success($result, string $id = null)
    {
        return [
            'jsonrpc' => self::JSON_RPC_VERSION,
            'result'  => $result,
            'id'      => $id,
        ];
    }

    public static function error($error)
    {
        return [
            'jsonrpc' => self::JSON_RPC_VERSION,
            'error'  => $error,
            'id'      => null,
        ];
    }
}
