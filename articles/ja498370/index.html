さてささいなこと-確認ダイアログのビューフラグメントを追加します：

ConfirmDialog.fragment.xml

    xmlns='sap.m'
    xmlns:core='sap.ui.core' >
    <Dialog
        id='confirmDialog'
        title=''
        type='Message'
        state='Warning'>
        <content>
            <Label text='  ?' />
        </content>
        <beginButton>
            <Button
                text=''
                press='handleConfirmBtn'/>
        </beginButton>
        <endButton>
            <Button
                text=''
                press='handleCancelBtn'/>
        </endButton>
    </Dialog>
</core:FragmentDefinition>

そしてそれにコントローラー：

ConfirmDialog.controller.js

sap.ui.define([
    "sap/ui/base/ManagedObject"
], function (ManagedObject) {
    "use strict";

    return ManagedObject.extend("project.ConfirmDialog", {

        constructor: function (oView, fnConfirmBtn) {
            this.__oView = oView;
            this.__fnConfirmBtn = fnConfirmBtn;
        },

        exit: function () {
            delete this.__oView;
            delete this.__fnConfirmBtn;
        },

        open: function () {
            const oView = this.__oView;
            let oDialog = oView.byId("confirmDialog");

            if (!oDialog) {
                const oFragmentController = {
                    handleConfirmBtn: () => {
                        this.__fnConfirmBtn();
                        oDialog.close();
                    },
                    handleCancelBtn: () => {
                        oDialog.close();
                    }
                };
                oDialog = sap.ui.xmlfragment(oView.getId(), "project.view.fragment.ConfirmDialog", oFragmentController);
                oView.addDependent(oDialog);
            }
            oDialog.open();
        }

    });
});

そして、初めてこのような通訳の醜い振る舞いを見たとき、黙って自分を疑い始めました。私は長い時間のための様々な枠組みで取り組んできましたが、私はそれのようなものに遭遇したことがありません：することが知られているのconstは、その存在全体のオブジェクトと配列の不変性を保証するものではありません-しかし、aSelectedも、どこにも言及されていません。それで、彼は現れて、割り当てられました、そして、今、彼はコールバックに転送されました。

しかし、私は長い間あなたを苦しめません。

それはすべてのjsプログラマーが愛するコンテキストとクロージャーについてです。

つまり、実際には、handleConfirmBtnの最初の実行時に、コンテキスト全体（aSelectedを含む）と一緒にハンドラへのリンクがまだあります。そしてその後の削除の確認とともに、呼ばれるのは彼女です。

エラーを修正する方法はそれほど単純で明確ではありませんでした。oFragmentController宣言を移動するだけでは不十分でした（最初の呼び出しの後、コンテキストは失われました）。そして、以下は最も簡潔な方法でした（私はopenメソッドのコードのみを提供します）：

open: function () {
    const oView = this.__oView;
    let oDialog = oView.byId("confirmDialog");

    if (!oDialog) {
        const oFragmentController = {
            handleConfirmBtn: function () {
                this.__fnConfirm();
                oDialog.close();
            },
            handleCancelBtn: function () {
                oDialog.close();
            }
        };
        oDialog = sap.ui.xmlfragment(oView.getId(), "project.view.fragment.ConfirmDialog", oFragmentController);
        oDialog.controller = oFragmentController;
        oView.addDependent(oDialog);
    }
    oDialog.controller.__fnConfirm = this.__fnConfirmBtn.bind(this);
    oDialog.open();
}

最後の4つの実行可能行に注意してください。このようにして、正しいコンテキストと共に現在のハンドラーへのポインターを「ロール」します。

概して、元のバージョンに追加する必要があったのは数行だけでした。
