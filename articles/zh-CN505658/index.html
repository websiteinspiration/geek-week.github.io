在启动和速度下，我们创建一个创建环境的标准函数（它只能运行一次）-setup（）在内部创建画布-createCanvas（600,600）并通过for循环启动工厂以创建星星，首先将自身限制为800。

function setup() {
  createCanvas(600, 600);
  for (let i = 0; i < 800; i++) {
    stars[i] = new Star();
  }
}


之后，我们创建第二个标准函数-draw（），不同之处在于它在无限循环中工作。我们通过background（）将背景涂成黑色，然后通过for循环计算星星（800）数组的长度，并引用数组中的每个星星，我们调用函数（在这种情况下为方法）update（）和show（）

function draw() {
  background("black");
  for (let i = 0; i < stars.length; i++) {
    stars[i].update();
    stars[i].show();
  }
}


就是这样，我们飞了，但是有两个问题。第一个-由于默认情况下坐标的参考点不在中心，而是在左上角，因此，我们仅看到不具有初始负坐标值的那部分恒星。其余的则不可见。



解决这个问题很容易-您只需要通过平移命令（宽度/ 2，高度/ 2）移动中心即可。



第二个问题是恒星在靠近观察者时不会改变其大小。在潜意识层面上，我们认为有些事情是不对的。

为了解决这个问题，让我们回到椭圆，创建另一个变量r，该变量也将取决于z并确定椭圆的第三和第四参数，这些参数负责水平和垂直直径。最终代码如下：

let stars = [];
let speed = 25;

function setup() {
  createCanvas(600, 600);
  for (let i = 0; i < 800; i++) {
    stars[i] = new Star();
  }
}

function draw() {
  background("black");
  translate(width / 2, height / 2);
  for (let i = 0; i < stars.length; i++) {
    stars[i].update();
    stars[i].show();
  }
}

function Star() {
  this.x = random(-width, width);
  this.y = random(-height, height);
  this.z = random(width);
 
  this.update = function() {
    this.z = this.z - speed;
    if (this.z < 1) {
      this.x = random(-width, width);
      this.y = random(-height, height);
      this.z = width;
    }
  };

  this.show = function() {
    fill("white");
    var sx = map(this.x / this.z, 0, 1, 0, width);
    var sy = map(this.y / this.z, 0, 1, 0, height);
    var r = map(this.z, 0, width, 10, 0);
    ellipse(sx, sy, r, r);
  };
}
