Dockをクリーンアップし、xCodeなしでアプリケーションを作成します



ステータスバーからアプリケーションを起動するプログラムを作成します。
端末、Swiftバージョン3.1以降、および任意のテキストエディタが必要です。
macOS Sierra 10.12.6とmacOS Catalina 10.15でテストしました。

ターミナル/アプリケーション/ユーティリティ/ターミナルを開き、ファイルを作成します。

touch toolbar.swift

toolbar.swiftファイルを開き、次のコードを記述します。

import AppKit
var app: NSApplication
var statusItem: NSStatusItem
#if swift(>=5.1)
    app = NSApplication.shared
    statusItem = NSStatusBar.system.statusItem(
        withLength: CGFloat(NSStatusItem.variableLength))
#else
    app = NSApplication.shared()
    statusItem = NSStatusBar.system().statusItem(withLength: CGFloat(32))
#endif

if #available(macOS 10.10, *) {
    statusItem.button?.title = "\u{2699}\u{FE0F}"
}

// extension

// menu

app.setActivationPolicy(.prohibited)
app.run()

コードの一部は、古いバージョンの言語との互換性を維持するために必要です。
アクティブなアプリケーションのアイコンがDockに表示されないように、app.setActivationPolicy（.prohibited）という行が必要です。

ファイルを保存し、次のコマンドでプログラムを実行します。

swift toolbar.swift

エラーがない場合は、ステータスパネルに「役に立たない」ボタンが表示されます。UTFの歯車記号「\ u {2699} \ u {FE0F}」を使用しましたが、statusItem.button？.Imageプロパティを使用して、必要な画像を追加できます。



端末のctrl-cをクリックしてプログラムを終了します。

そのような小さなアプリケーションにはAppDelegateクラスを作成しないことにしましたが、NSApplicationクラスを拡張しました。//拡張の後に次の行を追加します。

extension NSApplication {
    func runTask(_ appName: String, _ arg: String = "") {
        let task = Process()
        #if swift(>=5.1)
            task.executableURL = URL(fileURLWithPath: "/usr/bin/open")
        #else
            task.launchPath = "/usr/bin/open"
        #endif
        task.arguments = arg.isEmpty ?[appName] : [arg, appName]
        #if swift(>=5.1)
            do {try task.run()}
            catch {print(error)}
        #else
            task.launch()
        #endif
    }
    @objc func securityRun () {
        self.runTask("/System/Library/PreferencePanes/Security.prefPane")
    }
    @objc func diskRun () {
        self.runTask( "disk utility", "-a")
    }
    @objc func automatorRun () {
        self.runTask( "automator", "-a")
    }
}

この方法では、アプリケーションをより速く実行できるだけでなく、個々のSystemPreferenceページも実行できることに注意してください。

メソッドにはインターフェースが必要です。//メニュータイプの後：

let menu = NSMenu()
let items: [String] = ["Security", "Disk Utility", "Automator", "Quit"]

var sel: [Selector] = []
let shared = NSApplication.shared

#if swift(>=5.1)
    sel = [
        #selector(shared.securityRun), #selector(shared.diskRun),
        #selector(shared.automatorRun), #selector(shared.terminate)]
#else
    sel = [
        #selector(shared().securityRun), #selector(shared().diskRun),
        #selector(shared().automatorRun), #selector(shared().terminate)]
#endif

for i in 0..<items.count {
    menu.addItem(NSMenuItem(
        title: items[i], action: sel[i], keyEquivalent: ""))
    if i==items.count-2 {
        menu.addItem(NSMenuItem.separator())
    }
}
statusItem.menu = menu

すべてのメソッドについて、NSApplicationクラスのインスタンスからボタンとそれにアタッチされたメソッドを作成しました。ユーザーがターミナルを使用せずにプログラムを終了できるようにする必要があるため、「終了」ボタンを追加しました。

プログラムをコンパイルします。swiftの代わりに、swiftcと入力することに注意してください。

swiftc toolbar.swift

ツールバーファイルが現在のディレクトリに表示されました。次のコマンドで開始できます。

./toolbar

ツールバーファイルをダブルクリックで実行すると、ターミナルウィンドウがアプリケーションとともに表示されますが、これはあまり便利ではありません。

状況を修正してみましょう。すべてのコマンドをターミナルで起動します。

まず、アプリケーションのディレクトリツリーを作成します。

mkdir -p ToolBar.app/Contents/MacOS

アプリケーションへのアクセスを変更します。

chmod a+x ToolBar.app

アイコンのディレクトリを作成します。

mkdir ToolBar.app/Contents/Resources

標準アイコンの1つの名前をコピーして変更します。

cp /System/Library/CoreServices/CoreTypes.bundle/Contents/Resources/ToolbarAdvanced.icns ToolBar.app/Contents/Resources/AppIcon.icns

コンパイルされたツールバーファイルをMacOSディレクトリに移動します。

mv toolbar ToolBar.app/Contents/MacOS

アクセスモードをツールバーファイルに変更します。

chmod a+x ToolBar.app/Contents/MacOS/toolbar

アプリケーションのタイプに関する情報を含むPkgInfoファイルを作成します。-nフラグは、PkgInfoファイルに改行文字がないようにするために必要です。

echo -n "APPL????" > ToolBar.app/Contents/PkgInfo

アクセスモードをPkgInfoファイルに変更します。

chmod a+x ToolBar.app/Contents/PkgInfo

アプリケーション情報を含む最小限のInfo.plistを作成します。

echo '<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN" "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
<plist version="1.0">
<dict>
  <key>CFBundleIdentifier</key>
  <string>example.ToolBar</string>
  <key>CFBundleExecutable</key>
  <string>toolbar</string>
  <key>CFBundleIconFile</key>
  <string>AppIcon</string>
</dict>
</plist>' > ToolBar.app/Contents/Info.plist

ToolBar.appアプリケーションを/アプリケーション/ユーティリティに移動します。
ターミナルは管理者パスワードの入力を求めてきます。

sudo mv ToolBar.app /Applications/Utilities

これらの操作の後、新しいアプリケーションは/アプリケーション/ユーティリティに表示され、Launchpadから起動できます（何らかの理由で、ユーティリティを含むディレクトリはその他と呼ばれます）



アプリケーションを自動実行に追加するために残ります。

これを行うには、ファイル〜/ Library / LaunchAgents / example.ToolBar.plistを作成します。これを1つのコマンドにします。


echo '<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN" "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
<plist version="1.0">
<dict>
    <key>Label</key>
    <string>example.ToolBar</string>
    <key>ProgramArguments</key>
    <array>
        <string>open</string>
        <string>/Applications/Utilities/ToolBar.app</string>
        <string>--args</string>
        <string>-silent</string>
    </array>
    <key>RunAtLoad</key>
    <true/>
</dict>
</plist>' > ~/Library/LaunchAgents/example.ToolBar.plist

目的のファイルが表示されるかどうかを確認します。

ls ~/Library/LaunchAgents

再起動せずに自動実行を確認するには、アプリケーションを終了し、ターミナルでコマンドを実行します。

launchctl load -w ~/Library/LaunchAgents/example.ToolBar.plist

アプリケーションアイコンがステータスバーに表示されます。
アプリケーションを終了し、コマンドをアンロードします。

launchctl unload -w ~/Library/LaunchAgents/example.ToolBar.plist

これで、システムの再起動時にアプリケーションが起動し、Dockから余分なアイコンを削除できます。

ほとんどの場合、この方法で、iTunes用のミニプレーヤー、シンプルなToDoリスト、またはデスクトップ上のウィンドウを揃えるためのインターフェイスを作成できます。
