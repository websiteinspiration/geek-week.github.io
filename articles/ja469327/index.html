2を底とする対数を計算する1つの方法

対数の計算は、デジタル信号処理ではかなり一般的な操作です。多くの場合、たぶん、たたみ込み（累算を伴う乗算）と位相を伴う振幅のみを考慮する必要があります。通常、CORDICアルゴリズムは、FPGAの対数を計算するために使用されます。双曲線バージョンでは、テーブルと単純な算術演算のみが必要です。ただし、これは常に便利であるとは限りません。特にプロジェクトが大きく、クリスタルが小さく、最適化が始まるとダンスが始まります。そんな状況で、私はある日直面しなければなりませんでした。 RAMブロック（Cyclone IV）の両方のポートはすでにしっかりと機能しており、空きウィンドウはありません。双曲線CORDICに別のブロックを使用したくありませんでした。しかし、乗数があり、それに対して時間線図にまともなフリーウィンドウが表示されました。 1日を考えた後、次のアルゴリズムを作成しました。このアルゴリズムでは、テーブルは使用されませんが、乗算があります。より正確には2乗です。また、回路の二乗は一般的な乗算の場合よりも簡単なため、おそらくこのアルゴリズムは特殊なチップにとって興味深いものですが、FPGAには違いはありません。カットの下の詳細。

何が何であるかを説明すると、実数の方が簡単です。それらから始めましょう。後で整数の実装に進みます。

それがXの数だとしましょう。次のような数値Yを求めます。X=2Y。
また、Xは1から2の範囲にあると仮定します。Xは常に2のべき乗による乗算または除算によってこの区間に転送できるため、これは一般性をあまり制限しません。以下のためにY、これは簡単な整数を、加算または減算を意味します。したがって、Xは1から2までの間隔にあります。次に、Yは0から1までの間隔にあります。Yを無限2進分数として記述します。

Y=b020+b12−1+...+bn2−n+...


オッズ biこのレコードでは、Yのバイナリ表現のビットだけです。さらに、Yは 1より小さいので、b0= 0。

最初の方程式を二乗しましょう。X2=22Y以前と同様に、2Yのバイナリ表現を記述します。それは明らかです

2Y=b120+b22−1+...+bn2−(n−1)+...

それら。ビットbi同じままで、2人の力が移動しただけです。コウモリb0ゼロのため、ビューに存在しません。

2つのケースが考えられます

。1）X2>2、2Y> 1、 b1=1

2） X2<2、2Y <1 b1=0

最初のケースでは、Xの新しい値としてX2/2第二に- X2。

その結果、タスクは前者に削減されました。新しいXも1〜2の範囲にあり、新しいYは0〜1の範囲にありますが、結果の1ビットがわかりました。将来も同じステップを踏んで、Yのビットをできるだけ多く取得できます。

Cプログラムでどのように機能するか見てみましょう。

#include <stdio.h>
#include <math.h>
int main()
{
	double w=1.4;
	double s=0.0;
	double a=0.5;
	double u=w;
	for(int i=0; i<16; i++)
	{
		u=u*u;
		if(u>2)
		{
			u=u/2;
			s+=a;
		}
		a*=0.5;
	}
	w=log2(w);
	double err=100*abs(2*(s-w)/(s+w));
	printf("res=%f, log=%f, err=%f%c\n",s,w,err,'%');
	return 0;
}

16ビットの精度で対数を計算し、数学ライブラリが提供するものと比較しました。推定されたプログラム：

res = 0.485413、log = 0.485427、err = 0.002931％

結果はライブラリと一致し、0.003％の精度でアルゴリズムの効率を示しています。

整数の実装に移りましょう。

Nビットの2進数の符号なし数値が区間[0、1]を表すとします。便宜上、ユニット番号を考慮します2N、 だがしかし 2N−1、したがってデュース番号 2N+1。前のイメージと似たようなプログラムを作成しますが、整数を使用します。


#include <stdio.h>
#include <math.h>
#define DIG 	18            // 
#define N_BITS  16            //   
unsigned ONE=1<<(DIG-1);      //
unsigned TWO=ONE<<1;          //
unsigned SCALE=1<<(N_BITS+1);   // 
unsigned myLog(unsigned w)
{
	unsigned s=0;
	unsigned long long u=w;
	for(int i=0; i<N_BITS+1; i++)
	{
		s<<=1;
		u=(u*u)>>(DIG-1);  //    !
		if(u&TWO)   //     
		{
			u>>=1;
			s+=1;
		}
		printf("%X\n", (int)u);
	}
	return s;
}
int main()
{
	double w=1.2345678;
	unsigned iw=(unsigned)(ONE*w);
	double dlog=log2(w);
	unsigned ilog=myLog(iw);
	unsigned test=(unsigned)(SCALE*dlog);
	int err=abs((int)(ilog-test));
	printf("val=0x%X, res=0x%X, log=0x%X, err=%d\n",iw,ilog,test,err);
	return 0;
}

さまざまなビット深度（DIG）、計算精度（N_BITS）、および対数引数（w）を使用してプログラムを再生すると、すべてが正しく計算されていることがわかります。パラメータは、このソースに指定されて特に、プログラムが生成：

ヴァル= 0x27819、RES = 0x9BA5、ログイン= 0x9BA6、ERR = 1

これですべてがまったく同じことを行うことveril上のハードウェアの一部を実装するために準備ができていますCのmyLog関数。関数の変数sおよびuをループで出力し、verilogシミュレーターが生成するものと比較できます。これらの変数と鉄の実装との対応は非常に透過的で理解可能です。uは、反復中にXの新しい値を取得する作業レジスターです。s結果が累積されるシフトレジスタです。モジュールのインターフェースは次のようになります。

入力バスは、それぞれCyclone IVの乗算器の幅である18ビットを採用しています。モジュールの数値は正規化されているはずです。それら。上位ビットが1に等しい。私のプロジェクトでは、これは自動的に行われました。しかし、どちらの場合もノーマライザを実装することは、誰にとっても難しくないと思います。計算の精度はnbitsパラメータによって設定され、デフォルトは16です。モジュールはサイクルごとに1ビットをカウントし、16サイクルの間、16ビットの精度で対数を計算します。同じ精度でより正確に、より正確に同じ速度で実行する必要がある場合は、モジュールを複数のデバイスに分割して伝達することが誰にとっても難しくないことを願っています。

これが完全なモジュールとテストコードです
//--------------------- logarithm.v ------------------------------//
module logarithm(
		input                 clk,      //
		input                 wr,       //  
		input[17:0]           din,      //  
		output[nbits-1:0]     dout,     //  
		output                rdy       // 		
		);
	parameter nbits=16;       //    
	
	reg[4:0]  cnt;      //
	reg[17:0] acc;      // -
	reg[nbits-1:0] res; //
	
	always @(posedge clk)
		if(wr) cnt<=nbits+1;
		else
		if(cnt != 0) 
			cnt<=cnt-1;
		
	wire[35:0] square=acc*acc;  // 
	wire bit=square[35];        // 
	wire[17:0] next = bit ? square[35:18] : square[34:17]; //  
	
	always @(posedge clk)
		if(wr) acc<=din;
		else
		if(cnt != 0) begin
			acc<=next;
			#10 $display("%X", acc);
		end
		
	always @(posedge clk)
		if(wr) res<=0;
		else
		if(cnt != 0) begin
				res[nbits-1:1]<=res[nbits-2:0];
				res[0]<=bit;
			end
		
	assign dout=res;
	assign rdy=(cnt==0);
endmodule

//======================== testbench.v =====================//
module testbench();
	
	reg clk;      //
	always 
		#100 clk=~clk;
		
	reg wr;            //
	reg[17:0] din;     //
	wire rdy;          //
	wire[15:0] dout;   // 
		
	logarithm log2(
			.clk    (clk),
			.wr     (wr),
			.din    (din),
			.dout   (dout),
			.rdy    (rdy) 
			);
	
	//  n    
	task skipClk(integer n); 
		integer i;
		begin
			for(i=0; i<n; i=i+1)
				@(posedge clk);
		#10 ;
		end
	endtask	
	
	initial
		begin   //
			$dumpfile("testbench.vcd");
			$dumpvars(0, testbench);
			clk=0;
			wr=0;
			din=18'h27819;
			skipClk(3);
			wr=1;
			skipClk(1);
			wr=0;
			@(rdy);
			skipClk(3);
			$display("value=%X, result=%X", din, dout);
			$display("Done !");
			$finish;
		end
endmodule
