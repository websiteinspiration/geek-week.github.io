nRF52832、DIYプロジェクトのワイヤレス漏水センサー

HabrのDIYまたは日曜大工のセクションのすべての読者への挨拶！今日は私の次のプロジェクトについてお話ししたいと思います。この記事はバッテリー駆動の漏水検知器についてです。以前のプロジェクトと同様に、このデバイスはnRF52832マイクロコントローラーで実行されます。このセンサーには3つのバージョンがあり、3つのバージョンはすべてnRF52832で事前に作成されたモジュールを使用します。この記事では、HOLYIOTのYJ-17103モジュールを使用する中間バージョンについて説明します。



流体検出器は、チップSN74LVC1G00に実装されています。データシット。回路設計と動作原理について簡単に説明します。センサー電極1番はアースに接続され、センサー電極2番はSN74LVC1G00マイクロ回路のレッグAおよびBに100オームの抵抗を介して接続され、1M抵抗を介して3.3Vもこのラインに接続され、静電容量も回路に追加されます。マイクロ回路AとBのレッグで液体との接触がない場合、MKレッグに接続されたレッグYの論理ユニット（組み込みのコンパレーターを介して中断を検出するようにプログラムで構成されています）はそれぞれ論理ゼロです。液体との接触が発生してレッグAとBが低くなるとすぐに、SN74LVC1G00マイクロ回路のレッグYの信号も反転し、中断が発生し、MKがスリープ状態から解除されます。将来、チップSN74LVC1G00はチップSN74LVC1G14に置き換えられる可能性があります。データシット、そして多分そうではない:)内蔵コンパレータを介したMK脚からの液体の検出は計画されていません。

私の他のすべてのプロジェクトと同様に、これもArduinoプロジェクトであり、過去1年間（およそ）のすべてのプロジェクトと同様に、これもMysensorsプロジェクト用に作成されました。他の記事と同様に、この記事ではマイセンサーのトピックについて少し触れます。

Mysensorsは、ソフトウェア開発者のオープンソースコミュニティです。このプロトコルは、無線および有線ネットワークを作成するためにコミュニティによって開発されました。プロジェクトは元々Arduinoプラットフォーム用に開発されました。標準のMysensorsネットワークは、ゲート（ゲートウェイ）、再変換器、およびエンドデバイス（ノード）で構成されています。 1つのネットワークでは、最大254台のデバイスを使用できます。各デバイスには、最大254個のセンサー、センサー、作動ユニットを装備できます。他のデバイスでのネットワーク操作、データ処理、スクリプト実行、および対話は、UDコントローラーを使用して実行されます。一部のコントローラー（Majordomo）は複数のネットワークとMysensors（マルチゲート）の操作をサポートしているため、1つのコントローラーで制御されるネットワークよりもはるかに多くのネットワークが存在する可能性があります。

サポートされているハードウェアプラットフォーム：Linux / Raspberry Pi / Orange Pi | ATMega 328P | ESP8266 | ESP32 | nRF5x（Cortex M0、M4）| Arduino Zero（Cortex M0）で使用されるAtmel SAMD | Teensy3（MK66FX1M0VMD18）| STM32F1。

サポートされている無線送信機：NRF24L01 | RFM69 | RFM95（LoRa）| nRF5x

サポートされている有線通信タイプ：RS485

ゲートとコントローラー間のサポートされている通信タイプ：MQTT |シリアルUSB | Wifi |イーサネット| GSM

漏れセンサーに戻ります。このデバイスは、CR2430、CR2450、またはCR2477バッテリーを搭載しています。睡眠消費量は3μA未満です。伝送速度-250Kbps、10-15ms。送信時の消費電力は8mA以下です。理論的には、1つのバッテリーのバッテリー寿命は、バッテリーの自己放電時間とほぼ同じです。実際には、もちろん、登録、提示、充電レベルの定期的な送信があるので、すべてがバラ色ではありません。したがって、バッテリー寿命は値に近くなります-自己放電時間/ 2 :)。電源はバッテリから直接供給され、バッテリレベルはVDDピンから直接制御されます。センサーには、ネットワーク内のセンサーの登録を示し、サービスモードを示し、リーク検出を示すために、RGB LEDが取り付けられています。当然のことながら、LEDはまったく使用されないこともあれば、一部使用されないこともあります。

デバイスボードは、LUT法に従ってさらに製造するために作成されました。したがって、このオプションのニュアンスから、それはトラックの幅の増加、トラック間の距離の増加、層間移行のための領域の増加（より便利な穴あけのため）、およびボードの小さな領域のために空の領域を埋めることがないことです。その後、生産中の注文のオプションが作成されました。



デバイスケースは2つの部分で設計されました。ボードを取り付けるための場所のある上部カバーと、スチール製のコンタクトネジ用の2つの穴（ネジの頭にシリコンシーラントを使用するか、不要）で下部（バス）とボード上のボタン用の2つのチューブ（リセットとモード）を備えた下部（バス）。印刷はANICUBIC PHOTON SLA 3Dプリンターで行われました。印刷後、サンドペーパー320と1000を処理して、蓋と本体の底面の接合部に合わせました。

void loop() {
  if (nosleep == 0) {
    oldmillis = millis();
    sleep(SLEEP_TIME_W);
  }

  if (detection) {
    if (digitalRead(PIN_BUTTON) == 1 && button_flag == 0 && digitalRead(W_L_SENS) == 0) {
      //back side button detection
      button_flag = 1;
      nosleep = 1;
    }
    if (digitalRead(PIN_BUTTON) == 1 && button_flag == 1 && digitalRead(W_L_SENS) == 0) {
      digitalWrite(GREEN_LED, LOW);
      wait(10);
      digitalWrite(GREEN_LED, HIGH);
      wait(50);
    }
    if (digitalRead(PIN_BUTTON) == 0 && button_flag == 1 && digitalRead(W_L_SENS) == 0) {
      nosleep = 0;
      button_flag = 0;
      digitalWrite(GREEN_LED, HIGH);
      lpComp_reset();
    }

    if (digitalRead(W_L_SENS) == 1 && digitalRead(PIN_BUTTON) == 0) {
      //sens detection
      newmillis = millis();
      interrupt_time = newmillis - oldmillis;
      SLEEP_TIME_W = SLEEP_TIME_W - interrupt_time;
      send(sensMsg.set(detection), 1);
      wait(3000, 1, V_VAR1);
      if (AckG == 1) {
        while (timer < 10) {
          timer++;
          digitalWrite(BLUE_LED, LOW);
          wait(20);
          digitalWrite(BLUE_LED, HIGH);
          wait(30);
        }
        timer = 0;
        AckG = 0;
        wait(200);
      } else {
        while (timer < 10) {
          timer++;
          digitalWrite(RED_LED, LOW);
          wait(20);
          digitalWrite(RED_LED, HIGH);
          wait(30);
        }
        timer = 0;
        send(sensMsg.set(detection), 1);
        wait(3000, 1, V_VAR1);
        if (AckG == 1) {
          while (timer < 10) {
            timer++;
            digitalWrite(BLUE_LED, LOW);
            wait(20);
            digitalWrite(BLUE_LED, HIGH);
            wait(30);
          }
          timer = 0;
          AckG = 0;
        } else {
          while (timer < 10) {
            timer++;
            digitalWrite(RED_LED, LOW);
            wait(20);
            digitalWrite(RED_LED, HIGH);
            wait(30);
          }
          timer = 0;
        }
        lpComp_reset();
      }
    }

    if (SLEEP_TIME_W < 60000) {
      SLEEP_TIME_W = SLEEP_TIME;
      sendBatteryStatus();
    }
  }
  else {
    //if (detection == -1) {
    SLEEP_TIME_W = SLEEP_TIME;
    sendBatteryStatus();
  }
}

void sendBatteryStatus() {
  wait(100);
  batteryVoltage = hwCPUVoltage();
  wait(20);

  if (batteryVoltage > battery_vcc_max) {
    currentBatteryPercent = 100;
  }
  else if (batteryVoltage < battery_vcc_min) {
    currentBatteryPercent = 0;
  } else {
    currentBatteryPercent = (100 * (batteryVoltage - battery_vcc_min)) / (battery_vcc_max - battery_vcc_min);
  }
  sendBatteryLevel(currentBatteryPercent, 1);
  wait(3000, C_INTERNAL, I_BATTERY_LEVEL);
  if (AckB == 1) {
    AckB = 0;
    flag_lq = 1;
  } else {  
    sendBatteryLevel(currentBatteryPercent, 1); 
    wait(3000, C_INTERNAL, I_BATTERY_LEVEL); 
    if (AckB == 1) {
      AckB = 0;
      flag_lq = 1;
    }
  }
  //send(powerMsg.set(batteryVoltage), 1);
  //wait(2000, 1, V_VAR1);

  //sleep(10000); //
  if (flag_lq == 1) {
    linkQuality = calculationRxQuality();
    wait(50);
    sendSignalStrength(linkQuality, 1);
    wait(2000, 1, V_VAR1);
    if (AckL == 1) {
      AckL = 0;
    } else {
      sendSignalStrength(linkQuality, 1);
      wait(2000, 1, V_VAR1);
      if (AckL == 1) {
        AckG = 0;
      }
    }
    flag_lq = 0;
  }
}


void lpComp() {
  NRF_LPCOMP->PSEL = INTR_PIN;
  NRF_LPCOMP->ANADETECT = 1;
  NRF_LPCOMP->INTENSET = B0100;
  NRF_LPCOMP->ENABLE = 1;
  NRF_LPCOMP->TASKS_START = 1;
  NVIC_SetPriority(LPCOMP_IRQn, 15);
  NVIC_ClearPendingIRQ(LPCOMP_IRQn);
  NVIC_EnableIRQ(LPCOMP_IRQn);
}

void s_lpComp() {
  if ((NRF_LPCOMP->ENABLE) && (NRF_LPCOMP->EVENTS_READY)) {
    NRF_LPCOMP->INTENCLR = B0100;
  }
}

void r_lpComp() {
  NRF_LPCOMP->INTENSET = B0100;
}

nRF52832はプログラムによって低電力モード（DC-DCモード）で動作するように構成されており、MCは内部LPCOMPコンパレータを介したSN74LVC1G00マイクロ回路からの信号によってスリープからウェイクアップされます。デバイスには、デバイスのペアリング、デバイスのリセットなどのサービスモードを実装するための時計ボタンもあります。ボタンは、漏れ検知器と同じMKフットに巻かれています。両方のラインはショットキーダイオードで分離されています。モニタリングモードのチップSN74LVC1G00は何も消費しません。マイクロサーキットの電源管理は、MKのレッグから実行されます。

現在、これらのセンサーが機能する漏水コントローラーの開発はほぼ終了しています。
