  });

   this.style = null;

    return;

       this.dataset.mode = "deleted";
       return;

      return;
    }
    this.dataset.mode = "opened";
  });
 }

   this.translateY();
   return;

  this.style.transform = 'translateY(0)';
}

   return;
  }
  this.translateY();
}

  this.style.transform = `translateY(${pixels}px)`;
  this.style.transition = 'none';
 }
