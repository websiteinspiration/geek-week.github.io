PythonからDLLに2次元リストを渡す

みなさん、こんにちは。PythonのC / C ++の記事を少し補足することにしました。


int、bool、floatなどの標準型を渡すのは非常に簡単ですが、それほど必要ではありません。Python自体はすぐにそのようなデータに対処し、誰もがそのようなコードの一部をC / C ++ライブラリに転送する必要があるとは考えられません。


しかし、大きなデータ配列、またはさらに優れた2次元データ配列、またはオブジェクトの2次元配列の転送。


ここではすべてがそれほど明白ではありません。Pythonインタープリターにとって難しいコードセクションを大幅に高速化したい人のために強調できると思われる点がいくつかあります。


カットの下に示した例は、使用にはあまり役に立ちませんが、この手順のすべてのニュアンスを強調するには十分だと思います。


ライブラリファイルのソースコードをすぐに提供します。


py_list2c_array.h


#ifndef _PY_LIST_2_C_ARRAY_H_
#define _PY_LIST_2_C_ARRAY_H_

#include <stdio.h>

typedef struct {
    int value;
    wchar_t* name;
} Item;

extern "C" __declspec(dllexport)
int sum_diagonal(Item** field, size_t size);

#endif

py_list2c_array.cpp


ここでもすべてが標準ですが、wchar_t *のような行を印刷するためのwprintf（）関数の使用に注意しています。


// py_list2c_array.cpp:      DLL.
//

#include "stdafx.h"
#include "py_list2c_array.h"

extern "C" __declspec(dllexport)
int sum_diagonal(Item** field, size_t size) {
    int result = 0;
    for(size_t i=0; i<size;++i) {
        for(size_t j=0; j<size; ++j) {
            if(i == j) {
                result += field[i][j].value;
                wprintf(L"%s\n", field[i][j].name);
            }
        }
    }
    return result;
}

py_list2c_array.py


そして今、最も重要なこと。重要なポイントの説明とともに、Pythonスクリプトのソースコードを示します。


import ctypes

class PyItem:
    def __init__(self, value, name):
        self.value = value
        self.name = name

class CItem(ctypes.Structure):
    _fields_ = [
                 ('value', ctypes.c_int),
                 ('name', ctypes.c_wchar_p)
                ]

def create_list(size):
    return [[PyItem(int(str(i+1)+str(j+1)), 'item{}{}'.format(i+1, j+1)) for j in range(size)] for i in range(size)]

def py_list2c_array(py_list, size):
    rowType = CItem * size
    resultType = ctypes.POINTER(CItem) * size
    result = resultType()
    for i in range(size):
        row = rowType()
        for j in range(size):
            row[j] = CItem()
            row[j].value = py_list[i][j].value
            row[j].name = ctypes.c_wchar_p(py_list[i][j].name)
        result[i] = ctypes.cast(row, ctypes.POINTER(CItem))
    return ctypes.cast(result, ctypes.POINTER(ctypes.POINTER(CItem)))

if __name__ == '__main__':
    sLib = ctypes.cdll.LoadLibrary('./py_list2c_array.dll')

    size = 4
    py_list = create_list(size)
    c_array = py_list2c_array(py_list, size)

    sLib.sum_diagonal.argtypes = [ctypes.POINTER(ctypes.POINTER(CItem)), ctypes.c_size_t]
    sLib.sum_diagonal.restype = ctypes.c_int
    result = sLib.sum_diagonal(c_array, ctypes.c_size_t(size))
    print(': {}'.format(result))
