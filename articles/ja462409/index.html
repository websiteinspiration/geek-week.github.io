AVRマイクロコントローラー用のアセンブラーコードジェネレーターライブラリ


パート1.最初の知人


こんにちは、親愛なるカブロバイト。AVRシリーズの人気のあるマイクロコントローラーをプログラミングするための次の（利用可能な非常に多くの）プロジェクトに注目したいと思います。


これが必要な理由を説明するために多くのテキストを費やすことができますが、代わりに、他のソリューションとの違いの例を見てください。また、既存のプログラミングシステムとのすべての説明と比較は、必要に応じて、例を解析する過程で行われます。ライブラリは現在ファイナライズの段階にあるため、一部の関数の実装は完全に最適ではないように見える場合があります。また、このバージョンでプログラマーに割り当てられているタスクの一部は、さらに最適化または自動化されることになっています。


それでは始めましょう。提示された資料を完全な説明と見なすべきではなく、開発者がこのアプローチが読者にとってどれほど興味深いかを理解するために開発されたライブラリの機能の一部のデモンストレーションとしてのみ見なすべきであることをすぐに明確にしたいと思います。


私たちは一般的な慣習から逸脱せず、マイクロコントローラーの一種の「Hello world」という古典的な例から始めます。つまり、プロセッサレッグの1つに接続されているLEDを点滅させます。MicrosoftからVisualStudioを開いて（どのリリースでも可能です）、C＃用のコンソールアプリケーションを作成します。知らない人のために-仕事に十分なCommunity Editionは完全に無料です。


実際、テキスト自体は次のとおりです。


ソースコード例1
using NanoRTOSLib;
using System;

namespace ConsoleApp
{
    class Program
    {
        static void Main(string[] args)
        {
            var m = new Mega328();
            m.PortB[0].Mode = ePinMode.OUT;
            m.PortB.Activate();
            m.LOOP(m.TempL, (r, l) => m.GO(l), (r) => { m.PortB[0].Toggle();});
            Console.WriteLine(AVRASM.Text(m));
        }
    }
}

using NanoRTOSLib;
using System;

namespace ConsoleApp
{
    class Program
    {
        static void Main(string[] args)
        {
         var m = new Mega328();
            m.FCLK = 16000000;
            m.CKDIV8 = false;
            m.PortB.Direction(0x07);
            var bit1 = m.PortB[1];
            var bit2 = m.PortB[2];
            m.PortB.Activate();
            var tasks = new Parallel(m, 4);
            tasks.Heap = new StaticHeap(tasks, 64);
            var t1 = tasks.CreateTask((tsk) =>
            {
                var loop = AVRASM.NewLabel();
                bit1.Toggle();
                tsk.Delay(32);
                tsk.TaskContinue(loop);
            },"Task1");
            var t2 = tasks.CreateTask((tsk) =>
            {
                var loop = AVRASM.NewLabel();
                bit2.Toggle();
                tsk.Delay(48);
                tsk.TaskContinue(loop);
            }, "Task2");
            var ca = tasks.ContinuousActivate(tasks.AlwaysOn, t1);
            tasks.ActivateNext(ca, tasks.AlwaysOn, t2);
            ca.Dispose();
            m.EnableInterrupt();
            tasks.Loop();
            Console.WriteLine(AVRASM.Text(m));
        }
    }
}

using NanoRTOSLib;
using System;

namespace ConsoleApp
{
    class Program
    {
        static void Main(string[] args)
        {
            var m = new Mega328();
            m.FCLK = 16000000;
            m.CKDIV8 = false;

            var led7s = new Led_7();
            led7s.SegPort = m.PortC;
            led7s.Activate();

            m.PortD.Direction(0xFF);
            m.PortD.Activate();
            m.PortB[0].Mode = ePinMode.OUT;
            var tc77 = new TC77();
            tc77.CS = m.PortB[0];
            tc77.Port = m.SPI;

            m.Timer0.Clock = eTimerClockSource.CLK64;
            m.Timer0.Mode = eWaveFormMode.Normal;
            var reader = m.DREG("Temperature");
            var bcdRes = m.DREG("digits");
            var tmp = m.BYTE();

            var bcd = new BCD(reader, bcdRes);
            m.subroutines.Add(bcd);

            var os = new Parallel(m, 4);
            os.Heap = new StaticHeap(os, 64);

            var tmrSig = os.AddSignal(m.Timer0.OVF_Handler);
            var spiSig = os.AddSignal(m.SPI.Handler, () =>
            {
                m.SPI.Read(m.TempL);
                m.TempL.MStore(tmp);
            });

            var actuator = os.CreateTask((tsk) => 
            {
                var loop = AVRASM.NewLabel();
                tc77.ReadTemperatureAsync();
                tsk.Delay(16);
                tsk.TaskContinue(loop);
            }, "actuator");
            var treader = os.CreateTask((tsk) =>
            {
                var loop = AVRASM.NewLabel();
                tc77.ReadTemperatureCallback(os, reader, tmp);
                reader >>= 7;
                m.CALL(bcd);
                tsk.TaskContinue(loop);
            }, "reader");
            var display = os.CreateTask((tsk) =>
             {
                 var loop = AVRASM.NewLabel();
                 m.PortD.Write(0xFE);
                 m.TempQL.Load(bcdRes.Low);
                 m.TempQL &= 0x0F;
                 led7s.Show(m.TempQL);
                 os.AWAIT();
                 m.PortD.Write(0xFD);
                 m.TempQL.Load(bcdRes.Low);
                 m.TempQL >>= 4;
                 led7s.Show(m.TempQL);
                 os.AWAIT();
                 m.PortD.Write(0xFB);
                 m.TempQL.Load(bcdRes.High);
                 m.TempQL &= 0x0F;
                 led7s.Show(m.TempQL);
                 os.AWAIT();
                 m.PortD.Write(0xF7);
                 m.TempQL.Load(bcdRes.High);
                 m.TempQL >>= 4;
                 led7s.Show(m.TempQL);
                 os.AWAIT();
                 tsk.TaskContinue(loop);
             }, "display");

            var ct = os.ContinuousActivate(os.AlwaysOn, actuator);
            os.ActivateNext(ct, spiSig, treader);
            os.ActivateNext(ct, tmrSig, display);

            tc77.Activate();
            m.Timer0.Activate();
            m.EnableInterrupt();
            os.Loop();

            Console.WriteLine(AVRASM.Text(m));
        }
    }
}
