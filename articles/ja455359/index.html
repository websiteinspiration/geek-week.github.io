@discardableResult
func perform<Result, U, Z>(
    _ transform: (U, Z) throws -> Result,
    _ optional1: U?,
    _ optional2: Z?) rethrows -> Result? {

    guard
        let optional1 = optional1,
        let optional2 = optional2
        else {
            return nil
    }

    return try transform(optional1, optional2)
}
