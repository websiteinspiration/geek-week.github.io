パーティクルシステムを使用してゲーム内メッセージを表示する



仕事


ゲームThe Unlivingを開発するときは、パーティクルシステムを使用して、発生したダメージ、健康やエネルギーの不足、報酬の値、回復したヘルスポイントの数など、さまざまなメッセージを表示するタスクを設定します。Unity UIシステムの標準要素を使用する際に問題となる、メッセージの外観の効果やさらなる動作をカスタマイズする機会を増やすために、これを行うことが決定されました。

さらに、このアプローチは、メッセージのタイプごとに1つのパーティクルシステムインスタンスのみを使用することを意味するため、Unity UIを使用した同じメッセージの出力と比較して、生産性が大幅に向上します。

シェーダーを使用して、準備済みのテクスチャを正しいUV座標を使用して表示します。UV座標の情報は、Vector4リストの形式でParticleSystem.SetCustomParticleDataを使用して、2つのストリーム（頂点ストリーム）によってParticleSystemに送信されます。

私たちの実装では、10行10列の文字を含むテクスチャを使用しています。フォントとして任意の等幅フォントを使用できます。これは、メッセージ文字間の異なる間隔を避けるためです。

Vertex Streamに転送するVector4を作成

する文字セットを記述するために、SymbolsTextureData構造体を使用します。

chars配列は、左上隅からすべてのフォントテクスチャシンボルを追加するために、手動で入力する必要があります。

[Serializable]
public struct SymbolsTextureData
{
    //   
    public Texture texture;
    //    ,   - 
    public char[] chars;
    
    //Dictionary     -    
    private Dictionary<char, Vector2> charsDict;

    public void Initialize()
    {
        charsDict = new Dictionary<char, Vector2>();
        for (int i = 0; i < chars.Length; i++)
        {
            var c = char.ToLowerInvariant(chars[i]);
            if (charsDict.ContainsKey(c)) continue;
            //  ,    
            //    , ,     10.
            var uv = new Vector2(i % 10, 9 - i / 10);
            charsDict.Add(c, uv);
        }
    }

    public Vector2 GetTextureCoordinates(char c)
    {
        c = char.ToLowerInvariant(c);
        if (charsDict == null) Initialize();

        if (charsDict.TryGetValue(c, out Vector2 texCoord))
            return texCoord;
        return Vector2.zero;
    }
}

Unityのパーティクルシステムでは、2つのVector4ストリームの形式でカスタムデータを転送できます。



ストリームの座標のシフトを回避するために、UV2を使用して意図的にストリームを追加しました。これを行わない場合、C＃のCustom1-vectorのXおよびY座標は、ZおよびW TEXCOORD0シェーダーに対応します。したがって、Custom1.z = TEXCOORD1.x、Custom1.w = TEXCOORD1.yとなります。これは将来多くの不便を引き起こすでしょう。


前述のように、2つのVector4を使用して、メッセージの長さと文字のUV座標を伝えます。 Vector4にはfloat型の4つの要素が含まれているため、デフォルトで4 * 4 = 16バイトのデータをパックできます。なぜならメッセージにはメッセージの長さ（2桁の数字）と文字の座標（各文字の2桁の数字）のみが含まれるため、バイト型（0-255）の範囲は冗長です。小数点以下の桁数を使用しても問題ありません。

浮動小数点の精度は6〜9文字です。つまり、各Vector4座標の6ビットを安全に使用でき、データの整合性と精度を心配する必要はありません。実際には、7、8、9文字をパックしようとしましたが、フロートの精度は十分ではありません。

4バイトの標準バージョンとは対照的に、小数点以下の桁数を使用して、各フロートで最大6桁をパックすることがわかります。合計、1つのVector4には24桁の1桁の数字が含まれます。

ストリームでは2つのベクターを転送できるため、両方を使用して最大23文字のメッセージを送信します

Custom2.xyzw-メッセージの別の11文字+メッセージの長さ（最後の2文字）。

//   Vector2     float
public float PackFloat(Vector2[] vecs)
{
    if (vecs == null || vecs.Length == 0) return 0;            
    //      float
    var result = vecs[0].y * 10000 + vecs[0].x * 100000;
    if (vecs.Length > 1) result += vecs[1].y * 100 + vecs[1].x * 1000;
    if (vecs.Length > 2) result += vecs[2].y + vecs[2].x * 10;            
        return result;
}

//  Vector4    CustomData
private Vector4 CreateCustomData(Vector2[] texCoords, int offset = 0)
{
    var data = Vector4.zero;            
    for (int i = 0; i < 4; i++)
    {
        var vecs = new Vector2[3];                
        for (int j = 0; j < 3; j++)
        {
            var ind = i * 3 + j + offset;
            if (texCoords.Length > ind)
            {
                vecs[j] = texCoords[ind];
            }
            else
            {
                data[i] = PackFloat(vecs);
                i = 5; 
                break;
            }
        }
        if (i < 4) data[i] = PackFloat(vecs);
    }
    return data;
}

両方のブロックをSpawnParticle（）メソッドに追加すると、C＃部分の準備が整います。メッセージは、GPUによってパックされ、Vertex Stream内の2つのVector4の形式で送信されます。残っている最も興味深いことは、このデータを受け入れて正しく表示することです。

シェーダーコード

Shader "Custom/TextParticles"
{
    Properties
    {
        _MainTex ("Texture", 2D) = "white" {}
        //         10,    
        _Cols ("Columns Count", Int) = 10
        _Rows ("Rows Count", Int) = 10
    }
    SubShader
    {            
        Tags { "RenderType"="Opaque" "PreviewType"="Plane" "Queue" = "Transparent+1"}
        LOD 100
        ZWrite Off
        Blend SrcAlpha OneMinusSrcAlpha

        Pass
        {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            
            #include "UnityCG.cginc"
            
            struct appdata
            {
                float4 vertex : POSITION;
                fixed4 color : COLOR;
                float4 uv : TEXCOORD0;
                //    customData
                float4 customData1 : TEXCOORD1;
                float4 customData2 : TEXCOORD2;
            };           

            struct v2f
            {
                float4 vertex : SV_POSITION;
                fixed4 color : COLOR;
                float4 uv : TEXCOORD0;
                float4 customData1 : TEXCOORD1;
                float4 customData2 : TEXCOORD2;
            };
            
            uniform sampler2D _MainTex;
            uniform uint _Cols;
            uniform uint _Rows;
            
            v2f vert (appdata v)
            {
                v2f o;
                //        w- ?
                //       .
                //      100.
                float textLength = ceil(fmod(v.customData2.w, 100));

                o.vertex = UnityObjectToClipPos(v.vertex);
                //  UV ,   -   
                o.uv.xy = v.uv.xy * fixed2(textLength / _Cols, 1.0 / _Rows);
                o.uv.zw = v.uv.zw;
                o.color = v.color;                
                o.customData1 = floor(v.customData1);
                o.customData2 = floor(v.customData2);
                return o;
            }
            
            fixed4 frag (v2f v) : SV_Target
            {
                fixed2 uv = v.uv.xy;
                //   
                uint ind = floor(uv.x * _Cols);

                uint x = 0;
                uint y = 0;

                //  ,   
                //0-3 - customData1
                //4-7 - customData2
                uint dataInd = ind / 3;
                //   6     float
                uint sum = dataInd < 4 ? v.customData1[dataInd] : v.customData2[dataInd - 4];

                //  float      
                for(int i = 0; i < 3; ++i)
                {
                    if (dataInd > 3 & i == 3) break;
                    //  ,   10^2 = 99  ..
                    uint val = ceil(pow(10, 5 - i * 2));
                    x = sum / val;
                    sum -= x * val;

                    val = ceil(pow(10, 4 - i * 2));
                    y = sum / val;
                    sum -= floor(y * val);

                    if (dataInd * 3 + i == ind) i = 3;
                }                

                float cols = 1.0 / _Cols;
                float rows = 1.0 / _Rows;
                // UV-,  - , , 
                //     
                uv.x += x * cols - ind * rows;
                uv.y += y * rows;
                
                return tex2D(_MainTex, uv.xy) * v.color;
            }
            ENDCG
        }
    }
}
