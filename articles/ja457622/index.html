Qtパフォーマンスの測定

Aurora OSに関する一連の記事を続けることにしました（最近Sailfishと呼ばれるまで）。私はこのOSを使用しているときに、Qtとシステム全体のアプリケーションのパフォーマンスに関連するさまざまな観察結果を蓄積しました。これは、クリスマスツリーなどのさまざまなデバイスでハングアップしているため、起動時にすべての小さなことに気付きます。これは、Qtを使用している（またはすぐに使用する）同僚にとっても興味深いものであると思います。さらにテストできることを提案します。



私はQtでプログラミングし、同僚、iOS開発者、アプローチの類似点、相違点、利点についてよく話し合います。ある時点で、私たちはまだ言葉から行為に移り、測定を行うことにしました。エンターテインメントに参加する準備ができているAndroidプログラマーを見つけられなかったため、数値とテーブルを含む比較はSwiftとC ++についてのみ行われます。

Qt / C ++はメモリ管理メカニズムを課さず、Swift参照カウントが使用され、Javaではガベージコレクターですが、ユーザーはC ++で利用可能な機能のフレームワーク内でこの問題を決定することを思い出してください。したがって、プログラマーはメモリとより効率的に対話する機会があります。httpリクエストを送信したり、データベースからデータを読み取ったりするために、Qtは独自の強みに依存しており、OSが提供する既製のフレームワークを使用しません。例外-キーボードの操作、アプリケーションの開始ウィンドウの描画、通知の表示など。

//

import UIKit

class ViewController: UIViewController {

  @IBOutlet weak var digitTextField: UITextField!
  @IBOutlet weak var timeLabel: UILabel!

  @IBAction func startAction(_ sender: UIButton) {
    guard let text = digitTextField.text,
         let number = Int(text)
         else { return }
        prime(n: number)
    }

    func prime(n: Int) {
        let startTime = DispatchTime.now()
        let _ = PrimeSequence(upTo: n)
            .reduce(into: [], { $0.append($1) }) //     
            
        let endTime = DispatchTime.now()
        let time = (endTime.uptimeNanoseconds - startTime.uptimeNanoseconds)
        timeLabel.text = "\(time)"
    }
}

public struct PrimeSequence: Sequence {
    private let iterator: AnyIterator<Int>
    
    public init(upTo limit: Int) {
        self.iterator = AnyIterator(EratosthenesIterator(upTo: limit))
    }

    public func makeIterator() -> AnyIterator<Int> {
        return iterator
    }
}

private struct EratosthenesIterator: IteratorProtocol {
    let n: Int
    var composite: [Bool]
    var current = 2
    
    init(upTo n: Int) {
        self.n = n
        self.composite = [Bool](repeating: false, count: n + 1)
    }

    mutating func next() -> Int? {
        while current <= self.n {
            if !composite[current] {
                let prime = current
                for multiple in stride(from: current * current,
                                       through: self.n, by: current) {
                    composite[multiple] = true
                }
                current += 1
                return prime
            }
            current += 1
        }
        return nil
    }
}

Qtプログラム：
#include "eratosthenes.h"
#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <QVector>
#include <QDebug>
#include <vector>
#include <cmath>
 
Eratosthenes::Eratosthenes(QObject *parent)
{
	time = 0;
}          
 
void Eratosthenes::qtFunction(int n)
{
	clock_t start, end;
	start = clock();
 
	std::vector<int> lp = std::vector<int>(n + 1, 0);
	std::vector<int> pr;
	//      
	pr.reserve(std::sqrt(n) / 2);
 
	for (int i = 2; i <= n; ++i) {
    	        if (lp[i] == 0) {
        	        lp[i] = i;
        	        pr.emplace_back(i);
    	        }
 
    	        for (int j = 0; j < pr.size() && pr[j] <= lp[i] && i * pr[j] <= n; ++j) {
        	        lp[i * pr[j]] = pr[j];
    	        }
	}
 
	end = clock();
	time = (end - start) / (double)CLOCKS_PER_SEC;
	pTimeChanged();
 
	qDebug() << "  " << pr.size() << "" << time;
}

マイナス


もちろん、Qtではすべてがそれほどバラ色ではありません。ここでペイントしようとすると、短所があります。たとえば、AndroidでTextInputを使用すると、Qtはキーボードを操作する際に完全にユニークなスティックをホイールに配置するため、デバイスごとに松葉杖に特に敏感な完璧主義者を拷問することができます。1台の電話では画像が上になり、もう1台では静止しますが、EnterKeyは機能しません。3番目の電話では常に大文字のみが入力され、小文字に切り替えるように説得する方法はありません。あなたは無限に行くことができます。そして、これもすべて遅くなります！（うなり声はAndroidにのみ関連し、Sailfishにはそのような問題はなく、すべて正常に動作します）。さらに、Qtでは、アプリケーションのネイティブな外観を実現することは困難です。

結論


引き出すことができる主な結論：クロスプラットフォームツールであるQtは、ネイティブ開発ツールよりもパフォーマンスが劣ることはありません。これは、GUIに加えて、数学がまだたくさんあるプログラム、特に、ニュアンスが多く従業員が少ないエンタープライズアプリケーションに最適です。そのため、各OSで独立したバージョンを作成しません。機能は、ネイティブUIよりも企業ユーザーにとって重要です。Auroraにとって、Qtはアプリケーション開発のためのネイティブツールであり、おそらく他のパフォーマンスを向上させます。
