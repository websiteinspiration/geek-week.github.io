超音波GPS

超音波GPS。概念モデル


このような長い航海に出発する前に、膝の上で本当にすべてが可能かどうかを確認する価値があります。

この記事の内容：シンプルな超音波GPSを迅速かつ安価に作成する方法。

必要なデバイスのリスト


  • HC-SR04 3本
  • Arduino 1個
  • ワイヤーのコイル。

概念



図。1-デバイスの一般的なアイデア

部屋の上隅にはエミッターの役割を果たすHC-SR04が設置され、床にはレシーバーを上にして設置されています。3つ目はレシーバーの役割を果たします。

これはすべて、スキームに従って接続されています：


図。2-デバイス接続図

そしてもちろん、USB経由でコンピューターにArduinoを接続します。

すべての仕組み：

  1. レシーバーからエミッターまでの距離を測定します1
  2. 距離測定の開始に関する信号をレシーバー1とエミッター1に送信します（Trigレッグを引き寄せます）。
  3. レシーバーが長さを与えるまで待ちます。
  4. エミッタ2についても同じ手順を繰り返します。
  5. レシーバー座標を計算します。

学校の形状を思い出してください



図。3-問題の幾何学的表現

この図に基づいて、座標を計算するための式を作成します：



A、F-レシーバーに対するエミッターの高さ。
C、Eは、エミッターからレシーバーまでの距離を測定して得られる長さです。
Gはエミッタ間の距離です。

練習


2つのラジエーターを互いに適度な距離（3メートル）で天井の下に設置し、ラジエーターで作業エリアが形成される1点に向けます。

受信機を平らなもの（たとえば、ボードの一部）に取り付け、室内で不要な超音波ノイズが発生しないようにテープでテープで包みます。
上記のスキームに従ってすべてを接続します。

これは、エミッターのファスナーとレシーバーの基板のようになります


。4-モジュールのタイプ

プログラム


コードの重要な部分をさらに詳しく検討します。

エミッター1とレシーバーに、これらのデバイスのトリガー入力を低から高に、10マイクロ秒ずつ移動して低に戻し、距離の測定を強制的に開始させます。

  digitalWrite(trigPinRessiv, LOW);  digitalWrite(trigPinTransmit1, LOW);
  delayMicroseconds(5);
  digitalWrite(trigPinRessiv, HIGH);  digitalWrite(trigPinTransmit1, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPinRessiv, LOW);  digitalWrite(trigPinTransmit1, LOW);

通常、これらのデバイス自体が超音波信号を発信し、それが何かから反射されて戻ってくるまで待機します。しかし、私たちはそれらを欺いています。私たちの場合、一方は送信し、もう一方は受信します。受信したものはそれを彼の信号であると考えますが、これは別のデバイスからの信号であり、この他のデバイスまでの距離を示します。

センサーが超音波信号の飛行時間を通知し始めるまで待機します。

while (digitalRead(echoPinRessiv) == LOW);

信号の開始時刻を記録します。

timeStartSignal = micros();

センサーが超音波信号の飛行時間の通知を停止するまで待機します。

while (digitalRead(echoPinRessiv) == HIGH);

終了時間を書きます：

  timeEndSignal = micros();

単純な式を使用して、エミッターからレシーバーまでの距離を計算します。

  lenC = ((timeEndSignal-timeStartSignal)/58.00)*10.00;

部屋で超音波ノイズが収まるまで待ちます。

delay(100);

センサーが、測定された距離に正比例する期間、エコー出力をLowに下げることによって距離を教えてくれることは注目に値します。

2番目のエミッタについても同じことが繰り返されます。

直角三角形のルールを使用して、結果の距離を床面に投影します（図3）。

3次元座標から平面への遷移のソフトウェア式を実装します。式は上に示されています。

lenB = sqrt((lenC*2.00)*(lenC*2.00) - lenA*lenA);

残念ながら、エラーがあり、それらを削除するために私はそのような経験豊富な数式を思い付き、それを削除して何が得られるかを確認しました。

  measurementError = 26.437 - 0.08*lenC/10;
  lenB = (lenB + measurementError*10)/10.00;

センサー2についても同じことを繰り返し

ます。平面上の座標を計算します。

角度Alphaを見つけます。

  alfa = acos(((lenG*lenG + lenD*lenD - lenB*lenB)*1.00) / ((2*lenE*lenG)*1.00));

座標自体を見つけます。 

  koord_X = lenE*cos(1.57-alfa);
  koord_Y = lenE*cos(alfa);

座標の値が可能な範囲を超えている場合は、前の値に置き換えます。

  if((koord_X > 0) && (koord_X < 500) && (koord_Y > 0) && (koord_Y < 500))
  {
    oldKoord_X = koord_X;
    oldKoord_Y = koord_Y;
  }else{  
    koord_X = oldKoord_X;
    koord_Y = oldKoord_Y;
  }

6つの座標値のバッファーを作成し、常にシフトします：

  koord_X5 = koord_X4;
  koord_X4 = koord_X3;
  koord_X3 = koord_X2;
  koord_X2 = koord_X1;
  koord_X1 = koord_X;
  
  koord_Y5 = koord_Y4;
  koord_Y4 = koord_Y3;
  koord_Y3 = koord_Y2;
  koord_Y2 = koord_Y1;
  koord_Y1 = koord_Y;

過去6回の測定の平均値を取得します。

  averageKoord_X = (koord_X + koord_X1 + koord_X2 + koord_X3 + koord_X4 + koord_X5)/6;
  averageKoord_Y = (koord_Y + koord_Y1 + koord_Y2 + koord_Y3 + koord_Y4 + koord_Y5)/6;

座標をPCに送信します。

  Serial.println(averageKoord_X);
  Serial.println(averageKoord_Y);

機能：

float asin(float c)
float acos(float c)

コード全体：

int trigPinRessiv = 8;      
int echoPinRessiv = 9;      
int trigPinTransmit1 = 2;   
int trigPinTransmit2 = 3;  

int i;
long lenA = 2700;    //sensor height in mm
long lenG = 305;    // distance between sensors in cm

long koord_X,  koord_Y;
long koord_X1, koord_Y1;
long koord_X2, koord_Y2;
long koord_X3, koord_Y3;
long koord_X4, koord_Y4;
long koord_X5, koord_Y5;

long oldKoord_X = 0, oldKoord_Y = 0;
long averageKoord_X, averageKoord_Y;
long measurementError;
float alfa;
long timeStartSignal, timeEndSignal;
long lenC, lenE, lenB, lenD;

void setup() {
  Serial.begin (115200);
  pinMode(trigPinRessiv, OUTPUT);
  pinMode(echoPinRessiv, INPUT);
  pinMode(trigPinTransmit1, OUTPUT);
  pinMode(trigPinTransmit2, OUTPUT);
}
void loop()
{
  averageKoord_X = 0;
  averageKoord_Y = 0;
  
  digitalWrite(trigPinRessiv, LOW);  digitalWrite(trigPinTransmit1, LOW);
  delayMicroseconds(5);
  digitalWrite(trigPinRessiv, HIGH);  digitalWrite(trigPinTransmit1, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPinRessiv, LOW);  digitalWrite(trigPinTransmit1, LOW);
  
  while (digitalRead(echoPinRessiv) == LOW);   timeStartSignal = micros();
  while (digitalRead(echoPinRessiv) == HIGH);  timeEndSignal = micros();
  
  lenC = ((timeEndSignal-timeStartSignal)/58.00)*10.00;
  delay(100);

  digitalWrite(trigPinRessiv, LOW);  digitalWrite(trigPinTransmit2, LOW);
  delayMicroseconds(5);
  digitalWrite(trigPinRessiv, HIGH);  digitalWrite(trigPinTransmit2, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPinRessiv, LOW);  digitalWrite(trigPinTransmit2, LOW);
  
  while (digitalRead(echoPinRessiv) == LOW);  timeStartSignal = micros();
  while (digitalRead(echoPinRessiv) == HIGH);  timeEndSignal = micros();
  
  lenE = ((timeEndSignal-timeStartSignal)/58.00)*10.00;
  delay(100);


  lenB = sqrt((lenC*2.00)*(lenC*2.00) - lenA*lenA);
  measurementError = 26.437 - 0.08*lenC/10;
  lenB = (lenB + measurementError*10)/10.00;

  lenD = sqrt((lenE*2.00)*(lenE*2.00) - lenA*lenA);
  measurementError = 26.437 - 0.08*lenD/10;
  lenD = (lenD + measurementError*10)/10.00;
  
  alfa = acos(((lenG*lenG + lenD*lenD - lenB*lenB)*1.00)/((2*lenE*lenG)*1.00));
  
  koord_X = lenE*cos(1.57-alfa);
  koord_Y = lenE*cos(alfa);
  
  if((koord_X > 0) && (koord_X < 500) && (koord_Y > 0) && (koord_Y < 500))
  {
    oldKoord_X = koord_X;
    oldKoord_Y = koord_Y;
  }else{  
    koord_X = oldKoord_X;
    koord_Y = oldKoord_Y;
  }
  
  koord_X5 = koord_X4;
  koord_X4 = koord_X3;
  koord_X3 = koord_X2;
  koord_X2 = koord_X1;
  koord_X1 = koord_X;
  
  koord_Y5 = koord_Y4;
  koord_Y4 = koord_Y3;
  koord_Y3 = koord_Y2;
  koord_Y2 = koord_Y1;
  koord_Y1 = koord_Y;
  
  averageKoord_X = (koord_X + koord_X1 + koord_X2 + koord_X3 + koord_X4 + koord_X5)/6;
  averageKoord_Y = (koord_Y + koord_Y1 + koord_Y2 + koord_Y3 + koord_Y4 + koord_Y5)/6;
  
   
}

float asin(float c)
{
  float out;
  out = ((c+(pow(c,3))/6+(3*pow(c,5))/40+(5*pow(c,7))/112 +(35*pow(c,9))/1152 +(0.022*pow(c,11))+(0.0173*pow(c,13))+(0.0139*pow(c,15)) + (0.0115*pow(c,17))+(0.01*pow(c,19))));
  
  if(c >= .96 && c < .97)   {    out=1.287+(3.82*(c-.96));    }
  if(c>=.97 && c<.98)       {    out=(1.325+4.5*(c-.97));     }
  if(c>=.98 && c<.99)       {    out=(1.37+6*(c-.98));        }
  if(c>=.99 && c<=1)        {    out=(1.43+14*(c-.99));       }
  
  return out;
}



float acos(float c)
{
  float out;
  out=asin(sqrt(1-c*c));
  return out;
}

float atan(float c)
{
  float out;
  out=asin(c/(sqrt(1+c*c)));
  return out;
}
