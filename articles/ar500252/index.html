Kaspersky Security Center - مستوى جديد

في هبر ، كانت هناك بالفعل بعض الملاحظات حول أتمتة Kaspersky Security Center (KSC)


هنا و هنا


أقترح اليوم الغوص أكثر عمقًا والانتقال إلى مستوى جديد. استخدم: KSC Open API لـ Kaspersky Security Center 10+.


لمراقبة الحالة ، وتوزيع محطات العمل الجديدة حسب المجموعات ، كان من الضروري استخدام وحدة تحكم الإدارة لإنشاء التقارير ؛ ولديها وظائف واسعة النطاق ، ولكن سرعة تشغيلها تترك الكثير مما هو مرغوب فيه.


وكان ظهور الدعم لـ Open API بمثابة نسمة من الهواء النقي ، وعلى الفور تقرر استخدام هذا الحل ، وكالعادة ، تمت كتابة مكتبة بلغة البرمجة المفضلة لـ Go التي سمحت بالوصول إلى خادم KSC باستخدام واجهة برمجة التطبيقات التي وفرتها.


لاستخدام الحزمة في مشروعنا ، كالعادة ، نحتاج للحصول عليها:


go get -u github.com/pixfid/go-ksc/kaspersky


قم باستيراد الحزمة إلى المشروع الذي تم إنشاؤه.


import (
    "github.com/pixfid/go-ksc/kaspersky"
)

func GetAllGroups(ctx context.Context, client *kaspersky.Client) *FullGroupsInfos {
    groups := &FullGroupsInfos{} //    .

    groupParam := kaspersky.HGParams{
        WstrFilter: `
        (&
            (!"KLGRP_CHLDHST_CNT" = 0)
            (!"grp_from_unassigned" = true)
        )`,
        VecFieldsToReturn: []string{
            "id",
            "name",
            "grp_full_name",
            "creationDate",
            "KLGRP_CHLDHST_CNT",
            "KLSRV_HSTSTAT_CRITICAL",
            "KLSRV_HSTSTAT_WARNING",
        },
        PParams: kaspersky.PParams{
            KlsrvhSlaveRecDepth:    0,
            KlgrpFindFromCurVsOnly: true,
        },
        LMaxLifeTime: 100,
    }

    accessor, _, _ := client.HostGroup.FindGroups(ctx, groupParam) //  
    count, _, _ := client.ChunkAccessor.GetItemsCount(ctx, accessor.StrAccessor) //     

    _, _ = client.ChunkAccessor.GetItemsChunk(ctx, kaspersky.ItemsChunkParams{
        StrAccessor: accessor.StrAccessor,
        NStart:      0,
        NCount:      count.Int,
    } , groups) //    
    client.ChunkAccessor.Release(ctx, accessor.StrAccessor) // 

    return groups
}
