);

:


insert into task (id, period, next_run_time) values ($1, $2, $3);

:


with next_task as (
    select id from task
    where status = 0
      and next_run_time <= localtimestamp
    limit 1
    for update skip locked
)

set
    status = 1
from next_task
where task.id = next_task.id
returning task.id;

    status = 0,
