class _Counter {
  int _count;

  int get count => _count;

  _Counter(this._count)
      : this.onCounterUpd = BehaviorSubject<int>.seeded(_count);

  ///  .
  final BehaviorSubject<int> onCounterUpd;

  ///     ,   .
  Future incrementCounter() async {
    onCounterUpd.add(++_count);
  }
}
