           ARRAY [fs_user_990823086.name] AS fpath

    WHERE fs_user_990823086.parent_id IS NULL
    UNION ALL
    SELECT si.id,
           si.name,
           si.parent_id,
           si.owner,
           sp.fpath || si.name AS fpath
    FROM fs_user_990823086 si
             JOIN tree sp ON si.parent_id = sp.id
)
