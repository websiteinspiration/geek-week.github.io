    let handles:Vec<_> = (0..10).map(|_|{
        thread::spawn(||{
            let mut x = 0;
            for _ in 0..5_000_000 {
                x += 1

                 h.join().map_err(|_| "Could not join thread!\n").unwrap());
    }
    println!("Done!");
